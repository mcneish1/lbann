439  inline int get_adaptive_quantization_threads(El::Int width) {
440  int num_threads = omp_get_max_threads();
441  if (width <= 64) {
442  num_threads = 2;
443  } else if (width <= 128) {
444  num_threads = 8;
445  } else if (width <= 256) {
446  num_threads = 12;
447  } else if (width <= 1024) {
448  num_threads = 24;
449  }
450  return std::min(omp_get_max_threads(), num_threads);
451  }

462  inline int get_adaptive_quantization_copy_threads(El::Int width) {
