335  void fp_compute_im2col() {
336 
337  // Throw exception if pooling mode is not max or average pooling
338  if(m_pool_mode != pool_mode::max
339  && m_pool_mode != pool_mode::average) {
340  throw lbann_exception("pooling_layer: CPU pooling layer only implements max and average pooling");
341  }
342 
343  // Get local matrices
344  const Mat& prev_activations_local = this->m_prev_activations_v->LockedMatrix();
345  Mat& activations_local = this->m_activations_v->Matrix();
346 
347  // Get parameters
348  const int local_width = prev_activations_local.Width();
349  const int num_channels = this->m_prev_neuron_dims[0];
350  const int num_per_output_channel = this->m_num_neurons / num_channels;
351 
352  // Initialize max pool indices if needed
353  if(m_pool_mode == pool_mode::max) {
354  m_max_pool_indices.assign(this->m_num_neurons * local_width, 0);
355  }
356 
357  // Initialize matrices
358  Mat im2col_mat(m_pool_size * num_channels, num_per_output_channel);
359  Mat input_mat;
360 
361  // Iterate through data samples
362  for(int sample = 0; sample < local_width; ++sample) {
363 
364  // Construct im2col matrix from input
365  El::LockedView(input_mat, prev_activations_local,
366  El::ALL, El::IR(sample));
367  im2col(input_mat,
368  im2col_mat,
369  num_channels,
370  this->m_num_prev_neuron_dims - 1,
371  &this->m_prev_neuron_dims[1],
372  m_pool_pads.data(),
373  m_pool_dims.data(),
374  m_pool_strides.data());
375 
376  if(m_pool_mode == pool_mode::max) {
377  // Apply max pooling
378  DataType *activations_buffer = activations_local.Buffer(0, sample);
379  int *indices_buffer = &m_max_pool_indices[sample * this->m_num_neurons];
380  #pragma omp parallel for
381  for(int channel = 0; channel < num_channels; ++channel) {
382  for(int j = 0; j < num_per_output_channel; ++j) {
383  DataType *im2col_buffer = im2col_mat.Buffer(channel*m_pool_size, j);
384  DataType max_entry = -INFINITY;
385  int max_index = 0;
386  for(int i = 0; i < m_pool_size; ++i) {
387  const DataType current_entry = im2col_buffer[i];
388  if(current_entry > max_entry) {
389  max_entry = current_entry;
390  max_index = i;
391  }
392  }
393  const int output_index = j + channel * num_per_output_channel;
394  activations_buffer[output_index] = max_entry;
395  indices_buffer[output_index] = max_index;
396  }
397  }
398  }
399 
400  if(m_pool_mode == pool_mode::average) {
401  // Apply average pooling
402  DataType *activations_buffer = activations_local.Buffer(0, sample);
403  #pragma omp parallel for
404  for(int channel = 0; channel < num_channels; ++channel) {
405  for(int j = 0; j < num_per_output_channel; ++j) {
406  const DataType *im2col_buffer
407  = im2col_mat.LockedBuffer(channel*m_pool_size, j);
408  DataType output_entry = 0;
409  for(int i = 0; i < m_pool_size; ++i) {
410  output_entry += im2col_buffer[i];
411  }
412  output_entry /= m_pool_size;
413  const int output_index = j + channel * num_per_output_channel;
414  activations_buffer[output_index] = output_entry;
415  }
416  }
417  }
418 
419  }

424  void bp_compute_im2col() {
425 
426  // Throw exception if pooling mode is not max or average pooling
427  if(m_pool_mode != pool_mode::max
428  && m_pool_mode != pool_mode::average) {
429  throw lbann_exception("pooling_layer: CPU pooling layer only implements max and average pooling");
430  }
431 
432  // Get local matrices
433  const Mat& prev_error_signal_local = this->m_prev_error_signal_v->LockedMatrix();
434  Mat& error_signal_local = this->m_error_signal_v->Matrix();
435 
436  // Get parameters
437  const int local_width = prev_error_signal_local.Width();
438  const int num_channels = this->m_prev_neuron_dims[0];
439  const int num_per_input_channel = this->m_num_neurons / num_channels;
440 
441  // Initialize matrices
442  Mat im2col_mat(m_pool_size * num_channels, num_per_input_channel);
443  Mat output_mat;
444 
445  // Iterate through data samples
446  for(int sample = 0; sample < local_width; ++sample) {
447 
448  // Compute gradient w.r.t. im2col matrix for max pooling
449  if(m_pool_mode == pool_mode::max) {
450 
451  // Clear im2col matrix
452  El::Zero(im2col_mat);
453 
454  // Copy previous error signal to im2col matrix entries
455  // corresponding to max
456  const DataType *prev_error_signal_buffer
457  = prev_error_signal_local.LockedBuffer(0, sample);
458  const int *indices_buffer
459  = &m_max_pool_indices[sample * this->m_num_neurons];
460  #pragma omp parallel for
461  for(int channel = 0; channel < num_channels; ++channel) {
462  for(int j = 0; j < num_per_input_channel; ++j) {
463  const int input_index = j + channel * num_per_input_channel;
464  const int max_index = indices_buffer[input_index];
465  DataType *im2col_buffer = im2col_mat.Buffer(channel*m_pool_size, j);
466  im2col_buffer[max_index]
467  = prev_error_signal_buffer[input_index];
468  }
469  }
470 
471  }
472 
473  // Compute gradient w.r.t. im2col matrix for average pooling
474  if(m_pool_mode == pool_mode::average) {
475  const DataType *prev_error_signal_buffer
476  = prev_error_signal_local.LockedBuffer(0, sample);
477  #pragma omp parallel for
478  for(int channel = 0; channel < num_channels; ++channel) {
479  for(int j = 0; j < num_per_input_channel; ++j) {
480  DataType *im2col_buffer = im2col_mat.Buffer(channel*m_pool_size, j);
481  const int input_index = j + channel * num_per_input_channel;
482  const DataType output_entry
483  = prev_error_signal_buffer[input_index] / m_pool_size;
484  for(int i = 0; i < m_pool_size; ++i) {
485  im2col_buffer[i] = output_entry;
486  }
487  }
488  }
489 
490  }
491 
492  // Compute error signal (i.e. gradient w.r.t. input)
493  El::View(output_mat, error_signal_local, El::ALL, El::IR(sample));
494  col2im(im2col_mat,
495  output_mat,
496  num_channels,
497  this->m_num_prev_neuron_dims - 1,
498  &this->m_prev_neuron_dims[1],
499  m_pool_pads.data(),
500  m_pool_dims.data(),
501  m_pool_strides.data());
502 
503  }
504 
505  }
