52 void data_reader_merge_samples::load() {
53  // Load each subsidiary data reader.
54  for (auto&& reader : m_data_readers) {
55  reader->load();
56  }
57  // Compute the total number of samples and do some sanity checks.
58  int num_samples = 0;
59  int num_labels = m_data_readers[0]->get_num_labels();
60  int data_size = m_data_readers[0]->get_linearized_data_size();
61  int label_size = m_data_readers[0]->get_linearized_label_size();
62  const std::vector<int> data_dims = m_data_readers[0]->get_data_dims();
63  // Prepend a 0 to make things easier.
64  m_num_samples_psum.push_back(0);
65  for (auto&& reader : m_data_readers) {
66  m_num_samples_psum.push_back(reader->get_num_data());
67  num_samples += reader->get_num_data();
68  // TODO: It would be good to relax this.
69  // Large data sets may not have every label in every file.
70  if (num_labels != reader->get_num_labels()) {
71  throw lbann_exception(
72  "data_reader_merge_samples: data readers do not have the same number of labels");
73  }
74  if (data_size != reader->get_linearized_data_size()) {
75  throw lbann_exception(
76  "data_reader_merge_samples: data readers do not have the same data size");
77  }
78  if (label_size != reader->get_linearized_label_size()) {
79  throw lbann_exception(
80  "data_reader_merge_samples: data readers do not have the same label size");
81  }
82  if (data_dims != reader->get_data_dims()) {
83  throw lbann_exception(
84  "data_reader_merge_samples: data readers do not have the same data dims");
85  }
86  }
87  std::partial_sum(m_num_samples_psum.begin(), m_num_samples_psum.end(),
88  m_num_samples_psum.begin());
89  // Set up our indices.
90  // Note each subsidiary data reader presumably shuffled its indices as well.
91  // That's not strictly necessary, but does not impact anything.
92  m_shuffled_indices.resize(num_samples);
93  std::iota(m_shuffled_indices.begin(), m_shuffled_indices.end(), 0);
94  select_subset_of_data();
95 }

97 bool data_reader_merge_samples::fetch_datum(Mat& X, int data_id, int mb_idx,
98  int tid) {
99  // Find the right data reader to delegate to.
100  for (size_t i = 0; i < m_data_readers.size(); ++i) {
101  if (data_id < m_num_samples_psum[i + 1]) {
102  data_id -= m_num_samples_psum[i];
103  return m_data_readers[i]->fetch_datum(X, data_id, mb_idx, tid);
104  }
105  }
106  throw lbann_exception(
107  "data_reader_merge_samples: do not have data ID " +
108  std::to_string(data_id));
109 }
