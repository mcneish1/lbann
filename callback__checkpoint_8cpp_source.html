50 bool lbann_callback_checkpoint::need_checkpoint(model *m) {
51  /* TODO: since we're using clocks, this requires a bcast for each call,
52  * we could use number of samples processed to make a local decision */
53  // if none of our checkpoint conditions are set, assume we're not checkpointing
54  if (m_checkpoint_epochs == 0 &&
55  m_checkpoint_steps == 0 &&
56  m_checkpoint_secs == 0.0) {
57  return false;
58  }
59  // assume that we won't checkpoint
60  bool checkpoint_now = false;
61  lbann_comm *comm = m->get_comm();
62  // if at start of epoch and evenly divide
63  if (!checkpoint_now && m_checkpoint_epochs > 0) {
64  if (m_epoch_end) {
65  checkpoint_now = (m->get_cur_epoch() > 0) && (m->get_cur_epoch() % m_checkpoint_epochs == 0);
66  }
67  }
68  // if our current step is evenly divisable by checkpoint steps,
69  // take a checkpoint
70  if (!checkpoint_now && m_checkpoint_steps > 0) {
71  checkpoint_now = (m->get_cur_step() > 0) && (m->get_cur_step() % m_checkpoint_steps == 0);
72  }
73  // check the clock if time-based checkpoint is enabled
74  if (!checkpoint_now && m_checkpoint_secs != 0.0) {
75  // have rank 0 determine whether we should checkpoint
76  // to avoid issues with clock skew, we rely on rank 0 to make decision
77  if (comm->am_world_master()) {
78  // get the current time
79  EvalType current = MPI_Wtime();
80  // compute time next checkpoint is due
81  EvalType next = m_checkpoint_last + m_checkpoint_secs;
82  // determine whether it's time for a checkpoint
83  checkpoint_now = (current >= next);
84  }
85  // get checkpoint_now from rank 0
86  int flag = checkpoint_now ? 1 : 0;
87  MPI_Bcast(&flag, 1, MPI_INT, 0, MPI_COMM_WORLD);
88  checkpoint_now = (bool) flag;
89  }
90  m_epoch_end = false;
91  return checkpoint_now;
92 }

110 static bool read_latest(const char *dir, const char *name, int *epochLast, int *trainLast) {
111  // assume we don't have a file, we'll return -1 in that case
112  *epochLast = -1;
113  *trainLast = -1;
114  // define filename
115  char filename[1024];
116  sprintf(filename, "%s/%s", dir, name);
117  // open the file for reading
118  int fd = openread(filename);
119  if (fd != -1) {
120  // read epoch from file
121  char field[256];
122  read_string(fd, "shared.last", field, sizeof(field));
123  int ret = sscanf(field, "epoch=%d step=%d\n", epochLast, trainLast);
124  // close our file
125  closeread(fd, filename);
126  if(ret != 2) { return false; }
127  }
128  return true;

136 bool lbann_callback_checkpoint::checkpointShared(model *m) {
137  // if the checkpoint directory is not defined, bail
138  if (m_checkpoint_dir.length() == 0) {
139  return false;
140  }
141  // time how long this takes
142  El::Timer timer;
143  lbann_comm *comm = m->get_comm();
144  // get checkpoint directory
145  const char *dir = m_checkpoint_dir.c_str();
146  // read current epoch and step counters from model
147  int epoch = m->get_cur_epoch();
148  int step = m->get_cur_step();
149  // let user know we're saving a checkpoint
150  MPI_Barrier(MPI_COMM_WORLD);
151  if (comm->am_world_master()) {
152  timer.Start();
153  printf("Checkpoint: epoch %d step %d ...\n", epoch, step);
154  fflush(stdout);
155  }
156  // create top level directory
157  //const char* dir = trainParams.ParameterDir.c_str();
158  makedir(dir);
159  // create subdirectory for this epoch
160  char epochdir[1024];
161  snprintf(epochdir, sizeof(epochdir), "%s/shared.epoch.%d.step.%d", dir, epoch, step);
162  // start our checkpoint
163  persist p;
164  p.open_checkpoint(epochdir,m_checkpoint_per_rank);
165  // call virtual function to checkpoint model state
166  m->save_to_checkpoint_shared(p);
167  // close our checkpoint
168  p.close_checkpoint();
169  uint64_t bytes_count = p.get_bytes();
170  // write epoch number to current file, we do this at the end so as to only update
171  // this file when we know we have a new valid checkpoint
172  if (comm->am_world_master()) {
173  write_latest(dir, "last.shared.checkpoint", epoch, step);
174  }
175  // stop timer and report cost
176  MPI_Barrier(MPI_COMM_WORLD);
177  if (comm->am_world_master()) {
178  EvalType secs = timer.Stop();
179  EvalType bw = 0;
180  if (secs > 0.0) {
181  bw = EvalType(bytes_count) / (secs * 1024.0 * 1024.0);
182  }
183  printf("Checkpoint complete: Epoch=%d Step=%d (%f secs, %llu bytes, %f MB/sec)\n",
184  epoch, step, secs, (unsigned long long) bytes_count, bw
185  );
186  fflush(stdout);
187  }
188  // saved a checkpoint, update our last checkpoint time
189  m_checkpoint_last = MPI_Wtime();
190  return true;
191 }
192 bool lbann_callback_checkpoint::restartShared(model *m) {
193  // if the checkpoint directory is not defined, bail
194  if (m_checkpoint_dir.length() == 0) {
195  return false;
196  }
197  // get top level directory
198  const char *dir = m_checkpoint_dir.c_str();
199  // read epoch number from current file
200  int epoch, step;
201  lbann_comm *comm = m->get_comm();
202  if (comm->am_world_master()) {
203  read_latest(dir, "last.shared.checkpoint", &epoch, &step);
204  }
205  MPI_Bcast(&epoch, 1, MPI_INT, 0, MPI_COMM_WORLD);
206  MPI_Bcast(&step, 1, MPI_INT, 0, MPI_COMM_WORLD);
207  // if we couldn't find the latest epoch, just return
208  if (epoch < 0) {
209  return false;
210  }
211  // time how long this takes
212  El::Timer timer;
213  // let user know we're restarting from a checkpoint
214  MPI_Barrier(MPI_COMM_WORLD);
215  if (comm->am_world_master()) {
216  timer.Start();
217  printf("Restart: epoch %d ...\n", epoch);
218  fflush(stdout);
219  }
220  // get subdirectory for this epoch
221  char epochdir[1024];
222  sprintf(epochdir, "%s/shared.epoch.%d.step.%d", dir, epoch, step);
223  // open our checkpoint
224  persist p;
225  p.open_restart(epochdir,m_checkpoint_per_rank);
226  // call virtual function to restore model from checkpoint
227  m->load_from_checkpoint_shared(p);
228  // close our checkpoint
229  p.close_restart();
230  uint64_t bytes_count = p.get_bytes();
231  // let user know we've completed reading our restart
232  MPI_Barrier(MPI_COMM_WORLD);
233  if (comm->am_world_master()) {
234  EvalType secs = timer.Stop();
235  EvalType bw = 0.0;
236  if (secs > 0.0) {
237  bw = EvalType(bytes_count) / (secs * 1024.0 * 1024.0);
238  }
239  printf("Restart complete: Epoch=%d Step=%d (%f secs, %llu bytes, %f MB/sec)\n",
240  epoch, step, secs, (unsigned long long) bytes_count, bw
241  );
242  fflush(stdout);
243  }
244  return true;
245 }
