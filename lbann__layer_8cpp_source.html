99 void lbann::Layer::forwardProp() {
100  double fp_start = get_time();
101 
102  // Get incoming activations and convert matrix distribution if necessary
103  // Note that on assignment Elemental handles distribution conversion so a DistMatrixReadProxy is unnecessary
104  if(fp_input != NULL) { // Input layers will not have a valid fp_input
105  *m_prev_activations = *fp_input;
106  }
107  // Set the view for all of the standard matrices based on the
108  // current mini-batch size
109  fp_set_std_matrix_view();
110  // Apply connection regularization. (e.g. DropConnect).
111  for (regularizer* reg : regularizers) reg->fp_connections();
112  // Layer layer's linearity.
113  fp_linearity();
114  // Apply weight regularization (e.g. L2 normalization).
115  for (regularizer* reg : regularizers) reg->fp_weights();
116  // Apply activation function/nonlinearity.
117  fp_nonlinearity();
118  // Apply activation regularization (e.g. Dropout).
119  for (regularizer* reg : regularizers) reg->fp_activations();
120  fp_time += get_time() - fp_start;
121  return;
122 }

124 void lbann::Layer::backProp() {
125  double bp_start = get_time();
126 
127  // Get incoming loss and convert matrix distribution if necessary
128  // Note that on assignment Elemental handles distribution conversion so a DistMatrixReadProxy is unnecessary
129  if(bp_input != NULL) { // Target layers will not have a valid bp_input
130  *m_prev_error_signal = *bp_input;
131  }
132  // Set the view for all of the standard matrices based on the
133  // current mini-batch size
134  // bp_set_std_matrix_view();
135  // Backprop activation regularization.
136  for (regularizer* reg : regularizers) reg->bp_activations();
137  // Backprop the activation function/nonlinearity.
138  bp_nonlinearity();
139  // Backprop weight regularization.
140  for (regularizer* reg : regularizers) reg->bp_weights();
141  // Backprop the layer's linearity.
142  bp_linearity();
143  // Backprop connection regularization.
144  for (regularizer* reg : regularizers) reg->bp_connections();
145  bp_time += get_time() - bp_start;
146 }

148 void lbann::Layer::summarize(lbann_summary& summarizer, int64_t step) {
149  std::string prefix = "layer" + std::to_string(static_cast<long long>(Index)) + "/weights/";
150  // TODO: implement summarizer functions for other matrix distributions
151  const ElMat& wb = get_weights_biases();
152  summarizer.reduce_mean(prefix + "mean", wb, step);
153  summarizer.reduce_min(prefix + "min", wb, step);
154  summarizer.reduce_max(prefix + "max", wb, step);
155  summarizer.reduce_stdev(prefix + "stdev", wb, step);
156  prefix = "layer" + std::to_string(static_cast<long long>(Index)) + "/weights_gradient/";
157  const ElMat& wb_d = get_weights_biases_gradient();
158  summarizer.reduce_mean(prefix + "mean", wb_d, step);
159  summarizer.reduce_min(prefix + "min", wb_d, step);
160  summarizer.reduce_max(prefix + "max", wb_d, step);
161  summarizer.reduce_stdev(prefix + "stdev", wb_d, step);
162  prefix = "layer" + std::to_string(static_cast<long long>(Index)) + "/activations/";
163  const ElMat& acts = get_activations();
164  summarizer.reduce_mean(prefix + "mean", acts, step);
165  summarizer.reduce_min(prefix + "min", acts, step);
166  summarizer.reduce_max(prefix + "max", acts, step);
167  summarizer.reduce_stdev(prefix + "stdev", acts, step);
168  prefix = "layer" + std::to_string(static_cast<long long>(Index)) + "/";
169  summarizer.reduce_scalar(prefix + "fp_time", fp_time, step);
170  summarizer.reduce_scalar(prefix + "bp_time", bp_time, step);
171  reset_counters();
172 }

283 void lbann::Layer::fp_set_std_matrix_view() {
284  Int cur_mini_batch_size = neural_network_model->get_current_mini_batch_size();
285 
286  // Input layers will not have a valid fp_input
287  if(fp_input != NULL) {
288  View(*m_prev_activations_v, *m_prev_activations, ALL, IR(0, cur_mini_batch_size));
289  }
290  // Target layers will not have a valid bp_input
291  if(bp_input != NULL) {
292  View(*m_prev_error_signal_v, *m_prev_error_signal, ALL, IR(0, cur_mini_batch_size));
293  }
294  View(*m_weighted_sum_v, *m_weighted_sum, ALL, IR(0, cur_mini_batch_size));
295  View(*m_error_signal_v, *m_error_signal, ALL, IR(0, cur_mini_batch_size));
296  View(*m_activations_v, *m_activations, ALL, IR(0, cur_mini_batch_size));
297 
298  // Update the layer's effective mini-batch size so it averages properly.
299  if(cur_mini_batch_size != m_mini_batch_size) {
300  // When the current mini-batch is partial, check with the other
301  // models to figure out the entire size of the complete mini-batch
302  Int total_mini_batch_size = comm->intermodel_allreduce((Int) cur_mini_batch_size);
303  set_effective_minibatch_size(total_mini_batch_size);
304  }
305  else {
306  set_effective_minibatch_size(cur_mini_batch_size * comm->get_num_models());
307  }
308 }
