143  void setup_dims() override {
144 
145  // Initialize previous neuron tensor dimensions
146  base_convolution_layer::setup_dims();
147 
148  // Initialize convolution kernel dimensions
149  this->m_kernel_dims.insert(this->m_kernel_dims.begin(),
150  this->m_prev_neuron_dims[0]);
151 
152  // Check if previous neuron tensor dimensions are valid
153  #ifdef LBANN_DEBUG
154  if(this->m_num_neuron_dims != (int) this->m_kernel_dims.size() - 1) {
155  throw lbann_exception("deconvolution_layer: previous neuron tensor dimensions are unexpected");
156  }
157  #endif
158 
159  // Initialize neuron tensor dimensions
160  this->m_neuron_dims[0] = this->m_kernel_dims[1];
161  for(int i=0; i<this->m_num_neuron_dims-1; ++i) {
162  this->m_neuron_dims[i+1]
163  = ((this->m_prev_neuron_dims[i+1]-1) * this->m_conv_strides[i]
164  + this->m_kernel_dims[i+2] - 2*this->m_conv_pads[i]);
165  }
166  this->m_num_neurons = std::accumulate(this->m_neuron_dims.begin(),
167  this->m_neuron_dims.end(),
168  1,
169  std::multiplies<int>());
170 
171  // Get size of convolutional kernel
172  this->m_kernel_size = std::accumulate(m_kernel_dims.begin(),
173  m_kernel_dims.end(),
174  1,
175  std::multiplies<int>());
176 
177  }
