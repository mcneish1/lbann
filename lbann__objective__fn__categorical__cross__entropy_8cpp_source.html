50 double lbann::objective_functions::categorical_cross_entropy::compute_categorical_cross_entropy(ElMat &predictions_v,
51  ElMat &groundtruth_v) {
52 
53  // Compute categorical cross entropy on current process
54  double total_error = 0;
55  for(Int c = 0; c < groundtruth_v.LocalWidth(); c++) {
56  for(Int r = 0; r < groundtruth_v.LocalHeight(); r++) {
57  const DataType true_val = groundtruth_v.GetLocal(r,c);
58  if(true_val != DataType(0)) {
59  double pred_val = predictions_v.GetLocal(r,c);
60  total_error += - true_val * Log(pred_val);
61  }
62  }
63  }
64 
65  // Get categorical cross entropy by summing results from all processes
66  total_error = mpi::AllReduce(total_error, groundtruth_v.DistComm());
67  return total_error;
68 
69 }

74 
75  double total_error = compute_categorical_cross_entropy(predictions_v, groundtruth_v);
