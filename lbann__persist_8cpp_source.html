69 bool lbann::writeDist(int fd, const char* filename, const DistMat& M, uint64_t* bytes)
70 {
71  // TODO: store in network order
72 
73  // build our header
74  struct layer_header header;
75  header.rank = (uint64_t) M.Grid().Rank();
76  header.width = (uint64_t) M.Width();
77  header.height = (uint64_t) M.Height();
78  header.localwidth = (uint64_t) M.LocalWidth();
79  header.localheight = (uint64_t) M.LocalHeight();
80  header.ldim = (uint64_t) M.LDim();
81 
82  // write the header to the file
83  ssize_t write_rc = write(fd, &header, sizeof(header));
84  if (write_rc != sizeof(header)) {
85  // error!
86  }
87  *bytes += write_rc;
88 
89  // now write the data for our part of the distributed matrix
90  const Int localHeight = M.LocalHeight();
91  const Int localWidth = M.LocalWidth();
92  const Int lDim = M.LDim();
93  if(localHeight == lDim) {
94  // the local dimension in memory matches the local height,
95  // so we can write our data in a single shot
96  void* buf = (void*) M.LockedBuffer();
97  size_t bufsize = localHeight * localWidth * sizeof(DataType);
98  write_rc = write(fd, buf, bufsize);
99  if (write_rc != bufsize) {
100  // error!
101  }
102  *bytes += write_rc;
103  } else {
104  // TODO: if this padding is small, may not be a big deal to write it out anyway
105  // we've got some padding along the first dimension
106  // while storing the matrix in memory, avoid writing the padding
107  for(Int j = 0; j < localWidth; ++j) {
108  void* buf = (void*) M.LockedBuffer(0, j);
109  size_t bufsize = localHeight * sizeof(DataType);
110  write_rc = write(fd, buf, bufsize);
111  if (write_rc != bufsize) {
112  // error!
113  }
114  *bytes += write_rc;
115  }
116  }
117 
118  return true;
119 }

198 static void create_types(const El::DistMatrix<DataType> &M, MPI_Datatype* mattype, MPI_Datatype* viewtype)
199 {
200  // TODO: we could cache these datatypes on Matrix object
201 
202  // initialize return params to known values
203  *mattype = MPI_DATATYPE_NULL;
204  *viewtype = MPI_DATATYPE_NULL;
205 
206  // TODO: use TypeMap<>() and templating to figure this out
207  MPI_Datatype type = DataTypeMPI;
208 
209  // get global width and height of matrix
210  Int global_width = M.Width();
211  Int global_height = M.Height();
212 
213  // get local width and height, plus leading dimension of local matrix
214  Int W = M.LocalWidth();
215  Int H = M.LocalHeight();
216  Int LDim = M.LDim();
217 
218  // create a datatype to describe libelemental data in memory,
219  // data is stored in column-major order with a local height of H
220  // and a local width of W, also the leading dimension LDim >= H
221  // so there may be holes in our local buffer between consecutive
222  // columns which we need to account for
223 
224  // first we have H consecutive elements in a column
225  MPI_Datatype tmptype;
226  MPI_Type_contiguous(H, type, &tmptype);
227 
228  // then there may be some holes at then end of our column,
229  // since LDim >= H
230  MPI_Datatype coltype;
231  MPI_Aint extent = LDim * sizeof(DataType);
232  MPI_Type_create_resized(tmptype, 0, extent, &coltype);
233  MPI_Type_free(&tmptype);
234 
235  // finally we have W such columns
236  MPI_Type_contiguous(W, coltype, mattype);
237  MPI_Type_free(&coltype);
238  MPI_Type_commit(mattype);
239 
240  // create datatype to desribe fileview for a collective IO operation
241  // we will store matrix in column-major order in the file
242 
243  // get width and height of the process grid
244  int rank = M.Grid().Rank();
245  int ranks = M.Grid().Size();
246  int pheight = M.Grid().Height();
247  int pwidth = M.Grid().Width();
248 
249  // TODO: need to account for alignment if user has set this
250 
251  // create_darray expects processes to be in row-major order,
252  // find our global rank in row-major order
253  int prow = M.Grid().Row();
254  int pcol = M.Grid().Col();
255  int row_major_rank = prow * pwidth + pcol;
256 
257  int gsizes[2];
258  gsizes[0] = global_height;
259  gsizes[1] = global_width;
260  int distribs[2] = {MPI_DISTRIBUTE_CYCLIC, MPI_DISTRIBUTE_CYCLIC};
261  // TODO: if using block sizes > 1, then change dargs below (BlockHeight, BlockWidth)
262  int dargs[2] = {1, 1};
263  int psizes[2];
264  psizes[0] = pheight;
265  psizes[1] = pwidth;
266  MPI_Type_create_darray(ranks, row_major_rank, 2, gsizes, distribs, dargs, psizes, MPI_ORDER_FORTRAN, type, viewtype);
267  MPI_Type_commit(viewtype);
268 
269  return;
270 }

273 static void Write_MPI(const El::DistMatrix<DataType> &M, std::string basename = "DistMatrix", El::FileFormat format = El::BINARY, std::string title = "")
274 {
275  // TODO: error out if format != BINARY
276 
277  // TODO: use TypeMap<>() and templating to figure this out
278  MPI_Datatype type = DataTypeMPI;
279 
280  // define our file name
281  string filename = basename + "." + FileExtension(BINARY);
282  const char* path = filename.c_str();
283 
284  // get MPI communicator
285  MPI_Comm comm = M.Grid().Comm().comm;
286 
287  // get our rank
288  int rank = M.Grid().Rank();
289 
290  // first, delete the existing file
291  if (rank == 0) {
292  /*
293  int unlink_rc = unlink(path);
294  if (unlink_rc != 0) {
295  fprintf(stderr, "Error deleting file `%s'\n", path);
296  fflush(stderr);
297  }
298  */
299  MPI_File_delete(path, MPI_INFO_NULL);
300  }
301 
302  // get global width and height of matrix
303  Int global_width = M.Width();
304  Int global_height = M.Height();
305 
306  // define datatypes to describe local buffer and view into file
307  MPI_Datatype mattype, viewtype;
308  create_types(M, &mattype, &viewtype);
309 
310  // define hints for creating the file (e.g., number of stripes on Lustre)
311  MPI_Info info;
312  MPI_Info_create(&info);
313  MPI_Info_set(info, "striping_factor", "10");
314  //MPI_Info_set(info, "striping_factor", "80");
315  // TODO: specify number of writers?
316 
317  // open the file
318  MPI_File fh;
319  MPI_Status status;
320  char datarep[] = "native";
321  int amode = MPI_MODE_WRONLY | MPI_MODE_CREATE;
322  MPI_File_open(comm, path, amode, info, &fh);
323 
324  // done with the info object
325  MPI_Info_free(&info);
326 
327  // truncate file to 0 bytes
328 // MPI_File_set_size(fh, 0);
329 
330  // set our view to write header (height and width as unsigned 32-bit ints)
331  MPI_Offset disp = 0;
332  MPI_File_set_view(fh, disp, MPI_UINT32_T, MPI_UINT32_T, datarep, MPI_INFO_NULL);
333  if (rank == 0) {
334  uint32_t dimensions[2];
335  dimensions[0] = global_height;
336  dimensions[1] = global_width;
337  MPI_File_write_at(fh, 0, dimensions, 2, MPI_UINT32_T, &status);
338  }
339  disp += 2 * sizeof(uint32_t);
340 
341  // set view to write data
342  MPI_File_set_view(fh, disp, type, viewtype, datarep, MPI_INFO_NULL);
343 
344  // write our portion of the matrix, since we set our view using create_darray,
345  // all procs write at offset 0, the file view will take care of interleaving appropriately
346  const char* buf = (const char*) M.LockedBuffer();
347  MPI_File_write_at_all(fh, 0, buf, 1, mattype, &status);
348 
349  // close file
350  MPI_File_close(&fh);
351 
352  // free our datatypes
353  MPI_Type_free(&mattype);
354  MPI_Type_free(&viewtype);
355 
356  return;
357 }

360 static void Read_MPI(El::DistMatrix<DataType> &M, std::string filename, El::FileFormat format = El::BINARY, bool sequential = false)
361 {
362  // TODO: error out if format != BINARY
363 
364  // TODO: use TypeMap<>() and templating to figure this out
365  MPI_Datatype type = DataTypeMPI;
366 
367  // define our file name
368  const char* path = filename.c_str();
369 
370  // get MPI communicator
371  MPI_Comm comm = M.Grid().Comm().comm;
372 
373  // get our rank
374  int rank = M.Grid().Rank();
375 
376  // open the file
377  MPI_File fh;
378  MPI_Status status;
379  char datarep[] = "native";
380  int amode = MPI_MODE_RDONLY;
381  int rc = MPI_File_open(comm, path, amode, MPI_INFO_NULL, &fh);
382  if (rc != MPI_SUCCESS) {
383  if (rank == 0) {
384  cout << "Failed to open file `" << path << "'" << endl;
385  }
386  return;
387  }
388 
389  // set displacement to beginning of file
390  MPI_Offset disp = 0;
391 
392  // set our view to read header (height and width as unsigned 32-bit ints)
393  uint32_t dimensions[2];
394  MPI_File_set_view(fh, disp, MPI_UINT32_T, MPI_UINT32_T, datarep, MPI_INFO_NULL);
395  if (rank == 0) {
396  MPI_File_read_at(fh, 0, dimensions, 2, MPI_UINT32_T, &status);
397  }
398  disp += 2 * sizeof(uint32_t);
399 
400  // broadcast dimensions from rank 0
401  MPI_Bcast(dimensions, 2, MPI_UINT32_T, 0, comm);
402 
403  // resize matrix to hold data
404  Int global_height = dimensions[0];
405  Int global_width = dimensions[1];
406  M.Resize(global_height, global_width);
407 
408  // now define datatypes to describe local buffer and view into file
409  MPI_Datatype mattype, viewtype;
410  create_types(M, &mattype, &viewtype);
411 
412  // set view to write data
413  MPI_File_set_view(fh, disp, type, viewtype, datarep, MPI_INFO_NULL);
414 
415  // write our portion of the matrix, since we set our view using create_darray,
416  // all procs write at offset 0, the file view will take care of interleaving appropriately
417  char* buf = (char*) M.Buffer();
418  MPI_File_read_at_all(fh, 0, buf, 1, mattype, &status);
419 
420  // close file
421  MPI_File_close(&fh);
422 
423  // free our datatypes
424  MPI_Type_free(&mattype);
425  MPI_Type_free(&viewtype);
426 
427  return;
428 }
