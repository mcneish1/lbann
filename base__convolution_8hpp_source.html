103  base_convolution_layer(lbann_comm *comm,
104  int num_data_dims,
105  int num_output_channels,

107  std::vector<int> conv_pads,

109  bool has_bias,
110  cudnn::cudnn_manager *cudnn)
111  : learning(comm) {
112 
113  if (conv_dims.size() == 1) {
114  conv_dims.resize(num_data_dims, conv_dims[0]);
115  }
116  if (conv_pads.size() == 1) {
117  conv_pads.resize(num_data_dims, conv_pads[0]);
118  }
119  if (conv_strides.size() == 1) {
120  conv_strides.resize(num_data_dims, conv_strides[0]);
121  }
122  if ((int)conv_dims.size() != num_data_dims
123  && (int)conv_pads.size() != num_data_dims
124  && (int)conv_strides.size() != num_data_dims) {
125  std::stringstream err;

127  << "invalid number of convolution parameters "
128  << "(expected " << num_data_dims << " parameters, "
129  << "conv_dims has " << conv_dims.size() << ", "
130  << "conv_pads has " << conv_pads.size() << ", "
131  << "conv_strides has " << conv_strides.size() << ")";
132  throw lbann_exception(err.str());
133  }
134 
135  // Initialize convolution parameters
136  m_kernel_dims = conv_dims;
137  m_kernel_dims.insert(m_kernel_dims.begin(), num_output_channels);
138  m_conv_pads = conv_pads;

198  base_convolution_layer& operator=(const base_convolution_layer& other) {
199  learning::operator=(other);
200  m_kernel_dims = other.m_kernel_dims;
201  m_conv_pads = other.m_conv_pads;
202  m_conv_strides = other.m_conv_strides;
203  m_kernel_size = other.m_kernel_size;
204  m_bias_scaling_factor = other.m_bias_scaling_factor;
205 
206  // Copy matrices
207  #define COPY_MATRIX(src, dst) \
208  do { \
209  if(src != nullptr && dst != nullptr) { \
210  El::Copy(*src, *dst); \
211  } \
212  if(src != nullptr && dst == nullptr) { \
213  dst = src->Copy(); \
214  } \
215  if(src == nullptr && dst != nullptr) { \
216  delete dst; \
217  dst = nullptr; \
218  } \
219  } while(false)
220  COPY_MATRIX(other.m_kernel_weights_v, m_kernel_weights_v);
221  COPY_MATRIX(other.m_bias_weights_v, m_bias_weights_v);
222  COPY_MATRIX(other.m_kernel_weights_gradient, m_kernel_weights_gradient);
223  COPY_MATRIX(other.m_bias_weights_gradient, m_bias_weights_gradient);
224  #undef COPY_MATRIX
225 
226  #ifdef __LIB_CUDNN
227 
228  // Copy cuDNN objects
229  cudnn::copy_tensor_cudnn_desc(other.m_bias_cudnn_desc,
230  m_bias_cudnn_desc);
231  cudnn::copy_kernel_cudnn_desc(other.m_kernel_cudnn_desc,
232  m_kernel_cudnn_desc);
233  cudnn::copy_convolution_cudnn_desc(other.m_convolution_cudnn_desc,
234  m_convolution_cudnn_desc);
235 
236  // Copy GPU data
237  if (m_cudnn != nullptr) {
238  m_cudnn->deallocate_on_gpus(m_kernel_weights_gradient_d);
239  m_cudnn->deallocate_on_gpus(m_bias_weights_gradient_d);
240  m_kernel_weights_gradient_d = m_cudnn->copy(other.m_kernel_weights_gradient_d,
241  this->m_kernel_weights_gradient->Height(),
242  this->m_kernel_weights_gradient->Width());
243  m_bias_weights_gradient_d = m_cudnn->copy(other.m_bias_weights_gradient_d,
244  this->m_bias_weights_gradient->Height(),
245  this->m_bias_weights_gradient->Width());
246  }
247 
248  #endif // __LIB_CUDNN
249 
250  // Update views
251  setup_views();
252 
253  return *this;
254  }

297  void setup_data() override {
298  learning::setup_data();
299 
300  // Initialize default weights if none are provided
301  if (this->m_weights.size() > 2) {
302  std::stringstream err;
303  err << __FILE__ << " " << __LINE__ << " :: "
304  << "attempted to setup " << m_name << " with an invalid number of weights";
305  throw lbann_exception(err.str());
306  }
307  this->m_weights.resize(2, nullptr);
308  if (this->m_weights[0] == nullptr) {
309  this->m_weights[0] = new weights(this->m_comm, this->m_cudnn);
310  this->m_weights[0]->set_name(this->m_name + "_kernel_weights");
311  this->m_weights[0]->set_initializer(new he_normal_initializer(this->m_comm));
312  this->m_weights[0]->set_optimizer(m_neural_network_model->create_optimizer());
313  this->m_neural_network_model->add_weights(this->m_weights[0]);
314  }
315  if (this->m_weights[1] == nullptr) {
316  this->m_weights[1] = new weights(this->m_comm, this->m_cudnn);
317  this->m_weights[1]->set_name(this->m_name + "_bias_weights");
318  this->m_weights[1]->set_initializer(new constant_initializer(this->m_comm, DataType(0)));
319  this->m_weights[1]->set_optimizer(m_neural_network_model->create_optimizer());
320  this->m_neural_network_model->add_weights(this->m_weights[1]);
321  }
322 
323  // Initialize Glorot or He weight initialization
324  fan_in_fan_out_initializer* cast_initializer
325  = dynamic_cast<fan_in_fan_out_initializer*>(&this->m_weights[0]->get_initializer());
326  if (cast_initializer != nullptr) {
327  cast_initializer->set_fan_in(m_kernel_size / this->m_neuron_dims[0]);
328  cast_initializer->set_fan_out(m_kernel_size / this->m_prev_neuron_dims[0]);
329  }
330 
331  // Initialize bias
332  this->m_weights[1]->setup(this->m_neuron_dims[0], 1,
333  El::STAR, El::STAR);
334  El::Zeros(*this->m_bias_weights_gradient,
335  this->m_weights[1]->get_height(),
336  this->m_weights[1]->get_width());
337 
338  }

408  void apply_convolution_cudnn(bool during_forward_prop) {
409  #ifndef __LIB_CUDNN
410  throw lbann_exception("base_convolution_layer: cuDNN not detected");
411  #else
412 
413  // Useful constants
414  const DataType one = 1;
415  const DataType zero = 0;
416 
417  // Initialize input and output
418  cudnnTensorDescriptor_t input_cudnn_desc, output_cudnn_desc;
419  std::vector<DataType*> input_d, output_d;
420  if(during_forward_prop) {
421  input_cudnn_desc = this->m_prev_neurons_cudnn_desc;
422  output_cudnn_desc = this->m_neurons_cudnn_desc;
423  input_d = this->m_prev_activations_dv;
424  output_d = this->m_activations_d;
425  }
426  else {
427  input_cudnn_desc = this->m_neurons_cudnn_desc;
428  output_cudnn_desc = this->m_prev_neurons_cudnn_desc;
429  input_d = this->m_prev_error_signal_dv;
430  output_d = this->m_error_signal_d;
431  }
432 
433  // Get kernel weights
434  std::vector<DataType*> kernel_weights_d = m_weights[0]->get_values_gpu();
435 
436  // Perform convolution on each GPU
437  const int num_gpus = this->m_cudnn->get_num_gpus();
438  for(int i=0; i<num_gpus; ++i) {
439 
440  // Get work space
441  size_t work_space_size = this->m_cudnn->get_work_space_size(i);
442  if(work_space_size == 0) {
443  this->m_cudnn->set_maximum_work_space_size(i);
444  work_space_size = this->m_cudnn->get_work_space_size(i);
445  }
446  void *work_space = this->m_cudnn->get_work_space(i);
447 
448  // Determine convolution algorithm
449  cudnnConvolutionFwdAlgo_t convolution_cudnn_algorithm
450  = CUDNN_CONVOLUTION_FWD_ALGO_IMPLICIT_GEMM;
451  CHECK_CUDNN(cudnnGetConvolutionForwardAlgorithm(this->m_cudnn->get_handle(i),
452  input_cudnn_desc,
453  m_kernel_cudnn_desc,
454  m_convolution_cudnn_desc,
455  output_cudnn_desc,
456  CUDNN_CONVOLUTION_FWD_SPECIFY_WORKSPACE_LIMIT,
457  work_space_size,
458  &convolution_cudnn_algorithm));
459 
460  // Apply convolution
461  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
462  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
463  this->m_cudnn->get_stream(i)));
464  CHECK_CUDNN(cudnnConvolutionForward(this->m_cudnn->get_handle(i),
465  &one,
466  input_cudnn_desc,
467  input_d[i],
468  m_kernel_cudnn_desc,
469  kernel_weights_d[i],
470  m_convolution_cudnn_desc,
471  convolution_cudnn_algorithm,
472  work_space,
473  work_space_size,
474  &zero,
475  output_cudnn_desc,
476  output_d[i]));
477 
478  }
479 
480  #endif // #ifndef __LIB_CUDNN

830 
831  // Return immediately if there is no bias
832  if(m_bias_scaling_factor == DataType(0)) return;
833 
834  // Setup view
835  this->m_weights[1]->get_values_view(*m_bias_weights_v);
836

845 
846  // Apply bias to each output channel

852  = m_bias_scaling_factor * bias_weights_local(channel, 0);
853  for(El::Int col = 0; col < width_local; ++col) {
854  for(El::Int row = row_start; row < row_end; ++row) {
855  activations_local(row, col) += bias_term;

862  void compute_gradients_im2col(bool using_transposed_convolution) {
863 
864  // Get local matrices
865  const Mat& prev_activations_local = this->m_prev_activations_v->LockedMatrix();
866  const Mat& prev_error_signal_local = this->m_prev_error_signal_v->LockedMatrix();
867  Mat& kernel_weights_gradient_local = m_kernel_weights_gradient->Matrix();
868  Mat& bias_weights_gradient_local = m_bias_weights_gradient->Matrix();
869 
870  // Get convolution parameters
871  const El::Int width_local = prev_activations_local.Width();
872  const int num_input_channels = this->m_prev_neuron_dims[0];
873  const int num_output_channels = this->m_neuron_dims[0];
874  const int num_per_output_channel = this->m_num_neurons / num_output_channels;
875 
876  // Initialize weight gradients to zero
877  El::Zero(kernel_weights_gradient_local);
878  El::Zero(bias_weights_gradient_local);
879 
880  // Compute bias gradient
881  // Note: Sum is computed with Kahan summation
882  optimizer* bias_optimizer = this->m_weights[1]->get_optimizer();
883  if(m_bias_scaling_factor != DataType(0) && bias_optimizer != nullptr) {
884  #pragma omp parallel for
885  for(int channel = 0; channel < num_output_channels; ++channel) {
886  const El::Int row_start = channel * num_per_output_channel;
887  const El::Int row_end = (channel+1) * num_per_output_channel;
888  DataType sum = 0;
889  DataType correction = 0;
890  for(El::Int col = 0; col < width_local; ++col) {
891  for(El::Int row = row_start; row < row_end; ++row) {
892  DataType term = prev_error_signal_local(row, col);
893  term += correction;
894  const DataType next_sum = sum + term;
895  correction = term - (next_sum - sum);
896  sum = next_sum;
897  }
898  }
899  bias_weights_gradient_local(channel, 0) = m_bias_scaling_factor * sum;
900  }
901  bias_optimizer->allreduce_and_add_to_gradient(
902  *m_bias_weights_gradient,
903  DataType(1) / this->m_neural_network_model->get_effective_mini_batch_size());
904  }
905 
906  // Stop early if kernel is not being optimized
907  optimizer* kernel_optimizer = this->m_weights[0]->get_optimizer();
908  if (kernel_optimizer == nullptr) { return; }
909 
910  // Initialize im2col matrix
911  const int m = (using_transposed_convolution ?
912  m_kernel_size / num_input_channels :
913  m_kernel_size / num_output_channels);
914  const int n = (using_transposed_convolution ?
915  num_input_channels :
916  num_output_channels);
917  const int k = (using_transposed_convolution ?
918  this->m_num_prev_neurons / num_input_channels :
919  this->m_num_neurons / num_output_channels);
920  Mat im2col_matrix(m, k);
921 
922  // Compute kernel gradient contributions from each data sample
923  for(El::Int col = 0; col < width_local; ++col) {
924  if(using_transposed_convolution) {
925  const Mat prev_activations_col(k, n, prev_activations_local.LockedBuffer(0,col), k);
926  const Mat prev_error_signal_col
927  = El::LockedView(prev_error_signal_local, El::ALL, El::IR(col));
928  im2col(prev_error_signal_col,
929  im2col_matrix,
930  num_output_channels,
931  this->m_num_neuron_dims - 1,
932  &this->m_neuron_dims[1],
933  m_conv_pads.data(),
934  &m_kernel_dims[2],
935  m_conv_strides.data());
936  El::Gemm(El::NORMAL, El::NORMAL,
937  DataType(1), im2col_matrix, prev_activations_col,
938  DataType(1), kernel_weights_gradient_local);
939  }
940  else {
941  const Mat prev_activations_col
942  = El::LockedView(prev_activations_local, El::ALL, El::IR(col));
943  const Mat prev_error_signal_col(k, n, prev_error_signal_local.LockedBuffer(0,col), k);
944  im2col(prev_activations_col,
945  im2col_matrix,
946  num_input_channels,
947  this->m_num_prev_neuron_dims - 1,
948  &this->m_prev_neuron_dims[1],
949  m_conv_pads.data(),
950  &m_kernel_dims[2],
951  m_conv_strides.data());
952  El::Gemm(El::NORMAL, El::NORMAL,
953  DataType(1), im2col_matrix, prev_error_signal_col,
954  DataType(1), kernel_weights_gradient_local);
955  }
956  }
957 
958  // Scale and accumulate gradients
959  kernel_optimizer->allreduce_and_add_to_gradient(
960  *m_kernel_weights_gradient,
961  DataType(1) / this->m_neural_network_model->get_effective_mini_batch_size());
962 
963  }
