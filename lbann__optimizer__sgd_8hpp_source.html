83  void update_weight_bias_matrix(ElMat& WB_D, ElMat& WB) {
84 
85  // KERAS: lr = self.lr * (1.0 / (1.0 + self.decay * self.iterations))
86  lr = lr * (1.0 / (1.0 + decay * iterations));
87  iterations++;
88 
89  // velocity
90  // KERAS: v = self.momentum * m - lr * g # velocity
91  Scale(momentum, velocity);
92  Scale(lr, WB_D); // step
93  Axpy(-1, WB_D.Matrix(), velocity.Matrix()); // Update // TODO
94 
95  if(nesterov) {
96  //KERAS: new_p = p + self.momentum * v - lr * g
97  Copy(velocity, nesterov_ag);
98  Scale(momentum, nesterov_ag);
99  Axpy(-1, WB_D.Matrix(), nesterov_ag.Matrix()); // TODO
100  Axpy(1., nesterov_ag.Matrix(), WB.Matrix()); // TODO
101  }else {
102  //KERAS: new_p = p + v
103  Axpy(1., velocity.Matrix(), WB.Matrix()); // TODO
104  }
105 
106  }
