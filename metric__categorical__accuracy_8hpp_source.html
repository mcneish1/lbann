88  double compute_metric(AbsDistMat& predictions_v, AbsDistMat& groundtruth_v) override {
89  // Clear the contents of the intermediate matrices
90  El::Zeros(m_prediction_col_maxes, m_max_mini_batch_size, 1);
91  El::Zeros(m_replicated_prediction_col_maxes, m_max_mini_batch_size, 1);
92 
95  typedef typename std::conditional<T_layout == data_layout::MODEL_PARALLEL,
96  DistMat, StarVCMat>::type colnorm_cast_t;
97  ColumnMaxNorms((colnorm_cast_t) predictions_v, m_prediction_col_maxes_v);
98  El::Copy(m_prediction_col_maxes_v, m_replicated_prediction_col_maxes_v);
99 
100  El::Zeros(m_max_index, m_max_mini_batch_size, 1); // Clear the entire matrix
101 
103  for(int mb_index = 0; mb_index < predictions_v.LocalWidth(); mb_index++) {
104  int mb_global_index = predictions_v.GlobalCol(mb_index);
105  DataType sample_max = m_replicated_prediction_col_maxes_v.GetLocal(mb_global_index, 0);
106  for(int f_index = 0; f_index < predictions_v.LocalHeight(); f_index++) {
107  if(predictions_v.GetLocal(f_index, mb_index) == sample_max) {
108  m_max_index_v.Set(mb_global_index, 0, predictions_v.GlobalRow(f_index));
109  }
110  }
111  }
112 
113  El::Zeros(m_reduced_max_indices, m_max_mini_batch_size, 1); // Clear the entire matrix
116  m_comm->model_allreduce(m_max_index.Buffer(),
117  m_max_index.Height() * m_max_index.Width(),
118  m_reduced_max_indices.Buffer(), El::mpi::MAX);
119 
121  int num_errors = 0;
122 
123  // Copy(groundtruth_v, Y_local);
124 
128 
131  for(int mb_index= 0; mb_index < groundtruth_v.LocalWidth(); mb_index++) {
132  int targetidx = -1;
133  for(int f_index= 0; f_index < groundtruth_v.LocalHeight(); f_index++) {
134  if(groundtruth_v.GetLocal(f_index, mb_index) == DataType(1)) {
135  targetidx = groundtruth_v.GlobalRow(f_index);
136  }
137  }
138  if(targetidx != -1) {
139  int global_mb_index = groundtruth_v.GlobalCol(mb_index);
140  if(m_reduced_max_indices_v.Get(global_mb_index, 0) != targetidx) {
141  num_errors++;
142  }
143  }
144  }
145 
146  num_errors = m_comm->model_allreduce(num_errors);
147  return num_errors;
148  }

150  double report_metric(execution_mode mode) override {
151  statistics *stats = get_statistics(mode);
152  double errors_per_epoch = stats->m_error_per_epoch;
153  long samples_per_epoch = stats->m_samples_per_epoch;
154 
155  double accuracy = (double)(samples_per_epoch - errors_per_epoch) / samples_per_epoch * 100;
156 
157  return accuracy;
