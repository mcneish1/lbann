252 bool lbann::model::need_checkpoint()
253 {
254  /* TODO: since we're using clocks, this requires a bcast for each call,
255  * we could use number of samples processed to make a local decision */
256 
257  // if none of our checkpoint conditions are set, assume we're not checkpointing
258  if (m_checkpoint_epochs == 0 &&
259  m_checkpoint_steps == 0 &&
260  m_checkpoint_secs == 0.0)
261  {
262  return false;
263  }
264 
265  // assume that we won't checkpoint
266  int flag = 0;
267 
268  // if at start of epoch and evenly divide
269  if (flag == 0 && m_checkpoint_epochs > 0) {
270  if (at_epoch_start()) {
271  flag = (int) (m_current_epoch % m_checkpoint_epochs == 0);
272  }
273  }
274 
275  // if our current step is evenly divisable by checkpoint steps,
276  // take a checkpoint
277  if (flag == 0 && m_checkpoint_steps > 0) {
278  flag = (int) (m_current_step % m_checkpoint_steps == 0);
279  }
280 
281  // check the clock if time-based checkpoint is enabled
282  if (flag == 0 && m_checkpoint_secs != 0.0) {
283  // have rank 0 determine whether we should checkpoint
284  // to avoid issues with clock skew, we rely on rank 0 to make decision
285  if (m_rank == 0) {
286  // get the current time
287  double current = MPI_Wtime();
288 
289  // compute time next checkpoint is due
290  double next = m_checkpoint_last + m_checkpoint_secs;
291 
292  // determine whether it's time for a checkpoint
293  flag = (current >= next);
294  }
295 
296  // get flag from rank 0
297  MPI_Bcast(&flag, 1, MPI_INT, 0, MPI_COMM_WORLD);
298  }
299 
300  return (bool)flag;
301 }

304 static bool write_latest(const char* dir, const char* name, int epoch, int train)
305 {
306  // define filename
307  char filename[1024];
308  sprintf(filename, "%s/%s", dir, name);
309 
310  // open the file for writing
311  int fd = lbann::openwrite(filename);
312  if (fd != -1) {
313  lbann::write_uint32(fd, "epoch", (uint32_t)epoch);
314  lbann::write_uint32(fd, "train", (uint32_t)train);
315 
316  // close our file
317  lbann::closewrite(fd, filename);
318  }

320  return true;

324 static bool read_latest(const char* dir, const char* name, int* epochLast, int* trainLast)
325 {
326  // assume we don't have a file, we'll return -1 in that case
327  *epochLast = -1;
328  *trainLast = -1;
329 
330  // define filename
331  char filename[1024];
332  sprintf(filename, "%s/%s", dir, name);
333 
334  // open the file for reading
335  int fd = lbann::openread(filename);
336  if (fd != -1) {
337  // read epoch from file
338  uint32_t epoch;
339  lbann::read_uint32(fd, "epoch", &epoch);
340  *epochLast = (int) epoch;
341 
342  // read epoch from file
343  uint32_t train;
344  lbann::read_uint32(fd, "train", &train);
345  *trainLast = train;
346 
347  // close our file
348  lbann::closeread(fd, filename);
349  }
350 
351  return true;
352 }

361 bool lbann::model::checkpointShared()
362 {
363  // if the checkpoint directory is not defined, bail
364  if (m_checkpoint_dir.length() == 0) {
365  return false;
366  }
367 
368  // time how long this takes
369  Timer timer;
370 
371  // get checkpoint directory
372  const char* dir = m_checkpoint_dir.c_str();
373 
374  // read current epoch and step counters from model
375  int64_t epoch = m_current_epoch;
376  int64_t step = m_current_step;
377 
378  // let user know we're saving a checkpoint
379  MPI_Barrier(MPI_COMM_WORLD);
380  if (m_rank == 0) {
381  timer.Start();
382  printf("Checkpoint: epoch %d step %d ...\n", epoch, step);
383  fflush(stdout);
384  }
385 
386  // create top level directory
387  //const char* dir = trainParams.ParameterDir.c_str();
388  lbann::makedir(dir);
389 
390  // create subdirectory for this epoch
391  char epochdir[1024];
392  snprintf(epochdir, sizeof(epochdir), "%s/shared.epoch.%d.step.%d", dir, epoch, step);
393 
394  // start our checkpoint
395  lbann::persist p;
396  p.open_checkpoint(epochdir);
397 
398  // call virtual function to checkpoint model state
399  this->save_to_checkpoint_shared(p);
400 
401  // close our checkpoint
402  p.close_checkpoint();
403 
404  uint64_t bytes_count = p.get_bytes();
405 
406  // write epoch number to current file, we do this at the end so as to only update
407  // this file when we know we have a new valid checkpoint
408  if (m_rank == 0) {
409  write_latest(dir, "shared.last", epoch, step);
410  }
411 
412  // stop timer and report cost
413  MPI_Barrier(MPI_COMM_WORLD);
414  if (m_rank == 0) {
415  double secs = timer.Stop();
416  double bw = 0.0;
417  if (secs > 0.0) {
418  bw = ((double) bytes_count) / (secs * 1024.0 * 1024.0);
419  }
420  printf("Checkpoint complete: Epoch=%d Step=%d (%f secs, %llu bytes, %f MB/sec)\n",
421  epoch, step, secs, (unsigned long long) bytes_count, bw
422  );
423  fflush(stdout);
424  }
425 
426  // saved a checkpoint, update our last checkpoint time
427  m_checkpoint_last = MPI_Wtime();
428 
429  return true;
430 }

432 bool lbann::model::restartShared()
433 {
434  // if the checkpoint directory is not defined, bail
435  if (m_checkpoint_dir.length() == 0) {
436  return false;
437  }
438 
439  // get top level directory
440  const char* dir = m_checkpoint_dir.c_str();
441 
442  // read epoch number from current file
443  int epoch, step;
444  if (m_rank == 0) {
445  read_latest(dir, "shared.last", &epoch, &step);
446  }
447  MPI_Bcast(&epoch, 1, MPI_INT, 0, MPI_COMM_WORLD);
448  MPI_Bcast(&step, 1, MPI_INT, 0, MPI_COMM_WORLD);
449 
450  // if we couldn't find the latest epoch, just return
451  if (epoch < 0) {
452  return false;
453  }
454 
455  // time how long this takes
456  Timer timer;
457 
458  // let user know we're restarting from a checkpoint
459  MPI_Barrier(MPI_COMM_WORLD);
460  if (m_rank == 0) {
461  timer.Start();
462  printf("Restart: epoch %d ...\n", epoch);
463  fflush(stdout);
464  }
465 
466  // get subdirectory for this epoch
467  char epochdir[1024];
468  sprintf(epochdir, "%s/shared.epoch.%d.step.%d", dir, epoch, step);
469 
470  // open our checkpoint
471  lbann::persist p;
472  p.open_restart(epochdir);
473 
474  // call virtual function to restore model from checkpoint
475  this->load_from_checkpoint_shared(p);
476 
477  // close our checkpoint
478  p.close_restart();
479 
480  uint64_t bytes_count = p.get_bytes();
481 
482  // let user know we've completed reading our restart
483  MPI_Barrier(MPI_COMM_WORLD);
484  if (m_rank == 0) {
485  double secs = timer.Stop();
486  double bw = 0.0;
487  if (secs > 0.0) {
488  bw = ((double) bytes_count) / (secs * 1024.0 * 1024.0);
489  }
490  printf("Restart complete: Epoch=%d Step=%d (%f secs, %llu bytes, %f MB/sec)\n",
491  epoch, step, secs, (unsigned long long) bytes_count, bw
492  );
493  fflush(stdout);
494  }
495 
496  return true;
497 }
