39 void lbann_callback_timer::on_epoch_end(model* m) {
40  double end = get_time();
41  double epoch_time = end - epoch_start;
42  // Compute minibatch stats.
43  double mean = std::accumulate(batch_times.begin(), batch_times.end(), 0.0) /
44  batch_times.size();
45  auto minmax = std::minmax_element(batch_times.begin(), batch_times.end());
46  double stdev = 0.0;
47  for (const auto& t : batch_times) {
48  stdev += (t - mean) * (t - mean);
49  }
50  stdev = sqrt(stdev / (batch_times.size() - 1));
51  batch_times.clear();
52 
53  // Output.
54  lbann_comm* comm = m->get_comm();
55  if (comm->am_model_master()) {
56  if (comm->am_world_master()) {
57  std::vector<double> epoch_times(comm->get_num_models());
58  std::vector<double> means(comm->get_num_models());
59  std::vector<double> mins(comm->get_num_models());
60  std::vector<double> maxes(comm->get_num_models());
61  std::vector<double> stdevs(comm->get_num_models());
62  comm->intermodel_gather(epoch_time, epoch_times);
63  comm->intermodel_gather(mean, means);
64  comm->intermodel_gather(*(minmax.first), mins);
65  comm->intermodel_gather(*(minmax.second), maxes);
66  comm->intermodel_gather(stdev, stdevs);
67  for (int i = 0; i < comm->get_num_models(); ++i) {
68  std::cout << "Model " << i << " Epoch time: " << epoch_times[i] << "s; ";
69  std::cout << "Mean minibatch time: " << means[i] << "s; ";
70  std::cout << "Min: " << mins[i] << "s; ";
71  std::cout << "Max: " << maxes[i] << "s; ";
72  std::cout << "Stdev: " << stdevs[i] << "s" << std::endl;
73  }
74  } else {
75  comm->intermodel_gather(epoch_time, comm->get_intermodel_master());
76  comm->intermodel_gather(mean, comm->get_intermodel_master());
77  comm->intermodel_gather(*(minmax.first), comm->get_intermodel_master());
78  comm->intermodel_gather(*(minmax.second), comm->get_intermodel_master());
79  comm->intermodel_gather(stdev, comm->get_intermodel_master());
80  }
81  }
82 }
