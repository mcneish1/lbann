139 void weights::setup(int height,
140  int width,
141  El::Distribution col_dist,
142  El::Distribution row_dist) {
143 
144  // Check if weights has already been set up
145  if (m_values != nullptr) {
146  const El::DistData dist_data(*m_values);
147  if (m_height == height
148  && m_width == width
149  && dist_data.colDist == col_dist
150  && dist_data.rowDist == row_dist) {
151  return;
152  } else {
153  std::stringstream err;
154  err << __FILE__ << " " << __LINE__ << " :: "
155  << "attempted to setup " << m_name << " with "
156  << "height=" << height << ","
157  << "width=" << width << ","
158  << "col_dist=" << col_dist << ","
159  << "row_dist=" << row_dist << ", "
160  << "but the it is already setup with "
161  << "height=" << m_height << ","
162  << "width=" << m_width << ","
163  << "col_dist=" << dist_data.colDist << ","
164  << "row_dist=" << dist_data.rowDist;
165  throw lbann_exception(err.str());
166  }
167  }
168 
169  // Initialize weights matrix
170  m_height = height;
171  m_width = width;
172  m_values = m_initializer->construct_matrix(m_height,
173  m_width,
174  col_dist,
175  row_dist);
176 
177  // Setup GPU objects
178  if (m_cudnn != nullptr) {
179  setup_gpu();
180  }
181 
182  // Setup optimizer
183  if (m_optimizer != nullptr) {
184  m_optimizer->setup(*this);
185  }
186 
187 }

189 void weights::setup_gpu() {
190  #ifndef __LIB_CUDNN
191  std::stringstream err;
192  err << __FILE__ << " " << __LINE__ << " :: " << "cuDNN not detected";
193  throw lbann_exception(err.str());
194  #else
195 
196  // Check that distributed matrix is in STAR,STAR format
197  const El::DistData dist_data(*m_values);
198  if (dist_data.colDist != El::STAR || dist_data.rowDist != El::STAR) {
199  std::stringstream err;
200  err << __FILE__ << " " << __LINE__ << " :: "
201  << "attempted to setup weights with "
202  << "col_dist=" << dist_data.colDist << ","
203  << "row_dist=" << dist_data.rowDist << ", "
204  << "but weights with GPU support must have STAR,STAR format";
205  throw lbann_exception(err.str());
206  }
207 
208  // Copy weights matrix to GPU
209  m_cudnn->allocate_on_gpus(m_values_d, m_height, m_width);
210  m_cudnn->broadcast_to_gpus(m_values_d, m_values->LockedMatrix());
211 
212  #endif // __LIB_CUDNN
213 }

268 void weights::set_value(int row, int col, DataType value) {
269  if (m_cudnn == nullptr) {
270  if (m_values->IsLocal(row, col)) {
271  const El::Int local_row = m_values->LocalRow(row);
272  const El::Int local_col = m_values->LocalCol(col);
273  m_values->SetLocal(local_row, local_col, value);
274  }
275  } else {
276  #if __LIB_CUDNN
277  Mat cpu_value(1, 1);
278  cpu_value(0, 0) = value;
279  std::vector<DataType*> gpu_value = m_values_d;
280  for (DataType*& pointer : gpu_value) {
281  pointer += row + col * m_height;
282  }
283  m_cudnn->broadcast_to_gpus(gpu_value, cpu_value);
284  #endif // __LIB_CUDNN
285  }
286 }
