171  void fp_set_std_matrix_view() override {
172  // Use the predetermined size of the mini-batch to set the current
173  // batch size for the neural network
174  El::Int cur_mini_batch_size = get_current_mini_batch_size();
175  this->m_model->set_current_mini_batch_size(cur_mini_batch_size);
176 
177  // Use the precomputed size of the global mini-batch to set the
178  // current effective batch size across all models
179  int total_mini_batch_size = get_current_global_mini_batch_size();
180  this->m_model->set_effective_mini_batch_size(total_mini_batch_size);
181 
182  // Once the current mini-batch size is defined, set the standard view for activations only
183  El::View(*m_activations_v, *m_activations, El::ALL, El::IR(0, cur_mini_batch_size));
184  io_buffer->set_local_matrix_bypass(&this->m_activations_v->Matrix());
185  io_buffer->set_std_matrix_view(cur_mini_batch_size);
186  }

191  void fp_compute() override {
192  execution_mode mode = this->m_model->get_execution_mode();
193  int num_samples_in_batch = io_buffer->fetch_to_local_matrix(get_data_reader(), mode);
194 
195  if(dynamic_cast<partitioned_io_buffer*>(io_buffer) != nullptr) {
196  // Use the predetermined size of the mini-batch to set the current
197  // batch size for the neural network
198  num_samples_in_batch = get_current_mini_batch_size();
199 
200  update_num_samples_processed(num_samples_in_batch);
201  }else if(dynamic_cast<distributed_io_buffer*>(io_buffer) != nullptr) {
202  if(((distributed_io_buffer*) io_buffer)->is_current_root(mode)) {
204  update_num_samples_processed(num_samples_in_batch);
205  }
206 
209  this->m_model->set_current_mini_batch_size(Layer::m_comm->model_broadcast(((distributed_io_buffer*) io_buffer)->current_root_rank(mode), num_samples_in_batch));
210 
211  io_buffer->distribute_from_local_matrix(*this->m_activations, get_data_reader(), mode);
212  }else {
213  std::stringstream err;
214  err << __FILE__ << " " << __LINE__ << " :: "
215  << "could not fp_compute for I/O layers : encoutered generic_io_buffer type";
216  throw lbann_exception(err.str());
217  }
218  }
