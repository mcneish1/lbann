406 void model::evaluate(execution_mode mode) {
407 
408  // Return early if execution mode is invalid
409  if (!is_execution_mode_valid(mode)) return;
410 
411  // Initialize model for beginning of evaluation
412  set_execution_mode(mode);
413  m_objective_function->clear_history();
414  for (auto&& m : m_metrics) {
415  m->reset_metric();
416  }
417 
418  // Start evaluation
419  switch(mode) {
420  case execution_mode::validation:
421  do_validation_begin_cbs();
422  break;
423  case execution_mode::testing:
424  do_test_begin_cbs();
425  break;
426  default:
427  throw lbann_exception("Illegal execution mode in evaluate function");
428  }
429 
430  // Evaluate on mini-batches until data set is finished
431  while (!evaluate_mini_batch()) {}
432 
433  // Finish evaluation
434  switch(mode) {
435  case execution_mode::validation:
436  do_validation_end_cbs();
437  break;
438  case execution_mode::testing:
439  do_test_end_cbs();
440  break;
441  default:
442  throw lbann_exception("Illegal execution mode in evaluate function");
443  }
444 }

446 void model::train(int num_epochs) {
447  do_train_begin_cbs();
448  for (int epoch = 0; epoch < num_epochs; ++epoch) {
449 
450  // Stop if training has been terminated
451  if (get_terminate_training()) {
452  break;
453  }
454 
455  // Initialize model for beginning of training epoch
456  set_execution_mode(execution_mode::training);
457  m_objective_function->clear_history();
458  for (auto&& m : m_metrics) {
459  m->reset_metric();
460  }
461 
462  // Start epoch
463  ++m_current_epoch;
464  do_epoch_begin_cbs();
465 
466  // Train on mini-batches until data set is finished
467  while (!train_mini_batch()) {}
468 
469  // Evaluate model on validation set
470  evaluate(execution_mode::validation);
471 
472  // Finish epoch
473  do_epoch_end_cbs();
474 
475  }
476  do_train_end_cbs();
477 }

554 bool model::train_mini_batch() {
555  do_batch_begin_cbs();
556 
557  // Reset matrices
558  reset_layers();
559 
560  forward_prop();
561 
562  // Compute objective function value
563  m_objective_function->compute_value();
564 
565  // Compute gradients
566  m_objective_function->compute_gradient();
567  backward_prop();
568 
569  // Update weights
570  for (weights* w : m_weights) {
571  optimizer* opt = w->get_optimizer();
572  if (opt != nullptr) {
573  opt->step();
574  }
575  }
576 
577  const bool finished = update_layers();
578 
579  // Finish up
580  do_batch_end_cbs();
581  ++m_current_step;
582  return finished;
583 }

909 bool model::checkpointShared() {
910  // if the checkpoint directory is not defined, bail
911  if (m_checkpoint_dir.length() == 0) {
912  return false;
913  }
914 
915  // time how long this takes
916  Timer timer;
917 
918  // get checkpoint directory
919  const char *dir = m_checkpoint_dir.c_str();
920 
921  // read current epoch and step counters from model
922  int epoch = m_current_epoch;
923  int step = m_current_step;
924 
925  // let user know we're saving a checkpoint
926  MPI_Barrier(MPI_COMM_WORLD);
927  if (m_comm->am_world_master()) {
928  timer.Start();
929  printf("Checkpoint: epoch %d step %d ...\n", epoch, step);
930  fflush(stdout);
931  }
932 
933  // create top level directory
934  //const char* dir = trainParams.ParameterDir.c_str();
935  makedir(dir);
936 
937  // create subdirectory for this epoch
938  char epochdir[1024];
939  snprintf(epochdir, sizeof(epochdir), "%s/shared.epoch.%d.step.%d", dir, epoch, step);
940 
941  // start our checkpoint
942  persist p;
943  p.open_checkpoint(epochdir);
944 
945  // call virtual function to checkpoint model state
946  this->save_to_checkpoint_shared(p);
947 
948  // close our checkpoint
949  p.close_checkpoint();
950 
951  uint64_t bytes_count = p.get_bytes();
952 
953  // write epoch number to current file, we do this at the end so as to only update
954  // this file when we know we have a new valid checkpoint
955  if (m_comm->am_world_master()) {
956  write_latest(dir, "shared.last", epoch, step);
957  }
958

966  }
967  printf("Checkpoint complete: Epoch=%d Step=%d (%f secs, %llu bytes, %f MB/sec)\n",
968  epoch, step, secs, (unsigned long long) bytes_count, bw
969  );
970  fflush(stdout);
971  }
972 
973  // saved a checkpoint, update our last checkpoint time
974  m_checkpoint_last = MPI_Wtime();
975 
976  return true;
