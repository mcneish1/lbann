214 void model::replace_weights(std::vector<weights*>& new_weights) {
215 
216  // Check that number of weights is valid
217  if (new_weights.size() > m_weights.size()) {
218  std::stringstream err;
219  err << __FILE__ << " " << __LINE__ << " :: "
220  << "attempted to replace weights with an invalid number of weights "
221  << "(expected at most " << m_weights.size() << ", found " << new_weights.size() << ")";
222  throw lbann_exception(err.str());
223  }
224 
225  // Replace weights in list
226  std::vector<weights *> old_weights(m_weights.begin(),
227  m_weights.begin() + new_weights.size());
228  std::unordered_map<weights *,weights *> weights_map;
229  std::unordered_map<Layer *,Layer *> layer_map;
230  for (size_t i = 0; i < new_weights.size(); ++i) {
231  m_weights[i] = weights_map[old_weights[i]] = new_weights[i];
232  }
233  remap_pointers(layer_map, weights_map);
234 
235  // Delete old weights
236  for (const auto& w : old_weights) {
237  delete w;
238  }

302 void model::remap_pointers(const std::unordered_map<Layer *,Layer *>& layer_map,
303  const std::unordered_map<weights *,weights *>& weights_map) {
304 
305  // Fix pointers in objective function
306  if (m_objective_function != nullptr) {
307  auto layer_pointers = m_objective_function->get_layer_pointers();
308  for (auto& layer_pointer : layer_pointers) {
309  if (layer_map.count(layer_pointer) > 0) {
310  layer_pointer = layer_map.at(layer_pointer);
311  }
312  }
313  m_objective_function->set_layer_pointers(layer_pointers);
314  auto weights_pointers = m_objective_function->get_weights_pointers();
315  for (auto& weights_pointer : weights_pointers) {
316  if (weights_map.count(weights_pointer) > 0) {
317  weights_pointer = weights_map.at(weights_pointer);
318  }
319  }
320  m_objective_function->set_weights_pointers(weights_pointers);
321  }
322 
323  // Fix pointers in metrics
324  for (const auto& m : m_metrics) {
325  auto layer_pointers = m->get_layer_pointers();
326  for (auto& layer_pointer : layer_pointers) {
327  if (layer_map.count(layer_pointer) > 0) {
328  layer_pointer = layer_map.at(layer_pointer);
329  }
330  }
331  m->set_layer_pointers(layer_pointers);
332  }
333 
334  // Fix pointers in layers
335  for (const auto& l : m_layers) {
336  auto layer_pointers = l->get_layer_pointers();
337  for (auto& layer_pointer : layer_pointers) {
338  if (layer_map.count(layer_pointer) > 0) {
339  layer_pointer = layer_map.at(layer_pointer);
340  }
341  }
342  l->set_layer_pointers(layer_pointers);
343  auto weights_pointers = l->get_weights();
344  for (auto& weights_pointer : weights_pointers) {
345  if (weights_map.count(weights_pointer) > 0) {
346  weights_pointer = weights_map.at(weights_pointer);
347  }
348  }
349  l->set_weights(weights_pointers);
350  }

383 void model::setup_layer_topology() {
384 
385  // Perform breadth-first searches to find connected components
386  std::queue<const Layer *> layer_queue;
387  std::unordered_set<const Layer *> layer_set;
388  for (const auto& layer : m_layers) {
389  layer_queue.push(layer);
390  layer_set.insert(layer);
391  }
392  while (!layer_queue.empty()) {
393  const Layer *layer = layer_queue.front();
394  layer_queue.pop();
395  std::vector<const Layer *> relatives;
396  for (const auto& parent : layer->get_parent_layers()) {
397  relatives.push_back(parent);
398  }
399  for (const auto& child : layer->get_child_layers()) {
400  relatives.push_back(child);
401  }
402  for (const auto& relative : relatives) {
403  if (layer_set.count(relative) == 0) {
404  m_layers.push_back(const_cast<Layer *>(relative));
405  layer_queue.push(relative);
406  layer_set.insert(relative);
407  }
408  }
409  }
410 
411  // Make sure parent and child relationships are reciprocated
412  for (const auto& layer : m_layers) {
413  for (const auto& parent : layer->get_parent_layers()) {
414  const_cast<Layer *>(parent)->add_child_layer(layer);
415  }
416  for (const auto& child : layer->get_child_layers()) {
417  const_cast<Layer *>(child)->add_parent_layer(layer);
418  }
419  }
420 
421 }

434 void model::setup_weights() {
435 
436  // List of used and unused weights
437  std::unordered_set<weights *> weights_set(m_weights.begin(),
438  m_weights.end());
439  std::set<weights *> unused_weights(m_weights.begin(),
440  m_weights.end());
441 
442  // Find weights used by layers
443  for (const auto& layer : m_layers) {
444  for (const auto& w : layer->get_weights()) {
445  if (weights_set.count(w) == 0) {
446  m_weights.push_back(w);
447  weights_set.insert(w);
448  }
449  unused_weights.erase(w);
450  }
451  }
452 
453  // Find weights used by objective function
454  for (const auto& w : m_objective_function->get_weights_pointers()) {
455  if (weights_set.count(w) == 0) {
456  m_weights.push_back(w);
457  weights_set.insert(w);
458  }
459  unused_weights.erase(w);
460  }
461 
462  // Delete unused weights
463  for (const auto& w : unused_weights) {
464  m_weights.erase(std::remove(m_weights.begin(), m_weights.end(), w),
465  m_weights.end());
466  }
467 
468 }

474 void model::evaluate(execution_mode mode) {
475 
476  // Return early if execution mode is invalid
477  if (!is_execution_mode_valid(mode)) return;
478  if (mode != execution_mode::validation
479  && mode != execution_mode::testing) {
480  std::stringstream err;
481  err << __FILE__ << " " << __LINE__ << " :: "
482  << "invalid execution mode for evaluation";
483  throw lbann_exception(err.str());
484  }
485 
486  // Evaluate on all mini-batches
487  reset_epoch_statistics(mode);
488  reset_mode_and_model(mode);
489  do_evaluate_begin_cbs(mode);
490  while (!evaluate_mini_batch(mode)) {}
491  do_evaluate_end_cbs(mode);
492 }

494 void model::train(int num_epochs) {
495  do_train_begin_cbs();
496  for (int epoch = m_current_epoch; epoch < num_epochs; ++epoch) {
497 
498  // Stop if training has been terminated
499  if (get_terminate_training()) { break; }
500 
501  // Setup epoch
502  reset_mode_and_model(execution_mode::training);
503 
504  // Train on mini-batches
505  do_epoch_begin_cbs();
506  while (!train_mini_batch()) {}
507  // Once the epoch is complete, Increase the count
508  ++m_current_epoch;
509  do_epoch_end_cbs();
510  reset_epoch_statistics(execution_mode::training);
511 
512  // Evaluate on validation set
513  evaluate(execution_mode::validation);
514  }
515  do_train_end_cbs();
516 }

535 bool model::evaluate_mini_batch(execution_mode mode) {
536  do_batch_begin_cbs(mode);
537  forward_prop(mode);
538  m_objective_function->evaluate(mode);
539  for (const auto& m : m_metrics) {
540  m->evaluate(mode);
541  }
542  const bool finished = update_layers();
543  switch(m_execution_mode) {
544  case execution_mode::validation:
545  ++m_current_validation_step;
546  break;
547  case execution_mode::testing:
548  ++m_current_testing_step;
549  break;
550  default:
551  throw lbann_exception("Illegal execution mode in evaluate mini-batch function");
552  }
553  do_batch_end_cbs(mode);
554  return finished;
555 }

557 bool model::train_mini_batch() {
558  do_batch_begin_cbs(execution_mode::training);
559 
560  // Forward prop step
561  forward_prop(execution_mode::training);
562  m_objective_function->evaluate(execution_mode::training);
563  for (const auto& m : m_metrics) {
564  m->evaluate(execution_mode::training);
565  }
566 
567  // Backward prop step
568  clear_error_signals();
569  m_objective_function->differentiate();
570  backward_prop();
571  m_objective_function->compute_weight_regularization();
572 
573  // Update step
574  update_weights();
575  const bool finished = update_layers();
576 
577  ++m_current_step;
578  do_batch_end_cbs(execution_mode::training);
579  return finished;
580 }
