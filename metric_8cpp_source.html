82 void metric::setup(model& m) {
83 
84  // Set target layer if needed
85  if (m_target_layer == nullptr) {
86  const std::vector<Layer*> layers = m.get_layers();
87  for (int i = layers.size() - 1; i >= 0; --i) {
88  const generic_target_layer *target = dynamic_cast<const generic_target_layer*>(layers[i]);
89  if (target != nullptr) {
90  m_target_layer = target;
91  break;
92  }
93  }
94  if (m_target_layer == nullptr) {
95  std::stringstream err;
96  err << __FILE__ << " " << __LINE__ << " :: "
97  << "could not setup metric with target layer";
98  throw lbann_exception(err.str());
99  }
100  }

102 }

104 EvalType metric::evaluate(execution_mode mode) {
105 
106  // Check if target layer pointer has been setup
107  if (m_target_layer == nullptr) {
108  std::stringstream err;
109  err << __FILE__ << " " << __LINE__ << " :: "
110  << "attempted to evaluate metric without setting a target layer";
111  throw lbann_exception(err.str());
112  }
113 
114  // Evaluate objective function
115  const int mini_batch_size = m_target_layer->get_prediction().Width();
116  const EvalType total_value = evaluate_compute(m_target_layer->get_prediction(),
117  m_target_layer->get_ground_truth());
118 
119  // Record result in statistics and return
120  m_statistics[mode].add_value(total_value, mini_batch_size);
121  return total_value / mini_batch_size;
122 
123 }

179 bool metric::load_from_checkpoint_shared(persist& p) {
180  struct metric_statistics::packing_header training_header, validation_header, testing_header;
181  if (p.get_rank() == 0) {
182  m_statistics[execution_mode::training].unpack_scalars(p, &training_header);
183  m_statistics[execution_mode::validation].unpack_scalars(p, &validation_header);
184  m_statistics[execution_mode::testing].unpack_scalars(p, &testing_header);
185  }
186 
187  MPI_Bcast(&training_header, sizeof(training_header), MPI_BYTE, 0, MPI_COMM_WORLD);
188  MPI_Bcast(&validation_header, sizeof(validation_header), MPI_BYTE, 0, MPI_COMM_WORLD);
189  MPI_Bcast(&testing_header, sizeof(testing_header), MPI_BYTE, 0, MPI_COMM_WORLD);
190 
191  m_statistics[execution_mode::training].unpack_header(training_header);
192  m_statistics[execution_mode::validation].unpack_header(validation_header);
193  m_statistics[execution_mode::testing].unpack_header(testing_header);
194  return true;
