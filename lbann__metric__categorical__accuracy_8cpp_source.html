74 double lbann::metrics::categorical_accuracy::compute_metric(ElMat& predictions_v, ElMat& groundtruth_v) {
75 
76  // Clear the contents of the intermediate matrices
77  Zeros(YsColMax, m_max_mini_batch_size, 1);
78  Zeros(YsColMaxStar, m_max_mini_batch_size, 1);
79 
81  ColumnMax((DistMat)predictions_v, YsColMax_v);
82  Copy(YsColMax_v, YsColMaxStar_v);
83 
84  Zeros(m_max_index, m_max_mini_batch_size, 1); // Clear the entire matrix
85 
87  for(Int mb_index = 0; mb_index < predictions_v.LocalWidth(); mb_index++) {
88  int mb_global_index = predictions_v.GlobalCol(mb_index);
89  DataType sample_max = YsColMaxStar_v.GetLocal(mb_global_index, 0);
90  for(Int f_index = 0; f_index < predictions_v.LocalHeight(); f_index++) {
91  if(predictions_v.GetLocal(f_index, mb_index) == sample_max) {
92  m_max_index_v.Set(mb_global_index, 0, predictions_v.GlobalRow(f_index));
93  }
94  }
95  }
96 
97  Zeros(m_reduced_max_indices, m_max_mini_batch_size, 1); // Clear the entire matrix
100  comm->model_allreduce(m_max_index.Buffer(), m_max_index.Height() * m_max_index.Width(), m_reduced_max_indices.Buffer(), mpi::MAX);
101 
103  int num_errors = 0;
104 
105  // Copy(groundtruth_v, Y_local);
106 
110 
113  for(Int mb_index= 0; mb_index < groundtruth_v.LocalWidth(); mb_index++) {
114  Int targetidx = -1;
115  for(Int f_index= 0; f_index < groundtruth_v.LocalHeight(); f_index++) {
116  if(groundtruth_v.GetLocal(f_index, mb_index) == (DataType) 1.) {
117  targetidx = groundtruth_v.GlobalRow(f_index);
118  }
119  }
120  if(targetidx != -1) {
121  Int global_mb_index = groundtruth_v.GlobalCol(mb_index);
122  if(m_reduced_max_indices_v.Get(global_mb_index, 0) != targetidx) {
123  num_errors++;
124  }
125  }
126  }
127 
128  num_errors = comm->model_allreduce(num_errors);
129  return num_errors;
130 }
