40  switch(format) {
41  case matrix_format::MC_MR:
42  return new Adam<DistMat>(comm, lr, rho1, rho2, eps);
43  case matrix_format::CIRC_CIRC:
44  return new Adam<CircMat>(comm, lr, rho1, rho2, eps);
45  case matrix_format::STAR_STAR:
