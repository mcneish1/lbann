101 void lbann::deep_neural_network::train(int num_epochs, int evaluation_frequency)
102 {
103  do_train_begin_cbs();
104 
105  // Epoch main loop
106  for (int epoch = 0; epoch < num_epochs; ++epoch) {
107  // Check if training has been terminated
108  if (get_terminate_training()) break;
109 
110  // due to restart, may not always be at start of epoch
111  // use mini batch index in data reader to signify start of epoch
112  if (at_epoch_start()) {
113  ++m_current_epoch;
114  do_epoch_begin_cbs();
115  }
116 
118  m_execution_mode = execution_mode::training;
119  for (size_t l = 0; l < m_layers.size(); ++l) {
120  m_layers[l]->m_execution_mode = execution_mode::training;
121  }
122 
123  // Train on mini-batches until data set is traversed
124  // Note: The data reader shuffles the data after each epoch
125  for (auto&& m : metrics) { m->reset_metric(); }
126  bool finished_epoch;
127  do {
128  finished_epoch = train_mini_batch();
129 
130  // save a checkpoint if needed
131  if (need_checkpoint()) {
132  checkpointShared();
133  }
134  } while(!finished_epoch);
135  if(evaluation_frequency > 0
136  && (epoch + 1) % evaluation_frequency == 0)
137  {
138  // Evaluate model on validation set
139  // TODO: do we need validation callbacks here?
140  // do_validation_begin_cbs();
141  evaluate(execution_mode::validation);
142  // do_validation_end_cbs();
143 
144  // Set execution mode back to training
145  m_execution_mode = execution_mode::training;
146  for (size_t l = 0; l < m_layers.size(); l++) {
147  m_layers[l]->m_execution_mode = execution_mode::training;
148  }
149  }
150 
151  do_epoch_end_cbs();
152 
153  for (Layer* layer : m_layers) {
154  layer->epoch_reset();
155  }
156 
157  // save checkpoint after epoch
158  if (need_checkpoint()) {
159  checkpointShared();
160  }
161  }
162 
163  do_train_end_cbs();
164 }

166 bool lbann::deep_neural_network::train_mini_batch()
167 {
168  do_batch_begin_cbs();
169 
170  // Forward propagation
171  do_model_forward_prop_begin_cbs();
172  DataType L2NormSum = 0;
173  for (size_t l = 0; l < m_layers.size(); ++l) {
174  do_layer_forward_prop_begin_cbs(m_layers[l]);
175  m_layers[l]->forwardProp();
176  do_layer_forward_prop_end_cbs(m_layers[l]);
177  }
178  do_model_forward_prop_end_cbs();
179 
180  // Backward propagation
181  do_model_backward_prop_begin_cbs();
182  for (size_t l = m_layers.size(); l-- > 0;) {
183  do_layer_backward_prop_begin_cbs(m_layers[l]);
184  m_layers[l]->backProp();
185  do_layer_backward_prop_end_cbs(m_layers[l]);
186  }
187  do_model_backward_prop_end_cbs();
188 
190  for (size_t l = m_layers.size() - 1; l > 0; --l) {
191  m_layers[l]->update();
192  }
193  const bool data_set_processed = m_layers[0]->update();
194 
195  do_batch_end_cbs();
196  ++m_current_step; // Update the current step once the entire mini-batch is complete
197  return data_set_processed;
198 }

200 void lbann::deep_neural_network::evaluate(execution_mode mode)
201 {
202  switch(mode) {
203  case execution_mode::validation:
204  do_validation_begin_cbs(); break;
205  case execution_mode::testing:
206  do_test_begin_cbs(); break;
207  default:
208  throw lbann_exception("Illegal execution mode in evaluate function");
209  }
210 
211  // Set the execution mode
212  m_execution_mode = mode;
213  for (size_t l = 0; l < m_layers.size(); ++l) {
214  m_layers[l]->m_execution_mode = mode;
215  }
216 
217  // Evaluate on mini-batches until data set is traversed
218  // Note: The data reader shuffles the data after each epoch
219  for (auto&& m : metrics) { m->reset_metric(); }
220  bool finished_epoch;
221  do {
222  finished_epoch = evaluate_mini_batch();
223  } while(!finished_epoch);
224 
225  switch(mode) {
226  case execution_mode::validation:
227  do_validation_end_cbs(); break;
228  case execution_mode::testing:
229  do_test_end_cbs();
230  // Reset after testing.
231  for (Layer* layer : m_layers) {
232  layer->epoch_reset();
233  }
234  break;
235  default:
236  throw lbann_exception("Illegal execution mode in evaluate function");
237  }
238 
239  return;
240 }

242 bool lbann::deep_neural_network::evaluate_mini_batch()
243 {
244  do_batch_evaluate_begin_cbs();
245 
246  // forward propagation (mini-batch)
247  do_model_evaluate_forward_prop_begin_cbs();
248  for (size_t l = 0; l < m_layers.size(); l++) {
249  do_layer_evaluate_forward_prop_begin_cbs(m_layers[l]);
250  m_layers[l]->forwardProp();
251  do_layer_evaluate_forward_prop_end_cbs(m_layers[l]);
252  }
253  do_model_evaluate_forward_prop_end_cbs();
254 
255  // Update layers
256  // Note: should only affect the input and target layers
257  for (size_t l = m_layers.size() - 1; l > 0; --l) {
258  m_layers[l]->update();
259  }
260  const bool data_set_processed = m_layers[0]->update();
261  do_batch_evaluate_end_cbs();
262  switch(m_execution_mode) {
263  case execution_mode::validation:
264  ++m_current_validation_step;
265  break;
266  case execution_mode::testing:
267  ++m_current_testing_step;
268  break;
269  default:
270  throw lbann_exception("Illegal execution mode in evaluate mini-batch function");
271  }
272  return data_set_processed;
