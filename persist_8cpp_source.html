66 bool lbann::writeDist(int fd, const char *filename, const DistMat& M, uint64_t *bytes) {
67  // TODO: store in network order
68 
69  // build our header
70  struct layer_header header;
71  header.rank = (uint64_t) M.Grid().Rank();
72  header.width = (uint64_t) M.Width();
73  header.height = (uint64_t) M.Height();
74  header.localwidth = (uint64_t) M.LocalWidth();
75  header.localheight = (uint64_t) M.LocalHeight();
76  header.ldim = (uint64_t) M.LDim();
77 
78  // write the header to the file
79  ssize_t write_rc = write(fd, &header, sizeof(header));
80  if (write_rc != sizeof(header)) {
81  // error!
82  }
83  *bytes += write_rc;
84 
85  // now write the data for our part of the distributed matrix
86  const El::Int localHeight = M.LocalHeight();
87  const El::Int localWidth = M.LocalWidth();
88  const El::Int lDim = M.LDim();
89  if(localHeight == lDim) {
90  // the local dimension in memory matches the local height,
91  // so we can write our data in a single shot
92  auto *buf = (void *) M.LockedBuffer();
93  El::Int bufsize = localHeight * localWidth * sizeof(DataType);
94  write_rc = write(fd, buf, bufsize);
95  if (write_rc != bufsize) {
96  // error!
97  }
98  *bytes += write_rc;
99  } else {
100  // TODO: if this padding is small, may not be a big deal to write it out anyway
101  // we've got some padding along the first dimension
102  // while storing the matrix in memory, avoid writing the padding
103  for(El::Int j = 0; j < localWidth; ++j) {
104  auto *buf = (void *) M.LockedBuffer(0, j);
105  El::Int bufsize = localHeight * sizeof(DataType);
106  write_rc = write(fd, buf, bufsize);
107  if (write_rc != bufsize) {
108  // error!
109  }
110  *bytes += write_rc;
111  }
112  }
113 
114  return true;
115 }

119 bool lbann::readDist(int fd, const char *filename, DistMat& M, uint64_t *bytes) {
120  // read in the header
121  struct layer_header header;
122  ssize_t read_rc = read(fd, &header, sizeof(header));
123  if (read_rc != sizeof(header)) {
124  // error!
125  throw lbann_exception("Failed to read layer header");
126  }
127  *bytes += read_rc;
128 
129  // resize our global matrix
130  El::Int height = header.height;
131  El::Int width = header.width;
132  M.Resize(height, width);
133 
134  // TODO: check that header values match up
135 
136  if(M.ColStride() == 1 && M.RowStride() == 1) {
137  if(M.Height() == M.LDim()) {
138  auto *buf = (void *) M.Buffer();
139  El::Int bufsize = height * width * sizeof(DataType);
140  read_rc = read(fd, buf, bufsize);
141  if (read_rc != bufsize) {
142  // error!
143  throw lbann_exception("Failed to read layer data");
144  }
145  *bytes += read_rc;
146  } else {
147  for(El::Int j = 0; j < width; ++j) {
148  auto *buf = (void *) M.Buffer(0, j);
149  El::Int bufsize = height * sizeof(DataType);
150  read_rc = read(fd, buf, bufsize);
151  if (read_rc != bufsize) {
152  // error!
153  throw lbann_exception("Failed to read layer data");
154  }
155  *bytes += read_rc;
156  }
157  }
158  } else {
159  const El::Int localHeight = M.LocalHeight();
160  const El::Int localWidth = M.LocalWidth();
161  const El::Int lDim = M.LDim();
162  if(localHeight == lDim) {
163  auto *buf = (void *) M.Buffer();
164  El::Int bufsize = localHeight * localWidth * sizeof(DataType);
165  read_rc = read(fd, buf, bufsize);
166  if (read_rc != bufsize) {
167  // error!
168  throw lbann_exception("Failed to read layer data");
169  }
170  *bytes += read_rc;
171  } else {
172  for(El::Int jLoc = 0; jLoc < localWidth; ++jLoc) {
173  auto *buf = (void *) M.Buffer(0, jLoc);
174  El::Int bufsize = localHeight * sizeof(DataType);
175  read_rc = read(fd, buf, bufsize);
176  if (read_rc != bufsize) {
177  // error!
178  throw lbann_exception("Failed to read layer data");
179  }
180  *bytes += read_rc;
181  }
182  }
183  }
184  return true;
185 }
