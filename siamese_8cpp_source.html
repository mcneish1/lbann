43  // Sort layers topologically

46  // Determine layers in master head
47  int heads_start = m_layers.size();
48  int heads_end = -1;
49  for (size_t i = 0; i < m_layers.size(); ++i) {
50  if (m_layers[i]->is_fan_out_layer()) {
51  heads_start = i + 1;
52  break;
53  }
54  }
55  for (int i = m_layers.size() - 1; i >= 0; --i) {
56  if (m_layers[i]->is_fan_in_layer()) {
57  heads_end = i;
58  break;
59  }
60  }
61  if (heads_start > heads_end) {
62  throw lbann_exception("siamese_model: siamese models must have a fan-out layer before a fan-in layer");
63  }
64  std::vector<Layer*> master_head(m_layers.begin() + heads_start,
65  m_layers.begin() + heads_end);
66  const int master_head_end = heads_end;
67 
68  // Construct map from follower layers to master layers
69  std::unordered_map<Layer*,Layer*> follower_to_master_layer;
70  for (Layer* master_layer : m_layers) {
71  follower_to_master_layer[master_layer] = master_layer;
72  }
73 
74  // Duplicate heads
75  for (int i = 1; i < m_num_heads; ++i) {
76  std::string name_suffix = "_" + std::to_string(i);
77 
78  // Construct map from master layers to follower layers

87

99  heads_end++;

101  }

106  std::vector<Layer*> follower_layer_pointers;
107  for (Layer* master_layer_pointer : master_layer_pointers) {

109  follower_layer_pointers.push_back(follower_layer_pointer);
110  }
111  follower_layer->set_layer_pointers(follower_layer_pointers);
112  }
113 
114  // Fix pointers at start and end of head
115  m_layers[heads_start-1]->add_child_layer(follower_head.front());
116  m_layers[heads_end]->add_parent_layer(follower_head.back());
117 
118  }
119

122  Layer* layer = m_layers[i];
123  layer->set_model(this);
124  layer->setup();
