51 lbann::Optimizer *lbann::SGD_factory::create_optimizer(matrix_format format) {
52  switch(format) {
53  case matrix_format::MC_MR:
54  return new SGD<DistMat>(this->comm, this->lr, this->momentum, this->decay, this->nesterov);
55  case matrix_format::CIRC_CIRC:
56  return new SGD<CircMat>(this->comm, this->lr, this->momentum, this->decay, this->nesterov);
57  case matrix_format::STAR_STAR:
58  return new SGD<StarMat>(this->comm, this->lr, this->momentum, this->decay, this->nesterov);
59  case matrix_format::STAR_VC:
60  return new SGD<StarVCMat>(this->comm, this->lr, this->momentum, this->decay, this->nesterov);
61  default:
62  // TODO: throw an exception
63  printf("LBANN Error: unknown matrix distribution for SGD optimizer\n");
64  exit(-1);
65  }
