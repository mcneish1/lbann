91  void update_weight_bias_matrix(ElMat& WB_D, ElMat& WB) {
92  Copy(WB_D, WB_D_Temp);
93  // Square each entry of the WB_D matrix
94  EntrywiseMap(WB_D_Temp, std::function<DataType(const DataType&)>(_sq));
95  // Add squared value to WB_D_Cache
96  Axpy(1., WB_D_Temp, WB_D_Cache);
97 
98  Copy(WB_D_Cache, WB_D_Temp);
99  // Compute the inverse of the square root of the historical gradient (with a small perturbation)
100  EntrywiseMap(WB_D_Temp, std::function<DataType(const DataType&)>(_sqrt));
101  Copy(WB_D, WB_D_Temp2);
102  Hadamard(WB_D_Temp2, WB_D_Temp, WB_D);
103 
104  // WBL2NormSum = 0.0;
105  Axpy((DataType)-lr, WB_D, WB);
106  }
