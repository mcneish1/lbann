428  void fp_compute_gpu() {
429  #ifndef __LIB_CUDNN
430  throw lbann_exception("batch_normalization_layer: cuDNN not detected");
431  #else
432 
433  // Check execution mode
434  const bool is_training = this->m_model->get_execution_mode() == execution_mode::training;
435 
436  // Matrix parameters
437  const int num_gpus = this->m_cudnn->get_num_gpus();
438  const int height = this->m_prev_activations_v->Height();
439  const int width = this->m_prev_activations_v->Width();
440  const int local_width = this->m_prev_activations_v->LocalWidth();
441  const int num_channels = this->m_neuron_dims[0];
442  const int channel_size = this->m_num_neurons / num_channels;
443 
444  // Compute statistics
445  if(is_training) {
446 
447  // Get GPU objects
448  std::vector<DataType*> running_mean_d = m_weights[2]->get_values_gpu();
449  std::vector<DataType*> running_var_d = m_weights[3]->get_values_gpu();
450 
451  // Compute sums and sums of squares on GPUs
452  for(int i=0; i<num_gpus; ++i) {
453  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
454  const int col_start = std::min(i * this->m_mini_batch_size_per_gpu, local_width);
455  const int col_end = std::min((i+1) * this->m_mini_batch_size_per_gpu, local_width);
456  const int current_width = col_end - col_start;
457  batch_normalization_cuda
458  ::channel_sums_and_sqsums<DataType>(height,
459  current_width,
460  num_channels,
461  this->m_prev_activations_dv[i],
462  m_mean_d[i],
463  m_var_d[i],
464  this->m_cudnn->get_stream(i));
465  }
466 
467  // Accumulate sums and sums of squares
468  int samples_per_sum;
469  if (m_use_global_stats) {
470  this->m_cudnn->global_allreduce_on_gpus(m_mean_d,
471  num_channels,
472  1,
473  m_mean->RedundantComm());
474  this->m_cudnn->global_allreduce_on_gpus(m_var_d,
475  num_channels,
476  1,
477  m_var->RedundantComm());
478  samples_per_sum = channel_size * width;
479  } else {
480  this->m_cudnn->allreduce_on_gpus(m_mean_d, num_channels, 1);
481  this->m_cudnn->allreduce_on_gpus(m_var_d, num_channels, 1);
482  samples_per_sum = channel_size * local_width;
483  }
484 
485  // Compute minibatch statistics and running statistics
486  for(int i=0; i<num_gpus; ++i) {
487  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
488  batch_normalization_cuda
489  ::sums_to_statistics<DataType>(num_channels,
490  samples_per_sum,
491  m_decay,
492  m_mean_d[i],
493  m_var_d[i],
494  running_mean_d[i],
495  running_var_d[i],
496  this->m_cudnn->get_stream(i));
497  }
498 
499  }
500 
501  // Get GPU objects
502  std::vector<DataType*> scale_d = m_weights[0]->get_values_gpu();
503  std::vector<DataType*> bias_d = m_weights[1]->get_values_gpu();
504  std::vector<DataType*> mean_d = is_training ? m_mean_d : m_weights[2]->get_values_gpu();
505  std::vector<DataType*> var_d = is_training ? m_var_d : m_weights[3]->get_values_gpu();
506 
507  // Perform batch normalization with each GPU
508  for(int i=0; i<num_gpus; ++i) {
509  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
510  const int col_start = std::min(i * this->m_mini_batch_size_per_gpu, local_width);
511  const int col_end = std::min((i+1) * this->m_mini_batch_size_per_gpu, local_width);
512  const int current_width = col_end - col_start;
513  batch_normalization_cuda
514  ::batch_normalization<DataType>(height,
515  current_width,
516  num_channels,
517  this->m_prev_activations_dv[i],
518  mean_d[i],
519  var_d[i],
520  m_epsilon,
521  scale_d[i],
522  bias_d[i],
523  this->m_activations_d[i],
524  this->m_cudnn->get_stream(i));
525  }
526 
527  #endif // __LIB_CUDNN
528  }

530  void bp_compute_gpu() {
531  #ifndef __LIB_CUDNN
532  throw lbann_exception("batch_normalization_layer: cuDNN not detected");
533  #else
534 
535  // Check execution mode
536  const bool is_training = this->m_model->get_execution_mode() == execution_mode::training;
537 
538  // GPU objects
539  const int num_gpus = this->m_cudnn->get_num_gpus();
540  std::vector<DataType*> scale_d = m_weights[0]->get_values_gpu();
541  std::vector<DataType*> mean_d = is_training ? m_mean_d : m_weights[2]->get_values_gpu();
542  std::vector<DataType*> var_d = is_training ? m_var_d : m_weights[3]->get_values_gpu();
543 
544  // Matrix parameters
545  const int effective_mini_batch_size = this->m_model->get_effective_mini_batch_size();
546  const int height = this->m_prev_activations_v->Height();
547  const int width = this->m_prev_activations_v->Width();
548  const int local_width = this->m_prev_activations_v->LocalWidth();
549  const int num_channels = this->m_neuron_dims[0];
550 
551  // Compute local gradient contributions
552  for(int i=0; i<num_gpus; ++i) {
553  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
554  const int col_start = std::min(i * this->m_mini_batch_size_per_gpu, local_width);
555  const int col_end = std::min((i+1) * this->m_mini_batch_size_per_gpu, local_width);
556  const int current_width = col_end - col_start;
557  batch_normalization_cuda
558  ::batch_normalization_backprop1<DataType>(height,
559  current_width,

629  void fp_compute_cpu() {
630 
631  // Check execution mode
632  const bool is_training = this->m_model->get_execution_mode() == execution_mode::training;
633 
634  // Setup matrix views
635  this->m_weights[0]->get_values_view(*m_scale_v);
636  this->m_weights[1]->get_values_view(*m_bias_v);
637  this->m_weights[2]->get_values_view(*m_running_mean_v);
638  this->m_weights[3]->get_values_view(*m_running_var_v);
639 
640  // Local matrices
641  const Mat& prev_activations_local = this->m_prev_activations_v->LockedMatrix();
642  Mat& activations_local = this->m_activations_v->Matrix();
643 
644  // Matrix parameters
645  const int width = this->m_prev_activations_v->Width();
646  const El::Int local_width = this->m_prev_activations_v->LocalWidth();
647  const int num_channels = this->m_neuron_dims[0];
648  const int channel_size = this->m_num_neurons / num_channels;
649 
650  // Compute statistics
651  if(is_training) {
652 
653  // Local matrices
654  Mat& mean_local = m_mean->Matrix();
655  Mat& var_local = m_var->Matrix();
656  const Mat& running_mean_local = m_running_mean_v->LockedMatrix();
657  const Mat& running_var_local = m_running_var_v->LockedMatrix();
658  Mat& new_running_mean_local = m_mean_gradient->Matrix();
659  Mat& new_running_var_local = m_var_gradient->Matrix();
660 
661  // Compute sums and sums of squares
662  #pragma omp parallel for
663  for(int channel = 0; channel < num_channels; ++channel) {
664  auto sum = DataType(0);
665  auto sqsum = DataType(0);
666  const El::Int row_start = channel * channel_size;
667  const El::Int row_end = (channel+1) * channel_size;
668  for(El::Int col = 0; col < local_width; ++col) {
669  for(El::Int row = row_start; row < row_end; ++row) {
670  const DataType x = prev_activations_local(row, col);
671  sum += x;
672  sqsum += x * x;
673  }
674  }
675  mean_local(channel, 0) = sum;
676  var_local(channel, 0) = sqsum;
677  }
678  DataType num_samples;
679  if (m_use_global_stats) {
680  m_comm->allreduce(*m_mean, m_mean->RedundantComm(), El::mpi::SUM);
681  m_comm->allreduce(*m_var, m_var->RedundantComm(), El::mpi::SUM);
682  num_samples = channel_size * width;
683  } else {
684  num_samples = channel_size * local_width;
685  }
686 
687  // Compute minibatch statistics
688  // Note: new_running_mean and new_running_var are stored in
689  // m_mean_gradient and m_var_gradient
690  #pragma omp parallel for
691  for(int channel = 0; channel < num_channels; ++channel) {
692  const DataType mean = mean_local(channel, 0) / num_samples;
693  const DataType sqmean = var_local(channel, 0) / num_samples;
694  const DataType var = num_samples / (num_samples - DataType(1)) * std::max(sqmean - mean * mean, DataType(0));
695  const DataType old_running_mean = running_mean_local(channel, 0);
696  const DataType old_running_var = running_var_local(channel, 0);
697  const DataType new_running_mean = m_decay * old_running_mean + (DataType(1) - m_decay) * mean;
698  const DataType new_running_var = m_decay * old_running_var + (DataType(1) - m_decay) * var;
699  mean_local(channel, 0) = mean;
700  var_local(channel, 0) = var;
701  new_running_mean_local(channel, 0) = new_running_mean;
702  new_running_var_local(channel, 0) = new_running_var;
703  }
704  m_weights[2]->set_values(*m_mean_gradient);
705  m_weights[3]->set_values(*m_var_gradient);
706 
707  }
708 
709  // Local matrices
710  const Mat& mean_local = is_training ? m_mean->LockedMatrix() : m_running_mean_v->LockedMatrix();
711  const Mat& var_local = is_training ? m_var->LockedMatrix() : m_running_var_v->LockedMatrix();
712  const Mat& scale_local = m_scale_v->LockedMatrix();
713  const Mat& bias_local = m_bias_v->LockedMatrix();
714 
715  // Iterate through channels
716  #pragma omp parallel for
717  for(int channel = 0; channel < num_channels; ++channel) {
718 
719  // Get channel parameters
720  const DataType mean = mean_local(channel, 0);
721  const DataType var = var_local(channel, 0);
722  const DataType inv_stdev = 1 / std::sqrt(var + m_epsilon);
723  const DataType scale = scale_local(channel, 0);
724  const DataType bias = bias_local(channel, 0);
725 
726  // Apply batch normalization to inputs in channel
727  const El::Int row_start = channel * channel_size;
728  const El::Int row_end = (channel+1) * channel_size;
729  for(El::Int col = 0; col < local_width; ++col) {
730  for(El::Int row = row_start; row < row_end; ++row) {
731  const DataType x = prev_activations_local(row, col);
732  const DataType xhat = (x - mean) * inv_stdev;
733  const DataType y = scale * xhat + bias;
734  activations_local(row, col) = y;
735  }
736  }
737 
738  }
739 
740  }

742  void bp_compute_cpu() {
743 
744  // Check execution mode
745  const bool is_training = this->m_model->get_execution_mode() == execution_mode::training;
746 
747  // Setup matrix views
748  this->m_weights[0]->get_values_view(*m_scale_v);
749  this->m_weights[1]->get_values_view(*m_bias_v);
750  this->m_weights[2]->get_values_view(*m_running_mean_v);
751  this->m_weights[3]->get_values_view(*m_running_var_v);
752 
753  // Local matrices
754  const Mat& prev_activations_local = this->m_prev_activations_v->LockedMatrix();
755  const Mat& prev_error_signal_local = this->m_prev_error_signal_v->LockedMatrix();
756  Mat& error_signal_local = this->m_error_signal_v->Matrix();
757  const Mat& mean_local = is_training ? m_mean->LockedMatrix() : m_running_mean_v->LockedMatrix();
758  const Mat& var_local = is_training ? m_var->LockedMatrix() : m_running_var_v->LockedMatrix();
759  const Mat& scale_local = m_scale_v->LockedMatrix();
760  Mat& mean_gradient_local = m_mean_gradient->Matrix();
761  Mat& var_gradient_local = m_var_gradient->Matrix();
762  Mat& scale_gradient_local = m_scale_gradient->Matrix();
763  Mat& bias_gradient_local = m_bias_gradient->Matrix();
764 
765  // Matrix parameters
766  const int effective_mini_batch_size = this->m_model->get_effective_mini_batch_size();
767  const int width = this->m_prev_activations_v->Width();
768  const El::Int local_width = this->m_prev_activations_v->LocalWidth();
769  const int num_channels = this->m_neuron_dims[0];
770  const int channel_size = this->m_num_neurons / num_channels;
771 
772  // Compute local gradients
773  #pragma omp parallel for
774  for(int channel = 0; channel < num_channels; ++channel) {
775 
776  // Initialize channel parameters and gradients
777  const DataType mean = mean_local(channel, 0);
778  const DataType var = var_local(channel, 0);
779  const DataType scale = scale_local(channel, 0);
780  const DataType inv_stdev = 1 / std::sqrt(var + m_epsilon);
781  const DataType dvar_factor = inv_stdev * inv_stdev * inv_stdev / 2;
782  auto dmean = DataType(0);
783  auto dvar = DataType(0);
784  auto dscale = DataType(0);
785  auto dbias = DataType(0);
786 
787  // Compute gradient contributions from local entries
788  const El::Int row_start = channel * channel_size;
789  const El::Int row_end = (channel+1) * channel_size;
790  for(El::Int col = 0; col < local_width; ++col) {
791  for(El::Int row = row_start; row < row_end; ++row) {
792  const DataType x = prev_activations_local(row, col);
793  const DataType xhat = (x - mean) * inv_stdev;
794  const DataType dy = prev_error_signal_local(row, col);
795  dscale += dy * xhat;
796  dbias += dy;
797  const DataType dxhat = dy * scale;
798  dmean += - dxhat * inv_stdev;
799  dvar += - dxhat * (x - mean) * dvar_factor;
800  }
801  }
802  mean_gradient_local(channel, 0) = dmean;
803  var_gradient_local(channel, 0) = dvar;
804  scale_gradient_local(channel, 0) = dscale;
805  bias_gradient_local(channel, 0) = dbias;
806 
807  }
808 
809  // Accumulate gradients
810  if (is_training) {
811  if (m_use_global_stats) {
812  m_comm->allreduce(*m_mean_gradient,
813  m_mean_gradient->RedundantComm(),
814  El::mpi::SUM);
815  m_comm->allreduce(*m_var_gradient,
816  m_var_gradient->RedundantComm(),
817  El::mpi::SUM);
818  }
819  } else {
820  El::Zero(*m_mean_gradient);
821  El::Zero(*m_var_gradient);
822  }
823  optimizer* scale_optimizer = m_weights[0]->get_optimizer();
824  if (scale_optimizer != nullptr) {
825  scale_optimizer->stage_gradient_for_accumulation(
826  *m_scale_gradient,
827  DataType(1) / effective_mini_batch_size);
828  }
829  optimizer* bias_optimizer = m_weights[1]->get_optimizer();
830  if (bias_optimizer != nullptr) {
831  bias_optimizer->stage_gradient_for_accumulation(
832  *m_bias_gradient,
833  DataType(1) / effective_mini_batch_size);
834  }
835 
836  // Compute error signal
837  #pragma omp parallel for
838  for(int channel = 0; channel < num_channels; ++channel) {
839 
840  // Initialize channel parameters and gradients
841  const DataType mean = mean_local(channel, 0);
842  const DataType var = var_local(channel, 0);
843  const DataType scale = scale_local(channel, 0);
844  const DataType dmean = mean_gradient_local(channel, 0);
845  const DataType dvar = var_gradient_local(channel, 0);
846 
847  // Compute useful constants
848  const DataType num_samples = width * channel_size;
849  const DataType inv_stdev = 1 / std::sqrt(var + m_epsilon);
850  const DataType dmean_term = dmean / num_samples;
851  const DataType dvar_term = dvar * 2 / (num_samples - DataType(1));
852 
853  // Compute error signal for current channel
854  const El::Int row_start = channel * channel_size;
855  const El::Int row_end = (channel+1) * channel_size;
856  for(El::Int col = 0; col < local_width; ++col) {
857  for(El::Int row = row_start; row < row_end; ++row) {
858  const DataType x = prev_activations_local(row, col);
859  const DataType dy = prev_error_signal_local(row, col);
860  const DataType dxhat = dy * scale;
861  DataType dx = dxhat * inv_stdev;
862  dx += dmean_term;
863  dx += dvar_term * (x - mean);
864  error_signal_local(row, col) = dx;
865  }
866  }
867 
868  }
869 
870  }
