52 void greedy_layerwise_autoencoder::insert_mirror(uint32_t layer_index) {
53  // compute layer index for mirrror
54  size_t mirror_index = layer_index + 2;
55 
56  // build mirror layer
57  Layer *original_layer = m_layers[layer_index];
58  Layer *mirror_layer = NULL;
59  switch(original_layer->get_data_layout()){
60  case data_layout::MODEL_PARALLEL:
61  mirror_layer = new reconstruction_layer<data_layout::MODEL_PARALLEL>(m_comm, original_layer);
62  break;
63  case data_layout::DATA_PARALLEL:
64  mirror_layer = new reconstruction_layer<data_layout::DATA_PARALLEL>(m_comm, original_layer);
65  break;
66  default:
67  break;
68  }
69 
70  // insert mirror layer into model
71  insert(mirror_index, mirror_layer);
72 
73  //call base model set up at each phase to reindex and set appropriate matrices, fp and bp input
74  //assume that necessary layer parameters are set e.g., m_num_neurons when layers were constructed
75  setup_subset(layer_index, mirror_index+1); //set up all active layers
76 
77  // set flag to indicate that we have a mirror layer inserted
78  m_have_mirror = 1;
79 }

117 void greedy_layerwise_autoencoder::train(int num_epochs) {
118  while(m_end_index < m_layers.size()-1) {
119  set_end_index();
120  train_phase(num_epochs);
121  m_start_index = m_end_index;
122  m_reconstruction_layers.insert(m_reconstruction_layers.begin(),m_layers[m_end_index]);
123  // move on to the next (layerwise) phase
124  ++m_current_phase;
125  }
126  //evaluate and save all layers for i.e., (1) global cost (2) image reconstruction
127  evaluate(execution_mode::testing);
128  reset_phase();
129 }
