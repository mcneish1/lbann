89 void rmsprop::step_compute(AbsDistMat& values, const AbsDistMat& gradient) {
90 
91  // Get local matrix data
92  const int local_height = values.LocalHeight();
93  const int local_width = values.LocalWidth();
94  DataType* __restrict__ values_buffer = values.Buffer();
95  const int values_ldim = values.LDim();
96  const DataType* __restrict__ gradient_buffer = gradient.LockedBuffer();
97  const int gradient_ldim = gradient.LDim();
98  DataType* __restrict__ cache_buffer = m_cache->Buffer();
99  const int cache_ldim = m_cache->LDim();
100 
101  // Check if matrix data is contiguous
102  if (values_ldim != local_height
103  || gradient_ldim != local_height
104  || cache_ldim != local_height) {
105  // Update with non-contiguous data
106  #pragma omp parallel for collapse(2)
107  for (int j=0; j<local_width; ++j) {
108  for (int i=0; i<local_height; ++i) {
109  DataType& x = values_buffer[i+j*values_ldim];
110  const DataType g = gradient_buffer[i+j*gradient_ldim];
111  DataType& c = cache_buffer[i+j*cache_ldim];
112  c = m_decay_rate * c + (DataType(1) - m_decay_rate) * g * g;
113  x -= m_learning_rate * g / (std::sqrt(c) + m_eps);
114  }
115  }
116  } else {
117  // Update with contiguous data
118  #pragma omp parallel for
119  for (int i=0; i<local_height*local_width; ++i) {
120  DataType& x = values_buffer[i];
121  const DataType g = gradient_buffer[i];
122  DataType& c = cache_buffer[i];
123  c = m_decay_rate * c + (DataType(1) - m_decay_rate) * g * g;
124  x -= m_learning_rate * g / (std::sqrt(c) + m_eps);
125  }
126  }
127 }
