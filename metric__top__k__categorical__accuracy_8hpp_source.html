62  double compute_metric(AbsDistMat& predictions_v, AbsDistMat& ground_truth_v) override {
63  // This first computes the top k predictions within each column locally,
64  // then each column master gathers these, computes the global top k, and
65  // determines if an error was made.
66  El::Int num_errors = 0;
67  // Note: assumes structure is packed.
68  struct top_k_ele {
69  DataType val; // Predicted value.
70  DataType gt; // Ground truth.
71  };
72  const El::Int local_width = predictions_v.LocalWidth(); // minibatch dim
73  const El::Int local_height = predictions_v.LocalHeight(); // class dim
74  // Pack the top k predictions for each local column together.
75  std::vector<top_k_ele> local_top_k(m_top_k * local_width);
76  // Compute the top k entries locally.
77  std::vector<El::Int> local_indices(local_height);
78  std::iota(local_indices.begin(), local_indices.end(), 0);
79  for (El::Int mb_idx = 0; mb_idx < local_width; ++mb_idx) {
80  // Determine the top k local entries in this column.
81  std::partial_sort(
82  local_indices.begin(), local_indices.begin() + m_top_k,
83  local_indices.end(),
84  [mb_idx, &predictions_v] (El::Int a, El::Int b) -> bool {
85  return predictions_v.GetLocal(a, mb_idx) >
86  predictions_v.GetLocal(b, mb_idx); });
87  for (El::Int i = 0; i < m_top_k; ++i) {
88  El::Int idx = mb_idx*m_top_k + i;
89  local_top_k[idx].val = predictions_v.GetLocal(local_indices[i], mb_idx);
90  local_top_k[idx].gt = ground_truth_v.GetLocal(local_indices[i], mb_idx);
91  }
92  }
93  // Gather the data for each column to rank 0 within that column.
94  El::mpi::Comm col_comm = predictions_v.ColComm();
95  int col_comm_size = El::mpi::Size(col_comm);
96  if (El::mpi::Rank(col_comm) == 0) {
97  // This vector ends up being the concatenation of each local_top_k, and
98  // therefore accessing data for a single mini-batch requires computing the
99  // appropriate strides.
100  std::vector<top_k_ele> global_top_k(
101  m_top_k * local_width * col_comm_size);
102  m_comm->gather((DataType*) local_top_k.data(), 2*local_top_k.size(),
103  (DataType*) global_top_k.data(), col_comm);
104  // Compute the global top k elements in each column.
105  std::vector<El::Int> global_indices(m_top_k * col_comm_size);
106  std::iota(global_indices.begin(), global_indices.end(), 0);
107  for (El::Int mb_idx = 0; mb_idx < local_width; ++mb_idx) {
108  std::partial_sort(
109  global_indices.begin(), global_indices.begin() + m_top_k,
110  global_indices.end(),
111  [mb_idx, col_comm_size, &global_top_k, this]
112  (El::Int a, El::Int b) -> bool {
113  El::Int mb_offset = mb_idx * m_top_k;
114  El::Int a_proc_offset = (a/m_top_k) * m_top_k * col_comm_size;
115  El::Int a_idx = a_proc_offset + mb_offset + (a%m_top_k);
116  El::Int b_proc_offset = (b/m_top_k) * m_top_k * col_comm_size;
117  El::Int b_idx = b_proc_offset + mb_offset + (b%m_top_k);
118  return global_top_k[a_idx].val > global_top_k[b_idx].val;
119  });
120  // Check if there is a 1 ground truth label in the top k.
121  bool found = false;
122  for (El::Int i = 0; i < m_top_k; ++i) {
123  El::Int idx = global_indices[i];
124  idx = mb_idx*m_top_k + (i/m_top_k)*m_top_k*col_comm_size + (i%m_top_k);
125  if (global_top_k[idx].gt == DataType(1)) {
126  found = true;
127  break;
128  }
129  }
130  if (!found) {
131  ++num_errors;
132  }
133  }
134  } else {
135  m_comm->gather((DataType*) local_top_k.data(), 2*local_top_k.size(), 0,
136  col_comm);
137  }
138  return this->m_comm->model_allreduce(num_errors);
139  }

141  double report_metric(execution_mode mode) override {

144  long samples_per_epoch = stats->m_samples_per_epoch;
