55  double compute_metric(AbsDistMat& predictions_v, AbsDistMat& groundtruth_v) override {
56 
57  // Get local matrices and matrix parameters
58  const Mat& predictions_local = predictions_v.LockedMatrix();
59  const Mat& groundtruth_local = groundtruth_v.LockedMatrix();
60  const El::Int height = predictions_v.Height();
61  const El::Int width = predictions_v.Width();
62  const El::Int local_height = predictions_local.Height();
63  const El::Int local_width = predictions_local.Width();
64 
65  // Compute mean absolute deviation
66  double mad = 0.0;
67  for(El::Int col = 0; col < local_width; ++col) {
68  for(El::Int row = 0; row < local_height; ++row) {
69  const double pred_val = predictions_local(row, col);
70  const double true_val = groundtruth_local(row, col);
71  mad += std::abs(pred_val - true_val);
72  }
73  }
74  mad /= height * width;
75  mad = m_comm->allreduce(mad, predictions_v.DistComm());
76  return mad;
77 
78  }
