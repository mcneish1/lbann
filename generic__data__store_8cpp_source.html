74 void generic_data_store::get_my_indices() {
75  std::vector<int> indices;

77 
78  m_my_minibatch_indices.clear();
79  while (true) {
80  int n = m_reader->fetch_data_indices(indices);
81  if (n != (int)indices.size()) {
82  std::stringstream s2;
83  s2 << __FILE__ << " " << __LINE__ << " :: "
84  << " something is badly wrong!";
85  throw lbann_exception(s2.str());
86  }
87  for (auto t : indices) {
88  m_my_minibatch_indices.push_back(t);
89  }
90 
91  bool is_done = m_reader->update(is_active_reader, true);
92  if (!is_done) {
93  break;
94  }
95  }
