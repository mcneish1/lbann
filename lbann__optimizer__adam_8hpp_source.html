70  void update_weight_bias_matrix(ElMat& WB_D, ElMat& WB) {
71  // Update exponential decay rates.
72  cur_rho1 *= rho1;
73  cur_rho2 *= rho2;
74  // Compute the correction factor.
75  const float correction = std::sqrt(1.0f - cur_rho2) / (1.0f - cur_rho1);
76  // Update the biased first and second moments.
77  Scale(rho1, moment1_hist);
78  Axpy(1.0f - rho1, WB_D, moment1_hist);
79  Scale(rho2, moment2_hist);
80  EntrywiseMap(WB_D,
81  std::function<DataType(const DataType&)>(
82  [] (const DataType& x) { return x * x; }));
83  Axpy(1.0f - rho2, WB_D, moment2_hist);
84  // Compute the update. Use WB_D as a temporary as it's no longer needed.
85  Copy(moment2_hist, WB_D);
86  EntrywiseMap(WB_D,
87  std::function<DataType(const DataType&)>(
88  [this] (const DataType& x) { return 1.0f / std::sqrt(x + eps); }));
89  Hadamard(moment1_hist, WB_D, WB_D);
90  Axpy(-lr * correction, WB_D, WB);
91  }

97  bool saveToCheckpointShared(persist& p, int Index) {
98  char name[512];
99 
100  // current learning rate value

106  // current rho1 value

108  sprintf(name, "L%d_cur_rho1", Index);
109  p.write_float(persist_type::train, name, cur_rho1);
110  }
111 
112  // current rho2 value
113  if (p.m_rank == 0) {
114  sprintf(name, "L%d_cur_rho2", Index);
115  p.write_float(persist_type::train, name, cur_rho2);
116  }
117 
118  // checkpoint matrix for first moment
119  sprintf(name, "L%d_adam_moment1_%dx%d",
120  Index, moment1_hist.Height(), moment1_hist.Width());
121  bool rc1 = p.write_distmat(persist_type::train, name, (DistMat*)&moment1_hist);
122 
123  // checkpoint matrix for second moment
124  sprintf(name, "L%d_adam_moment2_%dx%d",
125  Index, moment2_hist.Height(), moment2_hist.Width());
126  bool rc2 = p.write_distmat(persist_type::train, name, (DistMat*)&moment2_hist);
127 
128  return (rc1 && rc2);
129  }
