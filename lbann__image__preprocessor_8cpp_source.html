61 void lbann_image_preprocessor::augment(Mat& pixels, unsigned imheight,
62  unsigned imwidth,
63  unsigned num_channels) {
64  bool do_transform = m_horizontal_flip || m_vertical_flip ||
65  m_rotation_range || m_horizontal_shift || m_vertical_shift ||
66  m_shear_range;
67  if (do_transform) {
68  cv::Mat sqpixels = cv_pixels(pixels, imheight, imwidth, num_channels);
69  rng_gen& gen = get_generator();
70  std::uniform_int_distribution<int> bool_dist(0, 1);
71  // Flips.
72  bool horiz_flip = bool_dist(gen) && m_horizontal_flip;
73  bool vert_flip = bool_dist(gen) && m_vertical_flip;
74  if (horiz_flip || vert_flip) {
75  if (horiz_flip && !vert_flip) {
76  flip(sqpixels, 1);
77  } else if (!horiz_flip && vert_flip) {
78  flip(sqpixels, 0);
79  } else {
80  flip(sqpixels, -1);
81  }
82  }
83  // Translations.
84  float x_trans = 0.0f;
85  float y_trans = 0.0f;
86  if (m_horizontal_shift) {
87  std::uniform_real_distribution<float> dist(-m_horizontal_shift,
88  m_horizontal_shift);
89  x_trans = dist(gen) * imwidth;
90  }
91  if (m_vertical_shift) {
92  std::uniform_real_distribution<float> dist(-m_vertical_shift,
93  m_vertical_shift);
94  y_trans = dist(gen) * imheight;
95  }
96  Mat trans_mat;
97  El::Diagonal(trans_mat, std::vector<DataType>({1.0f, 1.0f, 1.0f}));
98  trans_mat(0, 2) = x_trans;
99  trans_mat(1, 2) = y_trans;
100  // Shearing.
101  float shear = 0.0f;
102  if (m_shear_range) {
103  std::uniform_real_distribution<float> dist(-m_shear_range,
104  m_shear_range);
105  shear = dist(gen);
106  }
107  Mat shear_mat;
108  El::Zeros(shear_mat, 3, 3);
109  shear_mat(0, 0) = 1.0f;
110  shear_mat(2, 2) = 1.0f;
111  shear_mat(0, 1) = -std::sin(shear);
112  shear_mat(1, 1) = std::cos(shear);
113  // Rotation.
114  float rotate = 0.0f;
115  if (m_rotation_range) {
116  std::uniform_real_distribution<float> dist(-m_rotation_range,
117  m_rotation_range);
118  rotate = pi / 180.0f * dist(gen);
119  }
120  Mat rot_mat;
121  El::Zeros(rot_mat, 3, 3);
122  rot_mat(2, 2) = 1.0f;
123  rot_mat(0, 0) = std::cos(rotate);
124  rot_mat(0, 1) = -std::sin(rotate);
125  rot_mat(1, 0) = std::sin(rotate);
126  rot_mat(1, 1) = std::cos(rotate);
127  // Compute the final transformation.
128  Mat affine_mat_tmp(3, 3);
129  Mat affine_mat(3, 3);
130  El::Gemm(NORMAL, NORMAL, (DataType) 1.0, trans_mat, shear_mat,
131  (DataType) 0.0, affine_mat_tmp);
132  El::Gemm(NORMAL, NORMAL, (DataType) 1.0, affine_mat_tmp, rot_mat,
133  (DataType) 0.0, affine_mat);
134  affine_trans(sqpixels, affine_mat);
135  col_pixels(sqpixels, pixels, num_channels);
136  }
137 }

155 void lbann_image_preprocessor::mean_subtraction(
156  Mat& pixels, unsigned num_channels) {
157  const El::Int height = pixels.Height();
158  for (unsigned channel = 0; channel < num_channels; ++channel) {
159  // Compute the mean.
160  DataType mean = 0.0f;
161  for (unsigned i = channel; i < height; i += num_channels) {
162  mean += pixels(i, 0);
163  }
164  mean /= height / num_channels;
165  for (unsigned i = channel; i < height; i += num_channels) {
166  pixels(i, 0) -= mean;
167  }
168  }
169 }

171 void lbann_image_preprocessor::unit_variance(
172  Mat& pixels, unsigned num_channels) {
173  const El::Int height = pixels.Height();
174  for (unsigned channel = 0; channel < num_channels; ++channel) {
175  DataType mean = 0.0f;
176  DataType sqsum = 0.0f;
177  for (unsigned i = channel; i < height; i += num_channels) {
178  mean += pixels(i, 0);
179  sqsum += pixels(i, 0) * pixels(i, 0);
180  }
181  mean /= height / num_channels;
182  sqsum /= height / num_channels;
183  DataType std = sqsum - (mean * mean);
184  std = std::sqrt(std) + 1e-7; // Avoid division by 0.
185  for (unsigned i = channel; i < height; i += num_channels) {
186  pixels(i, 0) /= std;
187  }
188  }
189 }

192  // Pixels are in range [0, 255], normalize using that.

200 void lbann_image_preprocessor::z_score(Mat& pixels, unsigned num_channels) {
201  const El::Int height = pixels.Height();
202  for (unsigned channel = 0; channel < num_channels; ++channel) {
203  // Compute the mean and standard deviation.
204  DataType mean = 0.0f;
205  DataType sqsum = 0.0f;
206  for (unsigned i = channel; i < height; i += num_channels) {
207  mean += pixels(i, 0);
208  sqsum += pixels(i, 0) * pixels(i, 0);
209  }
210  mean /= height / num_channels;
211  sqsum /= height / num_channels;
212  DataType std = sqsum - (mean * mean);
213  std = std::sqrt(std) + 1e-7; // Avoid division by 0.
214  // Z-score is (x - mean) / std.
215  for (unsigned i = channel; i < height; i += num_channels) {
216  pixels(i, 0) = (pixels(i, 0) - mean) / std;
217  }
218  }

221 cv::Mat lbann_image_preprocessor::cv_pixels(const Mat& pixels,
222  unsigned imheight,
223  unsigned imwidth,
224  unsigned num_channels) {
225  if (num_channels == 1) {
226  cv::Mat m(imheight, imwidth, CV_32FC1);
227  for (unsigned y = 0; y < imheight; ++y) {
228  for (unsigned x = 0; x < imwidth; ++x) {
229  m.at<float>(y, x) = pixels(y * imwidth + x, 0);
230  }
231  }
232  return m;
233  } else if (num_channels == 3) {
234  cv::Mat m(imheight, imwidth, CV_32FC3);
235  for (unsigned y = 0; y < imheight; ++y) {
236  for (unsigned x = 0; x < imwidth; ++x) {
237  cv::Vec3f pixel;
238  unsigned offset = 3 * (y * imwidth + x);
239  pixel[0] = pixels(offset, 0);
240  pixel[1] = pixels(offset + 1, 0);
241  pixel[2] = pixels(offset + 2, 0);
242  m.at<cv::Vec3f>(y, x) = pixel;
243  }
244  }
245  return m;
246  } else {
247  throw lbann_exception(std::string{} + __FILE__ + " " +
248  std::to_string(__LINE__) +
249  "Only support 1 and 3 channels");
250  }

253 void lbann_image_preprocessor::col_pixels(const cv::Mat& sqpixels, Mat& pixels,
254  unsigned num_channels) {
255  unsigned imheight = sqpixels.rows;
256  unsigned imwidth = sqpixels.cols;
257  if (num_channels == 1) {
258  for (unsigned y = 0; y < imheight; ++y) {
259  for (unsigned x = 0; x < imwidth; ++x) {
260  pixels(y * imwidth + x, 0) = sqpixels.at<float>(y, x);
261  }
262  }
263  } else if (num_channels == 3) {
264  for (unsigned y = 0; y < imheight; ++y) {
265  for (unsigned x = 0; x < imwidth; ++x) {
266  cv::Vec3f pixel = sqpixels.at<cv::Vec3f>(y, x);
267  unsigned offset = 3 * (y * imwidth + x);
268  pixels(offset, 0) = pixel[0];
269  pixels(offset + 1, 0) = pixel[1];
270  pixels(offset + 2, 0) = pixel[2];
271  }
272  }
273  } else {
274  throw lbann_exception(std::string{} + __FILE__ + " " +
275  std::to_string(__LINE__) +
276  "Only support 1 and 3 channels");
277  }

286 void lbann_image_preprocessor::affine_trans(cv::Mat& sqpixels,
287  const Mat& trans) {
288  cv::Mat sqpixels_copy = sqpixels.clone();
289  // Construct the OpenCV transformation matrix.
290  cv::Mat cv_trans(2, 3, CV_32FC1);
291  cv_trans.at<float>(0, 0) = trans(0, 0);
292  cv_trans.at<float>(0, 1) = trans(0, 1);
293  cv_trans.at<float>(0, 2) = trans(0, 2);
294  cv_trans.at<float>(1, 0) = trans(1, 0);
295  cv_trans.at<float>(1, 1) = trans(1, 1);
296  cv_trans.at<float>(1, 2) = trans(1, 2);
297  cv::warpAffine(sqpixels_copy, sqpixels, cv_trans, sqpixels.size(),
298  cv::INTER_LINEAR, cv::BORDER_REPLICATE);
299 }

301 void lbann_image_preprocessor::internal_save_image(
302  Mat& pixels, const std::string filename, unsigned imheight, unsigned imwidth,
303  unsigned num_channels, bool scale) {
304  cv::Mat sqpixels = cv_pixels(pixels, imheight, imwidth, num_channels);
305  cv::Mat converted_pixels;
306  int dst_type = 0;
307  if (num_channels == 1) {
308  dst_type = CV_8UC1;
309  } else if (num_channels == 3) {
310  dst_type = CV_8UC3;
311  } // cv_pixels ensures no other case happens.
312  sqpixels.convertTo(converted_pixels, dst_type, scale ? 255.0f : 1.0f);
313  cv::imwrite(filename, converted_pixels);
314 }
