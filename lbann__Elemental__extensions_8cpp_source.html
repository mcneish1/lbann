92 template<typename F>
93 void ColumnMax( const Matrix<F>& X, Matrix<Base<F>>& norms )
94 {
95  // DEBUG_ONLY(CSE cse("ColumnMax"))
96  typedef Base<F> Real;
97  const Int m = X.Height();
98  const Int n = X.Width();
99  const F* XBuf = X.LockedBuffer();
100  const Int XLDim = X.LDim();
101 
102  norms.Resize( n, 1 );
103  Base<F>* normBuf = norms.Buffer();
104  for( Int j=0; j<n; ++j )
105  {
106  Real colMax = 0;
107  for( Int i=0; i<m; ++i )
108  colMax = Max(colMax,XBuf[i+j*XLDim]);
109  normBuf[j] = colMax;
110  }
111 }

158 template<typename F>
159 void ColumnMax
160 ( const DistSparseMatrix<F>& A, DistMultiVec<Base<F>>& norms )
161 {
162  //DEBUG_ONLY(CSE cse("ColumnMax"))
163  typedef Base<F> Real;
164  // Explicitly forming the transpose is overkill...
165  // The following would be correct but is best avoided.
166  /*
167  DistSparseMatrix<F> ATrans(A.Comm());
168  Transpose( A, ATrans );
169  RowMaxNorms( ATrans, norms );
170  */
171 
172  // Modify the communication pattern from an adjoint Multiply
173  // =========================================================
174  Zeros( norms, A.Width(), 1 );
175  // TODO: (Moon 9/12/16) I replaced the commented code since
176  // Elemental no longer has a DistSparseMultMeta class. I do not
177  // know if the code is correct.
178  // A.InitializeMultMeta();
179  // const auto& meta = A.multMeta;
180  const auto& meta = A.InitializeMultMeta();
181 
182  // Pack the send values
183  // --------------------
184  vector<Real> sendVals( meta.numRecvInds, 0 );
185  const Int numEntries = A.NumLocalEntries();
186  const F* values = A.LockedValueBuffer();
187  for( Int e=0; e<numEntries; ++e )
188  sendVals[meta.colOffs[e]] =
189  Max(sendVals[meta.colOffs[e]],values[e]);
190 
191  // Inject the updates into the network
192  // -----------------------------------
193  const Int numRecvInds = meta.sendInds.size();
194  vector<Real> recvVals( numRecvInds );
195  mpi::AllToAll
196  ( sendVals.data(), meta.recvSizes.data(), meta.recvOffs.data(),
197  recvVals.data(), meta.sendSizes.data(), meta.sendOffs.data(),
198  A.Comm() );
199 
200  // Form the maxima over all the values received
201  // --------------------------------------------
202  const Int firstLocalRow = norms.FirstLocalRow();
203  Real* normBuf = norms.Matrix().Buffer();
204  for( Int s=0; s<numRecvInds; ++s )
205  {
206  const Int i = meta.sendInds[s];
207  const Int iLoc = i - firstLocalRow;
208  normBuf[iLoc] = Max(normBuf[iLoc],recvVals[s]);
209  }
210 }
