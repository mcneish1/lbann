48 void lbann_callback_imcomm::setup(model* m) {
49  if (ct != NONE) {
50  bool add = layer_indices.size() == 0;
51  std::vector<Layer*>& layers = m->get_layers();
52  for (Layer* layer : layers) {
53  uint idx = layer->get_index();
54  if (add || layer_indices.find(idx) != layer_indices.end()) {
55  // Ensure index is present (overwrites if already there).
56  layer_indices.insert(idx);
57  // Update the layer's effective mini-batch size so it averages properly.
58  layer->set_effective_minibatch_size(
59  layer->get_minibatch_size() * m->get_comm()->get_num_models());
60  // Skip adding matrices when we don't need to.
61  if (!ct_does_quantization()) continue;
62  // TODO: handle case where weights_gradient is in other matrix distribution
63  DistMat& weights_gradient = (DistMat&) layer->get_weights_biases_gradient();
64  quantization_errors.emplace(idx, Mat{});
65  im_quantization_errors.emplace(idx, Mat{});
66  if (ct == ONEBIT_QUANTIZATION) {
67  // Set up gradient history and SGD optimizer for one-bit quantization.
68  gradhistories.emplace(idx, Mat{});
69  if (layer->optimizer != nullptr) {
70  if (typeid(*(layer->optimizer)) != typeid(Adagrad<DistMat>)) {
71  throw lbann_exception(
72  "lbann_callback_imcomm: Cannot do one-bit quantization for "
73  "layer that does not use Adagrad");
74  }
75  // TODO: This leaks the old optimizer.
76  layer->optimizer = new SGD<DistMat>(
77  layer->comm, layer->optimizer->get_learning_rate(),
78  0.0f, 0.0f, false);
79  layer->optimizer->setup(layer->m_weights->Width(), layer->m_weights->Height());
80  }
81  }
82  }
83  }
84  }

87 void lbann_callback_imcomm::on_epoch_end(model* m) {
88  lbann_comm* comm = m->get_comm();
89  if (comm->get_num_models() == 1 ||
90  m->get_execution_mode() != execution_mode::training) {
91  return; // No point with only one model.
92  }
93  if (ct_does_quantization()) {
94  std::vector<Layer*>& layers = m->get_layers();
95  for (size_t l = 0; l < layers.size(); ++l) {
96  if (layer_indices.find(layers[l]->get_index()) == layer_indices.end()) {
97  continue;
98  }
99  comm->intermodel_sum_matrix(quantization_errors[l]);
100  // TODO: handle case where weights_gradient is in other matrix distribution
101  DistMat& weights_gradient = (DistMat&) layers[l]->get_weights_biases_gradient();
102  Mat& local_mat = weights_gradient.Matrix();
103  local_mat = quantization_errors[l];
104  // Apply optimizer update again.
105  layers[l]->update();
106  quantization_errors[l].Empty();
107  }
108  }
109 }

111 void lbann_callback_imcomm::on_backward_prop_end(model* m) {
112  lbann_comm* comm = m->get_comm();
113  if (comm->get_num_models() == 1 ||
114  m->get_execution_mode() != execution_mode::training) {
115  return; // No point with only one model.
116  }
117  std::vector<Layer*>& layers = m->get_layers();
118  for (size_t l = 0; l < layers.size(); ++l) {
119  if (layer_indices.find(layers[l]->get_index()) == layer_indices.end()) {
120  continue;
121  }
122  double start_time = get_time();
123  // TODO: handle case where weights_gradient is in other matrix distribution
124  DistMat& weights_gradient = (DistMat&) layers[l]->get_weights_biases_gradient();
125  switch (ct) {
126  case NONE:
127  break;
128  case NORMAL:
129  comm->intermodel_sum_matrix(weights_gradient);
130  break;
131  case NORMAL_AR:
132  quantizer.intermodel_sum(comm, weights_gradient);
133  break;
134  case ONEBIT_QUANTIZATION:
135  quantizer.intermodel_sum_quantized(
136  comm, weights_gradient, quantization_errors[l], im_quantization_errors[l], true,
137  &(gradhistories[l]));
138  break;
139  case THRESH_QUANTIZATION:
140  // TODO: Don't hardcode thresholds.
141  quantizer.intermodel_sum_threshold_quantized(
142  comm, weights_gradient, quantization_errors[l], 0.01f, -0.01f,
143  im_quantization_errors[l], false);
144  break;
145  case COMPRESSED_THRESH_QUANTIZATION:
146  // TODO: Don't hardcode thresholds.
147  quantizer.intermodel_sum_threshold_quantized(
148  comm, weights_gradient, quantization_errors[l], 0.01f, -0.01f,
149  im_quantization_errors[l], true);
150  break;
151  case ADAPTIVE_THRESH_QUANTIZATION:
152  // TODO: Don't hardcode proportion.
153  quantizer.intermodel_sum_adaptive_threshold_quantized(
154  comm, weights_gradient, quantization_errors[l], 64,
155  im_quantization_errors[l]);
156  break;
157  /*case COMPRESSED_ADAPTIVE_THRESH_QUANTIZATION:
158  // TODO: Don't hardcode proportion.
159  quantizer.intermodel_sum_adaptive_threshold_quantized(
160  comm, weights_gradient, quantization_errors[l], 64,
161  im_quantization_errors[l], true);
162  break;*/
163  }
164  double im_time = get_time() - start_time;
165  if (summarizer != nullptr && ct != NONE) {
166  std::string prefix = "layer" + std::to_string(
167  static_cast<long long>(layers[l]->get_index())) + "/imcomm_";
168  summarizer->reduce_scalar(prefix + "time",
169  im_time, m->get_cur_step());
170  size_t bytes_sent = 0;
171  size_t bytes_received = 0;
172  if (ct_does_quantization()) {
173  bytes_sent = quantizer.get_bytes_sent();
174  bytes_received = quantizer.get_bytes_received();
175  } else {
176  // Use the same approximation the comm layer does.
177  bytes_sent = sizeof(DataType) * weights_gradient.LocalHeight() * weights_gradient.LocalWidth();
178  bytes_received = sizeof(DataType) * weights_gradient.LocalHeight() * weights_gradient.LocalWidth();
179  }
180  summarizer->reduce_scalar(prefix + "bytes_sent",
181  bytes_sent, m->get_cur_step());
182  summarizer->reduce_scalar(prefix + "bytes_received",
183  bytes_received, m->get_cur_step());
184  if (ct_does_quantization()) {
185  summarizer->reduce_scalar(prefix + "rs_bytes_sent",
186  quantizer.get_rs_bytes_sent(),
187  m->get_cur_step());
188  summarizer->reduce_scalar(prefix + "ag_bytes_sent",
189  quantizer.get_ag_bytes_sent(),
190  m->get_cur_step());
191  summarizer->reduce_scalar(prefix + "rs_bytes_received",
192  quantizer.get_rs_bytes_received(),
193  m->get_cur_step());
194  summarizer->reduce_scalar(prefix + "ag_bytes_received",
195  quantizer.get_ag_bytes_received(),
196  m->get_cur_step());
197  summarizer->reduce_scalar(prefix + "rs_send_trans_time",
198  quantizer.get_rs_send_trans_time(),
199  m->get_cur_step());
200  summarizer->reduce_scalar(prefix + "rs_recv_trans_time",
201  quantizer.get_rs_recv_trans_time(),
202  m->get_cur_step());
203  summarizer->reduce_scalar(prefix + "ag_recv_trans_time",
204  quantizer.get_ag_recv_trans_time(),
205  m->get_cur_step());
206  quantizer.reset_bytes_counters();
207  quantizer.reset_time_counters();
208  if (ct == ADAPTIVE_THRESH_QUANTIZATION) {
209  summarizer->reduce_scalar(prefix + "quantized_count",
210  quantizer.get_quantized_count(),
211  m->get_cur_step());
212  }
213  }
214  }
215  }
