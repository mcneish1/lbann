34 void lbann_callback_checknan::on_forward_prop_end(model* m, Layer* l) {
35  // Skip output layer.
36  if (l->get_index() == m->get_layers().size() - 1) {
37  return;
38  }
39  DistMat& acts = (DistMat&) l->get_activations();
40  if (!is_good(acts)) {
41  lbann_comm* comm = m->get_comm();

43  l->get_index() << " activations (step=" << m->get_cur_step() << ")" <<
44  std::endl;
45  throw lbann_exception("checknan: error in activations");
46  }

49 void lbann_callback_checknan::on_backward_prop_end(model* m, Layer* l) {
50  // Skip input/output layers.
51  if (l->get_index() == 0 || l->get_index() == m->get_layers().size() - 1) {

64 void lbann_callback_checknan::on_batch_end(model* m) {
65  std::vector<Layer*>& layers = m->get_layers();
66  // Skip input/output layers-- they don't have weights.
67  for (size_t i = 1; i < layers.size() - 1; ++i) {
68  Layer* l = layers[i];
69  DistMat& weights = (DistMat&) l->get_weights_biases();
70  if (!is_good(weights)) {
71  lbann_comm* comm = m->get_comm();
72  std::cout << "[" << comm->get_rank_in_world() << "]: error in layer " <<
73  l->get_index() << " weights (step=" << m->get_cur_step() << ")" <<
74  std::endl;
75  throw lbann_exception("checknan: error in weights");
76  }
77  }
78 }

80 bool lbann_callback_checknan::is_good(const DistMat& m) {
81  const Mat& lm = m.LockedMatrix();
82  const Int height = lm.Height();
83  const Int width = lm.Width();
84  for (Int col = 0; col < width; ++col) {
85  for (Int row = 0; row < height; ++row) {
86  const DataType val = lm(row, col);
87  if (std::isnan(val)) {
88  std::cout << "Found NaN at (" << row << ", " << col << ")!" << std::endl;
89  return false;
90  } else if (std::isinf(val)) {
91  std::cout << "Found inf (" << row << ", " << col << ")!" << std::endl;
92  return false;
93  }
94  }
95  }
96  return true;
97 }
