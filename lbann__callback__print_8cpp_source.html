52 void lbann_callback_print::on_epoch_end(model* m) {
53  lbann_comm* comm = m->get_comm();
54  if (comm->am_model_master()) {
56  for (auto&& metric : m->metrics) {
57  double train_score = metric->report_metric(execution_mode::training);
58  double validate_score = metric->report_metric(execution_mode::validation);
59  if (comm->am_world_master()) {
60  std::vector<double> train_scores(comm->get_num_models());
61  std::vector<double> validate_scores(comm->get_num_models());
62  comm->intermodel_gather(train_score, train_scores);
63  comm->intermodel_gather(validate_score, validate_scores);
64 
65  for (size_t i = 0; i < train_scores.size(); ++i) {
66  std::cout << "Model " << i;
67  std::cout << " @" << m->get_cur_step() << " steps";
68  std::cout << " Training " << _to_string(metric->type) << ": " << train_scores[i] << "%";
69  std::cout << " @" << m->get_cur_validation_step() << " validation steps Validation " << _to_string(metric->type) << ": " << validate_scores[i] << _disp_unit(metric->type);
70  std::cout << std::endl;
71  }
72  } else {
73  comm->intermodel_gather(train_score, comm->get_intermodel_master());
74  comm->intermodel_gather(validate_score, comm->get_intermodel_master());
75  }
76  }
77  for (Layer* layer : m->get_layers()) {
78  layer->epoch_print();
79  }
80  }
81 }
