171  virtual void fp_compute_cpu() {
172 
173  // Get local matrices and parameters
174  Mat& workspace_local = m_workspace_v->Matrix();
175  const Mat& prev_activations_local = this->m_prev_activations_v->LockedMatrix();
176  Mat& activations_local = this->m_activations_v->Matrix();
177  const El::Int local_height = activations_local.Height();
178  const El::Int local_width = activations_local.Width();
179 
180  // Find maximum entry in each column
181  #pragma omp parallel for
182  for(El::Int col = 0; col < local_width; ++col) {
183  DataType max_entry = prev_activations_local(0, col);
184  for(El::Int row = 1; row < local_height; ++row) {
185  max_entry = std::max(max_entry, prev_activations_local(row,col));
186  }
187  workspace_local(0, col) = max_entry;
188  }
189  m_comm->allreduce(*m_workspace_v, m_workspace_v->RedundantComm(),
190  El::mpi::MAX);
191 
192  // Exponentiate activations and compute column sums
193  // Note: Subtracting by the column max prevents activations from
194  // blowing up. Large negative values underflow to 0.
195  #pragma omp parallel for
196  for (El::Int col = 0; col < local_width; ++col) {
197  const DataType shift = workspace_local(0, col);
198  DataType sum = 0;
199  for (El::Int row = 0; row < local_height; ++row) {
200  const DataType prev_activations_entry = prev_activations_local(row, col);
201  const DataType activations_entry = std::exp(prev_activations_entry - shift);
202  activations_local(row, col) = activations_entry;
203  sum += activations_entry;
204  }
205  workspace_local(0, col) = sum;
206  }
207  m_comm->allreduce(*m_workspace_v, m_workspace_v->RedundantComm());
208 
209  // Divide activations by column sums
210  // Note: Small values are rounded to minimum output value to avoid
211  // denormalized floats.
212  El::IndexDependentMap(activations_local,
213  (std::function<DataType(El::Int,El::Int,const DataType&)>)
214  ([this,&workspace_local](El::Int r, El::Int c, const DataType& x)
215  ->DataType {
216  const DataType sum = workspace_local(0, c);
217  const DataType y = x / sum;
218 #ifdef LBANN_ENABLE_SOFTMAX_CUTOFF
219  return y > m_min_output ? y : m_min_output;
220 #else
221  return y;
222 #endif
223  }));
224  }

284  virtual void bp_compute_cpu() {
285  const Mat& activations_local = this->m_activations_v->LockedMatrix();
286  const Mat& prev_error_signal_local = this->m_prev_error_signal_v->Matrix();
287  Mat& error_signal_local = this->m_error_signal_v->Matrix();
288  Mat& workspace_local = m_workspace_v->Matrix();
289  const El::Int local_width = activations_local.Width();
290 
291  // Compute dot products
292  // Note: prev_error_signal^T activations
293  for(El::Int c=0; c<local_width; ++c) {
294  workspace_local(0, c) = El::Dot(prev_error_signal_local(El::ALL,El::IR(c)),
295  activations_local(El::ALL,El::IR(c)));
296  }
297  m_comm->allreduce(*m_workspace_v, m_workspace_v->RedundantComm());
298 
299  // Update error signal
300  // Note: error_signal := activations * (prev_error_signal - prev_error_signal^T activations)
301  El::IndexDependentFill(error_signal_local,
302  (std::function<DataType(El::Int,El::Int)>)
303  ([this,&activations_local,&prev_error_signal_local,&workspace_local]
304  (El::Int r, El::Int c)->DataType {
305  const DataType activations_entry = activations_local(r,c);
306  const DataType prev_error_signal_entry = prev_error_signal_local(r,c);
307  const DataType dot_product_entry = workspace_local(El::Int(0),c);
308 #ifdef LBANN_ENABLE_SOFTMAX_CUTOFF
309  if(activations_entry > m_min_output) {
310  return activations_entry * (prev_error_signal_entry
311  - dot_product_entry);
312  }
313  else {
314  return DataType(0);
315  }
316 #else
317  return activations_entry * (prev_error_signal_entry
318  - dot_product_entry);
319 #endif
320  }));
321 
322  }
