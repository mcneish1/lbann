388 uint lbann::sequential_model::add(const std::string layer_name,
389  const int layer_dim,
390  const activation_type activation,
391  const weight_initialization init,
392  std::vector<regularizer*> regularizers)
393 {
394  const int layer_index = m_layers.size();
395  Optimizer *optimizer = optimizer_fac->create_optimizer();
396 
397  // Get properties of previous layer
398  int prev_layer_dim = -1;
399  int prev_layer_index = -1;
400  if(m_layers.size() != 0) {
401  Layer* prev_layer = m_layers.back();
402  prev_layer_dim = prev_layer->NumNeurons;
403  prev_layer_index = prev_layer->Index;
404  }
405 
406  if (comm->am_model_master()) {
407  std::cout << "Adding a layer with input " << prev_layer_dim
408  << " and index " << layer_index
409  << " prev layer index " << prev_layer_index << std::endl;
410  }
411 
412  if(layer_name.compare("FullyConnected") == 0) {
413  Layer* new_layer
414  = layer_fac->create_layer<FullyConnectedLayer>("FullyConnected",
415  layer_index,
416  prev_layer_dim,
417  layer_dim,
418  m_mini_batch_size,
419  activation, init,
420  comm,
421  optimizer,
422  regularizers);
423  m_layers.push_back(new_layer);
424  } else if(layer_name.compare("Softmax") == 0) {
425  Layer* new_layer
426  = layer_fac->create_layer<SoftmaxLayer>("Softmax",
427  layer_index,
428  prev_layer_dim,
429  layer_dim,
430  m_mini_batch_size,
431  init,
432  comm,
433  optimizer);
434  m_layers.push_back(new_layer);
435  } else {
436  std::cout << "Unknown layer type " << layer_name << std::endl;
437  }
438 
439  return layer_index;
440 }
