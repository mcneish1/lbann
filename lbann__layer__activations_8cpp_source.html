149 DataType ELU_layer::elu(const DataType& z, DataType alpha) {
150  if (z > 0) {
151  return z;
152  } else {
153  return alpha * (std::exp(z) - 1);
154  }
155 }

157 DataType ELU_layer::eluPrime(const DataType& z, DataType alpha) {
158  if (z > 0) {
159  return 1.0f;
160  } else {
161  // elu(z, alpha) + alpha
162  return alpha * (std::exp(z) - 1) + alpha;
163  }
164 }

167 // There are three mechanisms for updating all fields of a distributed matrix
168 // 1) EntrywiseMap + independent reset of the bias row - Fastest (about ~30% faster)
169 // 2) IndexDependentMap (conditional application of function to mask bias row) - Slow
170 // IndexDependentMap(m,
171 // (std::function<double(int,int,double)>)[m](int r, int c, double z)->
172 // double{int bias_row = m.Height();
173 // if(r == bias_row - 1){return 1.0;}else{return sigmoid(z);}
174 // });
175 // 3) Dual nested loop over local matrix indices with global matrix lookups - Slow
