32 bool lbann::image_utils::loadIMG(const std::string& Imagefile, int& Width, int& Height, bool Flip, unsigned char *&Pixels) {
33 #ifdef __LIB_OPENCV

36  return false;

38 
39  Width = image.cols;
40  Height = image.rows;
41 
42  for (int y = 0; y < Height; y++) {
43  for (int x = 0; x < Width; x++) {
44  cv::Vec3b pixel = image.at<cv::Vec3b>(y, x);
45  int offset = (Flip) ? ((Height - 1 - y) * Width + x) : (y * Width + x);
46  Pixels[offset] = pixel[_LBANN_CV_BLUE_];
47  Pixels[offset + Height*Width] = pixel[_LBANN_CV_GREEN_];
48  Pixels[offset + 2*Height*Width] = pixel[_LBANN_CV_RED_];
49  }

179 bool lbann::image_utils::load_image(const std::string& filename,
180  int& Width, int& Height, int& Type, cv_process_patches& pp, std::vector<::Mat>& data) {
181 #ifdef __LIB_OPENCV

183 
184  std::vector<cv::Mat> patches;
185  bool ok = !image.empty() && pp.preprocess(image, patches);
186  if ((patches.size() == 0u) || (patches.size() != data.size())) {
187  return false;

189

191  ok = cv_utils::copy_cvMat_to_buf(patches[i], data[i], pp);

193 
194  // Disabling normalizer is needed because normalizer is not necessarily
195  // called during preprocessing but implicitly applied during data copying to
196  // reduce overhead.
197  pp.disable_normalizer();
198

200 
201  Width = patches[0].cols;
202  Height = patches[0].rows;
203  Type = patches[0].type();
