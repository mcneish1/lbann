121  void setup_dims() override {
122 
123  // Initialize previous layer dimensions with first parent layer
124  transform::setup_dims();
125 
126  // Check if concatenation axis is valid
127  if(m_concatenation_axis < 0
128  || m_concatenation_axis >= this->m_num_neuron_dims) {
129  throw lbann_exception("concatenation_layer: invalid concatenation axis");
130  }
131 
132  // Get concatenation axis indices corresponding to each parent layer
133  m_concatenation_points.empty();
134  m_concatenation_points.push_back(0);
135  m_concatenation_points.push_back(this->m_neuron_dims[m_concatenation_axis]);
136  for(size_t i=1; i<this->m_parent_layers.size(); ++i) {
137 
138  // Get parent layer dimensions
139  std::vector<int> parent_dims = this->m_parent_layers[i]->fp_output_dims(this);
140 
141  // Check if parent layer has valid dimensions
142  if((int) parent_dims.size() != this->m_num_neuron_dims) {
143  throw lbann_exception("concatenation_layer: parent layer has invalid number of dimensions");
144  }
145  for(size_t d=0; d<parent_dims.size(); ++d) {
146  if((int) d != m_concatenation_axis
147  && this->m_neuron_dims[d] != parent_dims[d]) {
148  throw lbann_exception("concatenation_layer: parent layer has invalid dimensions");
149  }
150  }
151 
152  // Get concatentation axis upper bound for parent layer
153  m_concatenation_points.push_back(m_concatenation_points.back()
154  + parent_dims[m_concatenation_axis]);
155 
156  }
157 
158  // Update neuron dimensions
159  this->m_neuron_dims[m_concatenation_axis] = m_concatenation_points.back();
160  this->m_num_neurons = std::accumulate(m_neuron_dims.begin(),
161  m_neuron_dims.end(),
162  1,
163  std::multiplies<int>());
164 
165  }

168  transform::setup_gpu();
169  #ifndef __LIB_CUDNN
170  throw lbann_exception("concatenation_layer: cuDNN not detected");
171  #else
172 
173  // Copy backward propagation output from GPUs if a parent layer is
174  // not using GPU implementation
175  for(size_t i=1; i<this->m_parent_layers.size(); ++i) {
176  if(!this->m_parent_layers[i]->using_gpus()) {
177  m_copy_bp_output_from_gpus = true;
178  }
179  }
180 
181  // Allocate workspace if needed
182  if(m_copy_bp_output_from_gpus) {
183  int max_slice_dim = 0;
184  for(size_t parent_index = 1;
185  parent_index < this->m_parent_layers.size();
186  ++parent_index) {
187  if(!this->m_parent_layers[parent_index]->using_gpus()) {
188  max_slice_dim = std::max(max_slice_dim,
189  m_concatenation_points[parent_index+1]
190  - m_concatenation_points[parent_index]);
191  }
192  }
193  int max_slice_size = (this->m_num_neurons
194  / this->m_neuron_dims[m_concatenation_axis]
195  * max_slice_dim);
196  size_t required_work_space = (max_slice_size
197  * this->m_mini_batch_size_per_gpu
198  * sizeof(DataType));
199  for(int i=0; i<this->m_cudnn->get_num_gpus(); ++i) {
200  if(required_work_space > this->m_cudnn->get_work_space_size(i)) {
201  this->m_cudnn->set_work_space_size(i, required_work_space);

235  void fp_compute_gpu() {
236  #ifndef __LIB_CUDNN
237  throw lbann_exception("concatenation_layer: cuDNN not detected");
238  #else
239 
240  // Split the activations tensor into slices of width 1 along the
241  // concatenation axis
242  const int num_slices
243  = std::accumulate(this->m_neuron_dims.begin(),
244  this->m_neuron_dims.begin() + m_concatenation_axis,
245  1,
246  std::multiplies<int>());
247  const int slice_unit_size
248  = std::accumulate(this->m_neuron_dims.begin() + m_concatenation_axis + 1,
249  this->m_neuron_dims.end(),
250  1,
251  std::multiplies<int>());
252  const int output_slice_dim = this->m_neuron_dims[m_concatenation_axis];
253  const int output_slice_size = output_slice_dim * slice_unit_size;
254 
255  // Copy entries in each child to error signal tensor
256  const int num_gpus = this->m_cudnn->get_num_gpus();
257  for(size_t parent_index = 0;
258  parent_index < this->m_parent_layers.size();
259  ++parent_index) {
260  const Layer* parent = this->m_parent_layers[parent_index];
261 
262  // Get child error signal on GPUs
263  std::vector<DataType*> input;
264  if(parent_index == 0) {
265  input = this->m_prev_activations_dv;
266  }
267  else {
269  std::vector<DataType*> work_spaces;
270  for(int i=0; i<num_gpus; ++i) {
271  work_spaces.push_back((DataType*)this->m_cudnn->get_work_space(i));
272  }
273  if(parent->using_gpus()) {
274  parent->get_gpu_fp_output(input, work_spaces, this);
275  }
276  else {
277  parent->get_fp_output(*this->m_prev_activations_v, this);
278  this->m_cudnn->scatter_to_gpus(work_spaces,
279  this->m_prev_activations_v->LockedMatrix(),
280  this->m_mini_batch_size_per_gpu);
281  input = work_spaces;
282  }
283  }
284 
285  // Split previous error signal tensor into slices
286  const int input_slice_dim = m_concatenation_points[parent_index+1] - m_concatenation_points[parent_index];
287  const int input_slice_size = input_slice_dim * slice_unit_size;
288  const int input_size = num_slices * input_slice_size;
289  const int slice_offset = m_concatenation_points[parent_index] * slice_unit_size;
290 
291  // Copy slices from previous error signal tensor into error signal tensor
292  for(int slice = 0; slice < num_slices; ++slice) {
293  std::vector<DataType*> input_slice(num_gpus), output_slice(num_gpus);
294  for(int i = 0; i < num_gpus; ++i) {
295  input_slice[i] = input[i] + slice * input_slice_size;
296  output_slice[i] = this->m_activations_d[i] + slice * output_slice_size + slice_offset;
297  }
298  this->m_cudnn->copy_on_gpus(output_slice,
299  input_slice,
300  input_slice_size,
301  this->m_mini_batch_size_per_gpu,
302  input_size,
303  this->m_num_neurons);
304  }
305 
306  }
307 
308  #endif // #ifndef __LIB_CUDNN

311  void fp_compute_cpu() {
312 
313  // Split the neuron tensor into slices of width 1 along the
314  // concatenation axis
315  const int num_slices
316  = std::accumulate(this->m_neuron_dims.begin(),
317  this->m_neuron_dims.begin() + m_concatenation_axis,
318  1,
319  std::multiplies<int>());
320  const int slice_unit_size
321  = std::accumulate(this->m_neuron_dims.begin() + m_concatenation_axis + 1,
322  this->m_neuron_dims.end(),
323  1,
324  std::multiplies<int>());
325  const int output_slice_dim = this->m_neuron_dims[m_concatenation_axis];
326  const int output_slice_size = output_slice_dim * slice_unit_size;
327 
328  // Copy entries in each parent to neuron tensor
329  for(size_t i = 0; i < this->m_parent_layers.size(); ++i) {
330 
331  // Split previous neuron tensor into slices
332  this->m_parent_layers[i]->get_fp_output(*this->m_prev_activations_v, this);
333  const int input_slice_dim = m_concatenation_points[i+1] - m_concatenation_points[i];
334  const int input_slice_size = input_slice_dim * slice_unit_size;
335  const int slice_offset_start = m_concatenation_points[i] * slice_unit_size;
336  const int slice_offset_end = m_concatenation_points[i+1] * slice_unit_size;
337 
338  // Copy slices from previous neuron tensor into neuron tensor
339  for(int slice = 0; slice < num_slices; ++slice) {
340  El::LockedView(*m_input_slice_v,
341  *this->m_prev_activations_v,
342  El::IR(slice * input_slice_size,
343  (slice+1) * input_slice_size),
344  El::ALL);
345  El::View(*m_output_slice_v,
346  *this->m_activations_v,
347  El::IR(slice * output_slice_size + slice_offset_start,
348  slice * output_slice_size + slice_offset_end),
349  El::ALL);
350  El::Copy(*m_input_slice_v, *m_output_slice_v);
351  }
352 
353  }

357  void get_bp_output(AbsDistMat& bp_output, const Layer* prev_layer) const override {
358 
359  // Check if input is in the list of parent layers
360  const int parent_index = (std::find(this->m_parent_layers.begin(),
361  this->m_parent_layers.end(),
362  prev_layer)
363  - this->m_parent_layers.begin());
364  if(parent_index >= (int) this->m_parent_layers.size()) {
365  transform::get_bp_output(bp_output, prev_layer);
366  }
367 
368  // Split the error signal tensor into slices of width 1 along the
369  // concatenation axis
370  const int num_slices
371  = std::accumulate(this->m_neuron_dims.begin(),
372  this->m_neuron_dims.begin() + m_concatenation_axis,
373  1,
374  std::multiplies<int>());
375  const int slice_unit_size
376  = std::accumulate(this->m_neuron_dims.begin() + m_concatenation_axis + 1,
377  this->m_neuron_dims.end(),
378  1,
379  std::multiplies<int>());
380  const int input_slice_dim = this->m_neuron_dims[m_concatenation_axis];
381  const int output_slice_dim = m_concatenation_points[parent_index+1] - m_concatenation_points[parent_index];
382  const int input_slice_size = input_slice_dim * slice_unit_size;
383  const int output_slice_size = output_slice_dim * slice_unit_size;
384  const int slice_offset_start = m_concatenation_points[parent_index] * slice_unit_size;
385  const int slice_offset_end = m_concatenation_points[parent_index+1] * slice_unit_size;
386 
387  if(num_slices == 1
388  && m_error_signal_v->DistData() == bp_output.DistData()) {
389  // Return view of error signal slice
390  El::LockedView(bp_output,
391  *this->m_error_signal_v,
392  El::IR(slice_offset_start, slice_offset_end),
393  El::ALL);
394  }
395  else {
396  // Copy slices from error signal tensor into output
397  bp_output.Empty(false);
398  bp_output.Resize(output_slice_size, m_error_signal_v->Width());
399  AbsDistMat* output_slice_v
400  = bp_output.Construct(bp_output.Grid(), bp_output.Root());
401  for(int slice = 0; slice < num_slices; ++slice) {
402  El::LockedView(*m_input_slice_v,
403  *this->m_error_signal_v,
404  El::IR(slice * input_slice_size + slice_offset_start,
405  slice * input_slice_size + slice_offset_end),
406  El::ALL);
407  El::View(*output_slice_v,
408  bp_output,
409  El::IR(slice * output_slice_size,
410  (slice+1) * output_slice_size),
411  El::ALL);
412  El::Copy(*m_input_slice_v, *output_slice_v);
413  }
414  delete output_slice_v;
415  }
