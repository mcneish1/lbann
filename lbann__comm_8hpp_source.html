332  template <typename T>
333  void broadcast(T* data, int count, std::vector<int>& dests, int root) {
334  mpi::Group bcast_group;
335  mpi::Comm bcast_comm;
336  std::vector<int> ranks;
337  ranks.push_back(root);
338  ranks.insert(ranks.end(), dests.begin(), dests.end());
339  create_group(ranks, bcast_group);
340  // Elemental doesn't expose this, so we have to reach into its internals.
341  // This lets us create a communicator without involving all of COMM_WORLD.
342  // Use a tag of 0; should not matter unless we're multi-threaded.
343  MPI_Comm_create_group(mpi::COMM_WORLD.comm, bcast_group.group, 0,
344  &(bcast_comm.comm));
345  int translated_root = mpi::Translate(mpi::COMM_WORLD, root, bcast_comm);
346  mpi::Broadcast(data, count, translated_root, bcast_comm);
347  mpi::Free(bcast_comm);
348  mpi::Free(bcast_group);
349  }

373  static inline bool is_sendable(const Mat& mat) {
374  // This assumes we do not transmit mat with a datatype smaller than
