89 bool lbann::image_utils::loadBMP(const char* Imagefile, int& Width, int& Height, int& BPP, bool Flip, unsigned char*& Pixels)
90 {
91  FILE *infile = fopen(Imagefile, "rb");
92  if (infile == NULL) {
93  fprintf(stderr, "can't open %s\n", Imagefile);
94  return false;
95  }
96 
97  // Read Bitmap File Header
98  BMP_FILEHEADER header;
99  fread(&header, sizeof(BMP_FILEHEADER), 1, infile);
100  if (header.bfType != BMP_HEADER_MARKER) {
101  fclose(infile);
102  return false;
103  }
104 
105  // Read Bitmap Info
106  int bisize = header.bfOffBits - sizeof(BMP_FILEHEADER);
107  BMP_INFO* info = (BMP_INFO*)malloc(bisize);
108  fread(info, bisize, 1, infile);
109 
110  // Check Palette Count
111  if (info->bmiHeader.biClrUsed != 0 || info->bmiHeader.biBitCount != 24) {
112  free(info);
113  fclose(infile);
114  return false;
115  }
116 
117  // Read DIB Bits
118  int bitrowsize = ((info->bmiHeader.biWidth * info->bmiHeader.biBitCount + 31) / 32) * 4;
119  int bitsize = bitrowsize * info->bmiHeader.biHeight;
120  unsigned char* bits = (unsigned char*)malloc(bitsize);
121  fread(bits, bitsize, 1, infile);
122 
123  // Set Pixels
124  Width = info->bmiHeader.biWidth;
125  Height = info->bmiHeader.biHeight;
126  BPP = 3;
127  Pixels = new unsigned char[Width * Height * BPP];
128  for (int y = 0; y < Height; y++) {
129  for (int x = 0; x < Width; x++) {
130  int offset = (Flip) ? ((Height - 1 - y) * Width + x) : (y * Width + x);
131  Pixels[offset * 3 + 0] = bits[y * bitrowsize + x * 3 + 2];
132  Pixels[offset * 3 + 1] = bits[y * bitrowsize + x * 3 + 1];
133  Pixels[offset * 3 + 2] = bits[y * bitrowsize + x * 3 + 0];
134  }
135  }
136 
137  free(info);
138  free(bits);
139  fclose(infile);
140 
141  return true;
142 }

154 bool lbann::image_utils::loadPGM(const char* Imagefile, int& Width, int& Height, int& BPP, bool Flip, unsigned char*& Pixels)
155 {
156  FILE *infile = fopen(Imagefile, "rb");
157  if (infile == NULL) {
158  fprintf(stderr, "can't open %s\n", Imagefile);
159  return false;
160  }
161 
162  char format[5];
163  fscanf(infile, "%s", format);
164  int width, height;
165  fscanf(infile, "%d%d", &width, &height);
166  int maxpixel;
167  fscanf(infile, "%d", &maxpixel);
168 
169  Width = width;
170  Height = height;
171  BPP = 1;
172  Pixels = new unsigned char[Width * Height * BPP];
173 
174  for (int y = 0; y < Height; y++) {
175  for (int x = 0; x < Width; x++) {
176  int offset = (Flip) ? ((Height - 1 - y) * Width + x) : (y * Width + x);
177  int pixel = fgetc(infile);
178  Pixels[offset] = (unsigned char)((double)pixel / maxpixel * 255.0);
179  }
180  }
181 
182  fclose(infile);
183  return true;
184 }
