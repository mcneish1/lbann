82 bool lbann::distributed_minibatch::is_data_set_processed(generic_data_reader *data_reader) {
83  // not just the ones in the last round. This will ensure that all readers, that had data
84  // will have distributed it.
85  int num_parallel_readers = data_reader->get_num_parallel_readers();
86  int num_iterations_per_epoch = data_reader->get_num_iterations_per_epoch();
87  int current_step_in_epoch = data_reader->get_current_step_in_epoch(); // Get the current step before the update function increments it
88 
89  bool is_active_reader = (m_comm->get_rank_in_model() < num_parallel_readers)
90  && ((m_comm->get_rank_in_model()+1)%num_parallel_readers == m_root);
91 
92  if(is_active_reader) {
93  if(m_local_data_valid) {
94  std::stringstream err;
95  err << __FILE__ << " "<< __LINE__
96  << " :: lbann_input_layer_distributed_minibatch: all valid data was not processed.";
97  throw lbann_exception(err.str());
98  }
99  }
100  m_local_reader_done = !(*update_data_reader_fn)(is_active_reader, data_reader);
101 
103  if(current_step_in_epoch == (num_iterations_per_epoch - 1)) {
104  m_local_reader_done = false;
105  m_root = 0;
106  m_num_data_per_epoch = 0;
109  return true;
110  } else {
111  return false;
112  }
113 }

118 int lbann::distributed_minibatch::compute_max_num_parallel_readers(long data_set_size, int mini_batch_size, int requested_num_parallel_readers) const {
119  int num_parallel_readers = requested_num_parallel_readers;
120 
123  if(m_comm->get_model_grid().Size() < num_parallel_readers) {
124  if (m_comm->am_model_master()) {
125  std::cout << "Warning the grid size " << m_comm->get_model_grid().Size()
126  << "is smaller than the number of requested parallel readers "
127  << num_parallel_readers << "." << std::endl;
128  }
129  num_parallel_readers = m_comm->get_model_grid().Size();
130  }
131 
133  if(data_set_size != 0) {
134  int max_num_parallel_readers = num_parallel_readers;
135  while(ceil((float)data_set_size / (float)(mini_batch_size * m_comm->get_num_models())) < max_num_parallel_readers) {
136  max_num_parallel_readers--;
137  }
138  if(max_num_parallel_readers != num_parallel_readers) {
139  std::cout << "Warning the training data set size " << data_set_size
140  << " is too small for the number of requested parallel readers "
141  << num_parallel_readers << ", using " << max_num_parallel_readers << "."
142  << std::endl;
143  }
144  return max_num_parallel_readers;
145  } else {
146  return 0;
147  }
148 }

150 void lbann::distributed_minibatch::calculate_num_iterations_per_epoch(int num_models, int model_rank, int max_mini_batch_size, generic_data_reader *data_reader) {
151  if(data_reader == nullptr) { return; }
152  // If the data reader does not have any data bail out (e.g. unused validation reader)
153  if(data_reader->get_use_percent() == double(0.0)) { return; }
154 
155  if(max_mini_batch_size > data_reader->get_num_data()) {
156  max_mini_batch_size = data_reader->get_num_data();
157  }
158 
160  int num_parallel_readers_per_model = compute_max_num_parallel_readers(data_reader->get_num_data(), max_mini_batch_size, m_requested_max_num_parallel_readers);
161  data_reader->set_num_parallel_readers(num_parallel_readers_per_model);
162  if(num_parallel_readers_per_model == 0) {
163  throw lbann_exception(
164  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
165  " :: distributed_minibatch: number of parallel readers is zero");
166  }
167 
169  int batch_stride = num_models * num_parallel_readers_per_model * max_mini_batch_size;
170  int base_offset = m_comm->get_rank_in_model() * num_models * max_mini_batch_size;
171  int model_offset = model_rank * max_mini_batch_size;
173  data_reader->set_mini_batch_size(max_mini_batch_size);
174  data_reader->set_stride_to_next_mini_batch(batch_stride);
175  data_reader->set_sample_stride(1);
176  data_reader->set_iteration_stride(num_parallel_readers_per_model);
177  data_reader->set_reset_mini_batch_index(m_comm->get_rank_in_model());
179  data_reader->set_base_offset(base_offset);
180  data_reader->set_model_offset(model_offset);
181  data_reader->set_initial_position();
182 
183  int min_stride_across_models = max_mini_batch_size * num_models;
184 
185  data_reader->set_global_mini_batch_size(min_stride_across_models);
186  data_reader->set_last_mini_batch_size(max_mini_batch_size);
187  data_reader->set_global_last_mini_batch_size(min_stride_across_models);
188 
189  int num_whole_mini_batches_per_model = floor(data_reader->get_num_data() / min_stride_across_models);
190  int num_whole_mini_batches_per_reader = floor(num_whole_mini_batches_per_model / num_parallel_readers_per_model);
191  int parallel_readers_with_extra_mini_batch = num_whole_mini_batches_per_model % num_parallel_readers_per_model;
192  int global_partial_mini_batch_size = data_reader->get_num_data() - (num_whole_mini_batches_per_model * min_stride_across_models);
193  int per_model_partial_mini_batch_size = global_partial_mini_batch_size / num_models;
194  int world_master_remainder_data = 0;
195 
196  // Compute how many full "parallel" mini-batches are available
197  int last_mini_batch_threshold = num_whole_mini_batches_per_model * min_stride_across_models;
198 
199  // BVE FIXME revisit this piece of code
200  if(m_comm->get_rank_in_model() < parallel_readers_with_extra_mini_batch) {
201  num_whole_mini_batches_per_reader += 1;
202  data_reader->set_last_mini_batch_size(max_mini_batch_size);
203  }
204 
205  int world_master_remainder_adjustment = data_reader->get_num_data()
206  - (num_whole_mini_batches_per_model * min_stride_across_models)
207  - (per_model_partial_mini_batch_size * num_models);
208  if(m_comm->am_world_master()) {
209  world_master_remainder_data = world_master_remainder_adjustment;
210  world_master_remainder_adjustment = 0;
211  }
212  per_model_partial_mini_batch_size += world_master_remainder_data;
213 
215  if(per_model_partial_mini_batch_size > 0) {
216  data_reader->set_last_mini_batch_size(per_model_partial_mini_batch_size);
217  data_reader->set_global_last_mini_batch_size(global_partial_mini_batch_size);
218  }
219 
220  // BVE FIXME this is wonky
221  if(global_partial_mini_batch_size != 0) {
222  data_reader->set_num_iterations_per_epoch(num_whole_mini_batches_per_model+1);
223  }else {
224  data_reader->set_num_iterations_per_epoch(num_whole_mini_batches_per_model);
225  }
226 
227  if(data_reader->get_last_mini_batch_size() > max_mini_batch_size) {
228  throw new lbann_exception("Error in calculating the partial mini-batch size, exceeds the max mini-batch size");
229  }
230 
233 
235  data_reader->set_stride_to_last_mini_batch(data_reader->get_stride_to_next_mini_batch());
236 
237  int last_mini_batch_offset = std::max(0, num_whole_mini_batches_per_reader - 1) * data_reader->get_stride_to_next_mini_batch();
238 
241  if(m_comm->get_rank_in_model() == parallel_readers_with_extra_mini_batch && per_model_partial_mini_batch_size > 0) {
242  data_reader->set_stride_to_last_mini_batch((last_mini_batch_threshold - data_reader->get_base_offset() - data_reader->get_model_offset() - last_mini_batch_offset)
243  + model_rank * per_model_partial_mini_batch_size + world_master_remainder_adjustment);
244  }
245 
246  //cout << "[" << m_comm->get_rank_in_world() << "] " << model_rank << " model rank, "<< m_comm->get_rank_in_model() << " rank in model, num_whole_mini_batches_per_model " << num_whole_mini_batches_per_model << " num_whole_mini_batches_per_reader " << num_whole_mini_batches_per_reader << "(m_num_mini_batches_per_reader=" << data_reader->get_num_mini_batches_per_reader() << ") parallel_readers_with_extra_mini_batch " << parallel_readers_with_extra_mini_batch << " partial_mini_batch_size=" << per_model_partial_mini_batch_size << " last mini bath size=" << data_reader->get_last_mini_batch_size() << " world_master_remainder_data=" << world_master_remainder_data << " threshold " << data_reader->get_last_mini_batch_threshold() << " with a last stride of " << data_reader->get_stride_to_last_mini_batch() << " and stride of " << data_reader->get_stride_to_next_mini_batch() << " and there are " << num_parallel_readers_per_model << " parallel readers per model" << " last mini batch offset = " << last_mini_batch_offset << " parallel reader with extra minibatch = " << parallel_readers_with_extra_mini_batch << " model bracket = " << (parallel_readers_with_extra_mini_batch * max_mini_batch_size + per_model_partial_mini_batch_size + world_master_remainder_data) <<" base ofset "<< data_reader->get_base_offset() << " model offset " << data_reader->get_model_offset() <<endl;
247 
248  return;
249 }
