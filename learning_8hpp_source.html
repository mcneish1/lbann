55  if (m_l2_regularization_factor != DataType(0)) {
56 
57  // Get local weight data
58  const DataType *weights_buffer = m_weights->LockedBuffer();
59  const int weights_ldim = m_weights->LDim();
60  const int local_height = m_weights->LocalHeight();
61  const int local_width = m_weights->LocalWidth();
62 
63  // Compute sum of squares
64  DataType sum = 0;
65  const int block_size = std::max((int) (64 / sizeof(DataType)), 1);
66  #pragma omp parallel for reduction(+:sum) collapse(2)
67  for(int col = 0; col < local_width; ++col) {
68  for(int block_start = 0; block_start < local_height; block_start += block_size) {
69  double block_sum = 0;
70  const int block_end = std::min(block_start + block_size, local_height);
71  for(int row = block_start; row < block_end; ++row) {
72  const DataType x = weights_buffer[row + col * weights_ldim];
73  block_sum += x * x;
74  }
75  sum += block_sum;
76  }
77  }
78  sum = m_comm->allreduce(sum, m_weights->DistComm());
79 
80  // Add regularization term to objective function
81  const DataType regularization_term = m_l2_regularization_factor * sum / 2;
82  this->m_neural_network_model->m_obj_fn->add_to_value(regularization_term);

98  virtual void group_lasso_regularize_objective_function() {
99  if (m_group_lasso_regularization_factor != DataType(0)) {
100 
101  // Get local weight data
102  const DataType *weights_buffer = m_weights->LockedBuffer();
103  const int weights_ldim = m_weights->LDim();
104  const int width = m_weights->Width();
105  const int local_height = m_weights->LocalHeight();
106  const int local_width = m_weights->LocalWidth();
107 
108  AbsDistMat *workspace;
109 
110  if (get_data_layout() == data_layout::MODEL_PARALLEL) {
111  workspace = new StarMRMat(this->m_comm->get_model_grid());
112  } else {
113  workspace = new StarVCMat(this->m_comm->get_model_grid());
114  }
115  workspace->Resize(1, width);
116  Mat& local_workspace = workspace->Matrix();
117 
118  // Compute sum of squares with Kahan summation
119  for (int col = 0; col < local_width; ++col) {
120  DataType sum = 0;
121  DataType correction = 0;
122  for (int row = 0; row < local_height; ++row) {
123  const DataType x = weights_buffer[row + col * weights_ldim];
124  const DataType term = x * x + correction;
125  const double next_sum = sum + term;
126  correction = term - (next_sum - sum);
127  sum = next_sum;
128  }
129  local_workspace(0, col) = sum;
130  }
131  m_comm->allreduce(*workspace, workspace->RedundantComm());
132 
133  // Add regularization term to objective function
134  DataType sumColL2Norms = 0;
135  for (int i = 0; i < local_width; i++) sumColL2Norms += std::sqrt(local_workspace(0, i));
136  m_comm->allreduce(sumColL2Norms, workspace->DistComm());
137  const DataType regularization_term = m_group_lasso_regularization_factor * sumColL2Norms;
138  this->m_neural_network_model->m_obj_fn->add_to_value(regularization_term);
139 
140  delete workspace;
141  }
142  }

145  virtual void group_lasso_regularize_gradient() {
146  if (m_group_lasso_regularization_factor != DataType(0)) {
147  //Group lasso gradient will be m_weights where each column c of m_weights is normalized by $||c||_{2}$.
148  //So first we compute $L_2$ norm of each column of m_weights as in group_lasso_regularize_objective_function.
149  const DataType *weights_buffer = m_weights->LockedBuffer();
150  const int weights_ldim = m_weights->LDim();
151  const int width = m_weights->Width();
152  const int local_height = m_weights->LocalHeight();
153  const int local_width = m_weights->LocalWidth();
154 
155  AbsDistMat *workspace;
156 
157  if (get_data_layout() == data_layout::MODEL_PARALLEL) {
158  workspace = new StarMRMat(this->m_comm->get_model_grid());
159  } else {
160  workspace = new StarVCMat(this->m_comm->get_model_grid());
161  }
162  workspace->Resize(1, width);
163  Mat& local_workspace = workspace->Matrix();
164 
165  // Compute sum of squares with Kahan summation
166  for (int col = 0; col < local_width; ++col) {
167  DataType sum = 0;
168  DataType correction = 0;
169  for (int row = 0; row < local_height; ++row) {
170  const DataType x = weights_buffer[row + col * weights_ldim];
171  const DataType term = x * x + correction;
172  const double next_sum = sum + term;
173  correction = term - (next_sum - sum);
174  sum = next_sum;
175  }
176  local_workspace(0, col) = sum;
177  }
178 
179  m_comm->allreduce(*workspace, workspace->RedundantComm());
180  El::EntrywiseMap(*workspace, std::function<DataType(const DataType&)>(
181  [](const DataType& x) {
182  return std::sqrt(x);
183  }));
184 
185  //update m_weights_graident using L2 norms of columns of weight matrix (in workspace.)
186  Mat& m_weights_gradient_local = m_weights_gradient->Matrix();
187  Mat& m_weights_local = m_weights->Matrix();
188 
189  El::IndexDependentMap(m_weights_gradient_local,
190  (std::function<DataType(El::Int,El::Int,const DataType&)>)
191  ([&m_weights_local, &local_workspace](El::Int r, El::Int c, const DataType& x)
192  -> DataType {
193  const DataType colL2Norm = local_workspace(0, c);
194  if (colL2Norm != DataType(0)) {
195  return x + m_weights_local(r, c)/colL2Norm;
196  } else {
197  return x;
198  };
199  }));
200  delete workspace;
201  }
202  }
