44 void dropout::fp_activations() {
45  // Terminate early if dropout is disabled
46  if (m_layer->m_execution_mode != execution_mode::training
47  || m_keep_prob < 0.0f) return;
48 
49  // Get local activations
50  ElMat* acts = m_layer->m_activations;
51  const Int local_height = acts->LocalHeight();
52  const Int local_width = acts->LocalWidth();
53  const Int global_height = acts->Height();
54 
55 #ifdef LBANN_PROCDET_DROPOUT
56  bernoulli_fill_procdet(m_cur_mask, acts->Height(), acts->Width(),
57  m_keep_prob);
58  m_cur_mask *= 1.0 / m_keep_prob;
59  if (acts->GlobalRow(local_height - 1) == global_height - 1) {
60  for (Int col = 0; col < local_width; ++col) {
61  m_cur_mask.SetLocal(local_height - 1, col, 1.0f);
62  }
63  }
64  Hadamard(*acts, m_cur_mask, *acts);
65 #else
66  Mat local_acts = acts->Matrix();
67 
68  // Construct dropout mask
69  // Note: Construct Bernoulli matrix and scale by
70  // 1/m_keep_prob. Entries corresponding to bias row are set to 1
71  // to ensure that mask doesn't affect bias row. This
72  // implementation assumes 'acts' is in MC,MR; Star,VC; Star,VR; or
73  // similar format.
74  Bernoulli(m_cur_mask, local_height, local_width, m_keep_prob);
75  m_cur_mask *= 1.0 / m_keep_prob;
76  if (acts->GlobalRow(local_height - 1) == global_height - 1) {
77  for (Int col = 0; col < local_width; ++col) {
78  m_cur_mask.Set(local_height - 1, col, 1.0f);
79  }
80  }
81  // Apply dropout mask to local activations
82  Hadamard(local_acts, m_cur_mask, local_acts);
83 #endif // LBANN_PROCDET_DROPOUT
84 }

86 void dropout::bp_activations() {
87  // Terminate early if dropout is disabled
88  if (m_layer->m_execution_mode != execution_mode::training
89  || m_keep_prob < 0.0f) return;
90 
91 #ifdef LBANN_PROCDET_DROPOUT
92  Hadamard(*(m_layer->m_prev_error_signal), m_cur_mask, *(m_layer->m_prev_error_signal));
93 #else
94  // Re-weight the incoming loss using dropout mask
95  Mat local_prev_error_signal = m_layer->m_prev_error_signal->Matrix();
96  Hadamard(local_prev_error_signal, m_cur_mask, local_prev_error_signal);
97 #endif // LBANN_PROCDET_DROPOUT
98 }
