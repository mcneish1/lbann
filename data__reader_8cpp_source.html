58 int lbann::generic_data_reader::fetch_data(Mat& X) {
59  int nthreads = omp_get_max_threads();
60  if(!position_valid()) {
61  throw lbann_exception(
62  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
63  + " :: generic data reader load error: !position_valid"
64  + " -- current pos = " + std::to_string(m_current_pos)
65  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices");
66  }
67 
70  #pragma omp parallel for schedule(static, 1)
71  for (int t = 0; t < nthreads; t++) {
72  preprocess_data_source(omp_get_thread_num());
73  }
74 
75  int loaded_batch_size = get_loaded_mini_batch_size();
76  const int end_pos = std::min(static_cast<size_t>(m_current_pos+loaded_batch_size),
77  m_shuffled_indices.size());
78  const int mb_size = std::min(
79  El::Int{((end_pos - m_current_pos) + m_sample_stride - 1) / m_sample_stride},
80  X.Width());
81 
82  El::Zeros(X, X.Height(), X.Width());
83  El::Zeros(m_indices_fetched_per_mb, mb_size, 1);
84  #pragma omp parallel for
85  for (int s = 0; s < mb_size; s++) {
86  // Catch exceptions within the OpenMP thread.
87  try {
88  int n = m_current_pos + (s * m_sample_stride);
89  int index = m_shuffled_indices[n];
90  bool valid = fetch_datum(X, index, s, omp_get_thread_num());
91  if (!valid) {
92  throw lbann_exception(
93  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
94  " :: generic data reader load error: datum not valid");
95  }
96  m_indices_fetched_per_mb.Set(s, 0, index);
97  } catch (lbann_exception& e) {
98  lbann_report_exception(e);
99  } catch (std::exception& e) {
100  El::ReportException(e);
101  }
102  }
103 
106  #pragma omp parallel for schedule(static, 1)
107  for (int t = 0; t < nthreads; t++) {
108  postprocess_data_source(omp_get_thread_num());
109  }
110 
111  return mb_size;
112 }

198 bool generic_data_reader::update(bool is_active_reader) {
199  bool reader_not_done = true; // BVE The sense of this should be fixed
200  m_current_mini_batch_idx++;
201 
202  if(is_active_reader) {
203  m_current_pos = get_next_position();
204 
206  El::Zeros(m_indices_fetched_per_mb, m_indices_fetched_per_mb.Width(), 1);
207 
208  m_loaded_mini_batch_idx += m_iteration_stride;
209  }
210  if (m_loaded_mini_batch_idx >= m_num_iterations_per_epoch) {
211  reader_not_done = false;
212  }
213  if (m_current_mini_batch_idx == m_num_iterations_per_epoch) {
214  if ((get_rank() < m_num_parallel_readers) && (m_current_pos < (int)m_shuffled_indices.size())) {
215  throw lbann_exception(
216  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
217  + " :: generic data reader update error: the epoch is complete,"
218  + " but not all of the data has been used -- current pos = " + std::to_string(m_current_pos)
219  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices");
220  }
221  if (m_shuffle) {
222  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(),
223  get_data_seq_generator());
224  }
225  set_initial_position();
226  }
227  return reader_not_done;
228 }

265 void generic_data_reader::select_subset_of_data() {
266  if(m_shuffle) {
267  std::shuffle(m_shuffled_indices.begin(), m_shuffled_indices.end(), get_data_seq_generator());
268  }
269 
270  size_t count = get_absolute_sample_count();
271  double use_percent = get_use_percent();
272  if (count == 0 and use_percent == 0.0) {
273  throw lbann_exception(
274  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
275  " :: generic_data_reader::select_subset_of_data() get_use_percent() "
276  + "and get_absolute_sample_count() are both zero; exactly one "
277  + "must be zero");
278  }
279  if (!(count == 0 or use_percent == 0.0)) {
280  throw lbann_exception(
281  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
282  " :: generic_data_reader::select_subset_of_data() get_use_percent() "
283  "and get_absolute_sample_count() are both non-zero; exactly one "
284  "must be zero");
285  }
286 
287 
288  if (count != 0) {
289  if(count > static_cast<size_t>(get_num_data())) {
290  throw lbann_exception(
291  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
292  " :: generic_data_reader::select_subset_of_data() - absolute_sample_count=" +
293  std::to_string(count) + " is > get_num_data=" +
294  std::to_string(get_num_data()));
295  }
296  m_shuffled_indices.resize(get_absolute_sample_count());
297  }
298 
299  if (use_percent) {
300  m_shuffled_indices.resize(get_use_percent()*get_num_data());
301  }
302 
303  long unused = get_validation_percent()*get_num_data(); //get_num_data() = m_shuffled_indices.size()
304  long use_me = get_num_data() - unused;
305  if (unused > 0) {
306  m_unused_indices=std::vector<int>(m_shuffled_indices.begin() + use_me, m_shuffled_indices.end());
307  m_shuffled_indices.resize(use_me);
308  }
309 
310  if(!m_shuffle) {
311  std::sort(m_shuffled_indices.begin(), m_shuffled_indices.end());
312  std::sort(m_unused_indices.begin(), m_unused_indices.end());
313  }
314 }

323 bool generic_data_reader::saveToCheckpointShared(persist& p, const char *name) const {
324  // rank 0 writes the training state file
325  if (p.get_rank() == 0) {
326  char fieldname[1024];
327 
328  // record minibatch index
329  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
330  p.write_uint64(persist_type::train, fieldname, (uint64_t) m_current_mini_batch_idx);
331 
332  // get size of list of training examples
333  int size = m_shuffled_indices.size();
334 
335  // record size of ShuffleIndices
336  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
337  p.write_uint64(persist_type::train, fieldname, (uint64_t) size);
338 
339  // TODO: each model may have a different position, need to gather and write these
340  // record current position within training data
341  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
342  p.write_uint64(persist_type::train, fieldname, (uint64_t) m_current_pos);
343 
344  // write list of indices
345  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
346  p.write_int32_contig(persist_type::train, fieldname, &m_shuffled_indices[0], (uint64_t) size);
347  }
348 
349  return true;

353 bool lbann::generic_data_reader::loadFromCheckpointShared(persist& p, const char *name) {
354  // rank 0 reads the training state file
355  if (p.get_rank() == 0) {
356  char fieldname[1024];
357 
358  // record minibatch index
359  uint64_t val;
360  snprintf(fieldname, sizeof(fieldname), "%s_current_mini_batch_idx", name);
361  p.read_uint64(persist_type::train, fieldname, &val);
362  m_current_mini_batch_idx = (int) val;
363 
364  // get size of ShuffleIndices
365  snprintf(fieldname, sizeof(fieldname), "%s_data_size", name);
366  p.read_uint64(persist_type::train, fieldname, &val);
367  int size = (int) val;
368 
369  // get current position within data
370  snprintf(fieldname, sizeof(fieldname), "%s_data_position", name);
371  p.read_uint64(persist_type::train, fieldname, &val);
372  m_current_pos = (int) val;
373 
374  // resize shuffled index array to hold values
375  m_shuffled_indices.resize(size);
376 
377  // read list of indices
378  snprintf(fieldname, sizeof(fieldname), "%s_data_indices", name);
379  p.read_int32_contig(persist_type::train, fieldname, &m_shuffled_indices[0], (uint64_t) size);
380  }
381 
382  // broadcast minibatch index
383  MPI_Bcast(&m_current_mini_batch_idx, 1, MPI_INT, 0, MPI_COMM_WORLD);
384 
385  // TODO: with multiple readers, make this a scatter
386  // broadcast current position
387  MPI_Bcast(&m_current_pos, 1, MPI_INT, 0, MPI_COMM_WORLD);
388 
389  // broadcast values from rank 0
390  int size = m_shuffled_indices.size();
391  MPI_Bcast(&size, 1, MPI_INT, 0, MPI_COMM_WORLD);
392 
393  // resize shuffled index array to hold values
394  if (p.get_rank() != 0) {
395  m_shuffled_indices.resize(size);
396  }
397 
398  // broadcast index array
399  MPI_Bcast(&m_shuffled_indices[0], size, MPI_INT, 0, MPI_COMM_WORLD);
400 
401  return true;
402 }
