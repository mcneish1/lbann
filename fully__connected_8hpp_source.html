175  fully_connected_layer& operator=(const fully_connected_layer& other) {
176  learning::operator=(other);
177  m_bias_scaling_factor = other.m_bias_scaling_factor;
178 
179  // Copy matrices
180  #define COPY_MATRIX(src, dst) \
181  do { \
182  if(src != nullptr && dst != nullptr) { \
183  El::Copy(*src, *dst); \
184  } \
185  if(src != nullptr && dst == nullptr) { \
186  dst = src->Copy(); \
187  } \
188  if(src == nullptr && dst != nullptr) { \
189  delete dst; \
190  dst = nullptr; \
191  } \
192  } while(false)
193  COPY_MATRIX(other.m_matrix_weights_v, m_matrix_weights_v);
194  COPY_MATRIX(other.m_bias_weights_v, m_bias_weights_v);
195  COPY_MATRIX(other.m_matrix_weights_gradient, m_matrix_weights_gradient);
196  COPY_MATRIX(other.m_bias_weights_gradient, m_bias_weights_gradient);
197  #undef COPY_MATRIX
198  setup_views(); // Update views.
199  #ifdef __LIB_CUDNN
200  if (m_cudnn != nullptr) {
201  m_cudnn->deallocate_on_gpus(m_matrix_weights_gradient_d);
202  m_cudnn->deallocate_on_gpus(m_bias_weights_gradient_d);
203  m_matrix_weights_gradient_d = m_cudnn->copy(other.m_matrix_weights_gradient_d,
204  this->m_matrix_weights_gradient->Height(),
205  this->m_matrix_weights_gradient->Width());
206  m_bias_weights_gradient_d = m_cudnn->copy(other.m_bias_weights_gradient_d,
207  this->m_bias_weights_gradient->Height(),
208  this->m_bias_weights_gradient->Width());
209  cudnn::copy_tensor_cudnn_desc(other.m_bias_weights_desc,
210  m_bias_weights_desc);
211  cudnn::copy_tensor_cudnn_desc(other.m_activations_desc,
212  m_activations_desc);
213  }
214  #endif // __LIB_CUDNN
215 
216  return *this;
217  }

264  void setup_data() override {
265  learning::setup_data();
266 
267  // Initialize default weights if none are provided
268  if (this->m_weights.size() > 2) {
269  std::stringstream err;
270  err << __FILE__ << " " << __LINE__ << " :: "
271  << "attempted to setup " << m_name << " with an invalid number of weights";
272  throw lbann_exception(err.str());
273  }
274  this->m_weights.resize(2, nullptr);
275  if (this->m_weights[0] == nullptr) {
276  this->m_weights[0] = new weights(this->m_comm, this->m_cudnn);
277  this->m_weights[0]->set_name(this->m_name + "_matrix_weights");
278  this->m_weights[0]->set_initializer(new he_normal_initializer(this->m_comm));
279  this->m_weights[0]->set_optimizer(m_neural_network_model->create_optimizer());
280  this->m_neural_network_model->add_weights(this->m_weights[0]);
281  }
282  if (this->m_weights[1] == nullptr) {
283  this->m_weights[1] = new weights(this->m_comm, this->m_cudnn);
284  this->m_weights[1]->set_name(this->m_name + "_bias_weights");
285  this->m_weights[1]->set_optimizer(m_neural_network_model->create_optimizer());
286  this->m_neural_network_model->add_weights(this->m_weights[1]);
287  }
288 
289  // Initialize Glorot or He weight initialization
290  fan_in_fan_out_initializer* cast_initializer
291  = dynamic_cast<fan_in_fan_out_initializer*>(&this->m_weights[0]->get_initializer());
292  if (cast_initializer != nullptr) {
293  cast_initializer->set_fan_in(this->m_num_prev_neurons);
294  cast_initializer->set_fan_out(this->m_num_neurons);
295  }
296 
297  // Setup weights
298  if (this->get_data_layout() == data_layout::MODEL_PARALLEL) {
299  this->m_weights[0]->setup(this->m_num_neurons,
300  this->m_num_prev_neurons,
301  El::MC, El::MR);
302  this->m_weights[1]->setup(this->m_num_neurons, 1,
303  El::MC, El::STAR);
304  }
305  else if (this->get_data_layout() == data_layout::DATA_PARALLEL) {
306  this->m_weights[0]->setup(this->m_num_neurons,
307  this->m_num_prev_neurons,
308  El::STAR, El::STAR);
309  this->m_weights[1]->setup(this->m_num_neurons, 1,
310  El::STAR, El::STAR);
311  }
312  else {
313  std::stringstream err;
314  err << __FILE__ << " " << __LINE__ << " :: "
315  << "attempted to setup " << m_name << " with an invalid data layout";
316  throw lbann_exception(err.str());
317  }
318 
319  // Setup weight gradients
320  El::Zeros(*this->m_matrix_weights_gradient,
321  this->m_weights[0]->get_height(),
322  this->m_weights[0]->get_width());
323  El::Zeros(*this->m_bias_weights_gradient,
324  this->m_weights[1]->get_height(),
325  this->m_weights[1]->get_width());
326 
327  }

335  void setup_gpu() override {
336  learning::setup_gpu();
337 #ifndef __LIB_CUDNN
338  throw lbann_exception("fully_connected_layer: CUDA not detected");
339 #else
340 
341  // Allocate GPU memory
342  this->m_cudnn->allocate_on_gpus(this->m_matrix_weights_gradient_d,
343  this->m_matrix_weights_gradient->Height(),
344  this->m_matrix_weights_gradient->Width());
345  if(m_bias_scaling_factor != DataType(0)) {
346  this->m_cudnn->allocate_on_gpus(this->m_bias_weights_gradient_d,
347  this->m_bias_weights_gradient->Height(),
348  this->m_bias_weights_gradient->Width());
349  }
350 
351  // CUDNN setup
352  // NOTE: Setting tensor dimensions as (1, 1, X, Y), where X is the
353  // mini batch size, and bias dimensions as (1, 1, 1, Y) does not
354  // work. Calls to cudnnAddTensor return
355  // CUDNN_STATUS_NOT_SUPPORTED. Setting X as the dimension of C
356  // works, though they should be mathematically the same.
357  FORCE_CHECK_CUDNN(cudnnCreateTensorDescriptor(&m_bias_weights_desc));
358  FORCE_CHECK_CUDNN(cudnnSetTensor4dDescriptor(m_bias_weights_desc,
359  CUDNN_TENSOR_NCHW,
360  cudnn::get_cudnn_data_type(),
361  1, 1, 1,
362  m_bias_weights_v->Height()));
363  FORCE_CHECK_CUDNN(cudnnCreateTensorDescriptor(&m_activations_desc));
364  FORCE_CHECK_CUDNN(cudnnSetTensor4dDescriptor(m_activations_desc,
365  CUDNN_TENSOR_NCHW,
366  cudnn::get_cudnn_data_type(),
367  1, m_mini_batch_size_per_gpu, 1,
368  m_bias_weights_v->Height()));
369 
370 #endif // __LIB_CUDNN

382  void fp_compute_cpu() {
383  // Apply weight matrix
384  m_weights[0]->get_values_view(*m_matrix_weights_v);
385  fp_compute_weights<device::CPU>();
386 
387  // Apply bias if needed
388  if(m_bias_scaling_factor != DataType(0)) {
389  m_weights[1]->get_values_view(*m_bias_weights_v);
390  const Mat& local_bias_weights = m_bias_weights_v->LockedMatrix();
391  El::IndexDependentMap(this->m_activations_v->Matrix(),
392  (std::function<DataType(El::Int,El::Int,const DataType&)>)
393  ([this,&local_bias_weights](El::Int r, El::Int c,const DataType& z)
394  ->DataType {
395  return z + m_bias_scaling_factor * local_bias_weights(r, 0);
396  }));
397  }
398  }

400  void fp_compute_cuda() {
401 #ifndef __LIB_CUDNN
402  throw lbann_exception("fully_connected: CUDA not detected");
403 #else
404  // Apply weight matrix
405  fp_compute_weights<device::CUDA>();
406 
407  // Apply bias if needed
408  if(m_bias_scaling_factor != DataType(0)) {
409  std::vector<DataType*> bias_weights_d = m_weights[1]->get_values_gpu();
410  const int num_gpus = this->m_cudnn->get_num_gpus();
411  for (int i = 0; i < num_gpus; ++i) {
412  CHECK_CUDA(cudaSetDevice(this->m_cudnn->get_gpu(i)));
413 #if 1
414  const DataType one = 1;
415  CHECK_CUDNN(cudnnSetStream(this->m_cudnn->get_handle(i),
416  this->m_cudnn->get_stream(i)));
417  FORCE_CHECK_CUDNN(cudnnAddTensor(this->m_cudnn->get_handle(i),
418  &m_bias_scaling_factor,
419  m_bias_weights_desc,
420  bias_weights_d[i],
421  &one,
422  m_activations_desc,
423  m_activations_d[i]));
424 #else
425  fully_connected_cuda::add_tensor(m_bias_scaling_factor,
426  bias_weights_d[i],
427  m_bias_weights_v->Height(), 1,
428  DataType(1), m_activations_d[i],
429  this->m_activations_v->Height(),
430  this->m_mini_batch_size_per_gpu);
431 #endif
432  }
433  }
434 #ifdef LBANN_DEBUG
435  this->m_cudnn->check_error();
436 #endif
437 #endif

448  void bp_compute_cpu() {
449  // Compute the error signal and gradients.
450  bp_compute_weights<device::CPU>();
451 
452  // Compute bias update if needed
453  optimizer* bias_optimizer = this->m_weights[1]->get_optimizer();
454  if(m_bias_scaling_factor != DataType(0) && bias_optimizer != nullptr) {
455  El::RowSum(this->m_prev_error_signal_v->LockedMatrix(),
456  m_bias_weights_gradient->Matrix());
457  El::Scale(m_bias_scaling_factor /
458  this->m_neural_network_model->get_effective_mini_batch_size(),
459  *m_bias_weights_gradient);
460  bias_optimizer->allreduce_and_add_to_gradient(*m_bias_weights_gradient);
461  }
462 
463  }

551 template<> template<device dev> inline void
552 fully_connected_layer<data_layout::MODEL_PARALLEL>::bp_compute_weights() {
553  // Compute the partial delta update for the next lower layer
554  El::Gemm(El::TRANSPOSE, El::NORMAL, DataType(1),
555  *this->m_matrix_weights_v,
556  *this->m_prev_error_signal_v,
557  DataType(0),
558  *this->m_error_signal_v);
559 
560  // Compute update for activation weights
561  optimizer* matrix_optimizer = this->m_weights[0]->get_optimizer();
562  if (matrix_optimizer != nullptr) {
563  El::Gemm(El::NORMAL, El::TRANSPOSE, DataType(1)/
564  this->m_neural_network_model->get_effective_mini_batch_size(),
565  *this->m_prev_error_signal_v,
566  *this->m_prev_activations_v,
567  DataType(0),
568  *m_matrix_weights_gradient);
569  matrix_optimizer->add_to_gradient(*m_matrix_weights_gradient);
570  }
571 }
