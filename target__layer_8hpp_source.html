69  void setup_dims() override {
70  io_layer::setup_dims();
71  if (this->is_for_regression()) {
72  this->m_neuron_dims = get_data_dims();
73  this->m_num_neuron_dims = this->m_neuron_dims.size();
74  this->m_num_neurons = std::accumulate(this->m_neuron_dims.begin(),
75  this->m_neuron_dims.end(),
76  1,
77  std::multiplies<int>());
78  } else {
79  this->m_num_neurons = get_linearized_label_size();

81  this->m_neuron_dims.assign(1, this->m_num_neurons);
