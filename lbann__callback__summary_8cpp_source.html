39 void lbann_callback_summary::on_batch_end(model* m) {
40  m->summarize(*summarizer);
41  // Note that these comm stats are a running sum, so they count from the last
42  // time we reset and thus are over the whole batch_interval period.
43  // Bytes sent/received are the sum of the bytes sent/received by every rank
44  // in a model.
45  lbann_comm* comm = m->get_comm();
46  size_t bytes_sent = comm->get_bytes_sent();
47  size_t bytes_received = comm->get_bytes_received();
48  size_t model_barriers = comm->get_num_model_barriers();
49  size_t intermodel_barriers = comm->get_num_intermodel_barriers();
50  size_t global_barriers = comm->get_num_global_barriers();
51  comm->reset_stats_counters();
52  summarizer->sum_reduce_scalar("bytes_sent", bytes_sent, m->get_cur_step());
53  summarizer->sum_reduce_scalar("bytes_received", bytes_received,
54  m->get_cur_step());
55  summarizer->reduce_scalar("model_barriers", model_barriers,
56  m->get_cur_step());
57  summarizer->reduce_scalar("intermodel_barriers", intermodel_barriers,
58  m->get_cur_step());
59  summarizer->reduce_scalar("global_barriers", global_barriers,
60  m->get_cur_step());
61 }

63 void lbann_callback_summary::on_epoch_end(model* m) {
64  for (auto&& metric : m->metrics) {
65  double train_score = metric->report_metric(execution_mode::training);
66  string phase = "train_";
67  phase += _to_string(metric->type);
68  summarizer->reduce_scalar(phase, train_score, m->get_cur_step());
69  }
70  for (const auto& layer : m->get_layers()) {
71  std::string prefix = "layer" + std::to_string(layer->get_index()) + "/";
72  summarizer->reduce_histogram(prefix + "WB", layer->get_weights_biases(),
73  m->get_cur_step());
74  }
75  summarizer->flush();
76 }

78 void lbann_callback_summary::on_test_end(model* m) {
79  lbann_comm* comm = m->get_comm();
80  for (auto&& metric : m->metrics) {
81  double test_score = metric->report_metric(execution_mode::testing);
82  string phase = "test_";
83  phase += _to_string(metric->type);
84  summarizer->reduce_scalar(phase, test_score, m->get_cur_step());
85  }
86  // Reset counters incremented during test phase.
87  comm->reset_stats_counters();
88  for (auto&& layer : m->get_layers()) {
89  layer->reset_counters();
90  }
91 }
