69 void loss_function::setup(objective_function& obj_fn) {
70  objective_function_term::setup(obj_fn);
71 
72  // Check that loss function has one target layer and no weights
73  // Note: if target layer is not specified, choose the latest target
74  // layer in the model.
75  if (m_layers.size() > 1) {
76  std::stringstream err;
77  err << __FILE__ << " " << __LINE__ << " :: "
78  << "attempted to setup loss function with invalid number of target layers "
79  << "(expected 1, found " << m_layers.size() << ")";
80  throw lbann_exception(err.str());
81  }
82  if (m_layers.empty()) {
83  std::vector<Layer*> layers
84  = this->m_objective_function->get_model()->get_layers();
85  for (int i = layers.size() - 1; i >= 0; --i) {
86  if (dynamic_cast<target_layer*>(layers[i]) != nullptr) {
87  m_layers.push_back(layers[i]);
88  break;
89  }
90  }
91  }
92  if (m_layers.size() != 1
93  || dynamic_cast<target_layer*>(m_layers[0]) == nullptr) {
94  std::stringstream err;
95  err << __FILE__ << " " << __LINE__ << " :: "
96  << "could not setup loss function with target layer";
97  throw lbann_exception(err.str());
98  }
99  if (m_weights.size() != 0) {
100  std::stringstream err;
101  err << __FILE__ << " " << __LINE__ << " :: "
102  << "attempted to setup loss function with invalid number of weights "
103  << "(expected 0, found " << m_weights.size() << ")";
104  throw lbann_exception(err.str());
105  }
106 
107  // Initialize gradient matrix
108  auto *target = (target_layer*) m_layers[0];
109  const AbsDistMat& ground_truth = target->get_activations();
110  m_gradient = ground_truth.Construct(ground_truth.Grid(),
111  ground_truth.Root());
112  El::Zeros(*m_gradient, ground_truth.Height(), ground_truth.Width());
113 
114 }

125 void loss_function::compute_gradient() {
126  if (m_scale_factor == DataType(0)) { return; }
127  auto *target = (target_layer*) m_layers[0];
128  const AbsDistMat& prediction = target->get_prediction();
129  const AbsDistMat& ground_truth = target->get_ground_truth();
130  El::Zeros(*m_gradient, prediction.Height(), prediction.Width());
131  differentiate(prediction, ground_truth, *m_gradient);
132  target->add_to_error_signal(*m_gradient, m_scale_factor);
133 }
