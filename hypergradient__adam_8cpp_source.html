139 void hypergradient_adam::step_compute(AbsDistMat& values,
140  const AbsDistMat& gradient) {
141 
142  // Precompute the bias correction.
143  m_current_beta1 *= m_beta1;
144  m_current_beta2 *= m_beta2;
145  const DataType correction = std::sqrt(DataType(1) - m_current_beta2) /
146  (DataType(1) - m_current_beta1);
147 
148  // Get local matrix data
149  const int local_height = values.LocalHeight();
150  const int local_width = values.LocalWidth();
151  DataType* __restrict__ values_buffer = values.Buffer();
152  const int values_ldim = values.LDim();
153  const DataType* __restrict__ gradient_buffer = gradient.LockedBuffer();
154  const int gradient_ldim = gradient.LDim();
155  DataType* __restrict__ moment1_buffer = m_moment1->Buffer();
156  const int moment1_ldim = m_moment1->LDim();
157  DataType* __restrict__ moment2_buffer = m_moment2->Buffer();
158  const int moment2_ldim = m_moment2->LDim();
159  DataType* __restrict__ old_gradient_buffer = m_old_gradient->Buffer();
160  const int old_gradient_ldim = m_old_gradient->LDim();
161 
162  // Compute the learning rate update.
163  DataType lr_update = El::Dot(gradient, *m_old_gradient);
164  m_learning_rate += m_hyper_learning_rate * lr_update;
165 
166  // Check if matrix data is contiguous.
167  if (values_ldim != local_height
168  || gradient_ldim != local_height
169  || moment1_ldim != local_height
170  || moment2_ldim != local_height
171  || old_gradient_ldim != local_height) {
172  // Non-contiguous data.
173  #pragma omp parallel for collapse(2)
174  for (int j = 0; j < local_width; ++j) {
175  for (int i = 0; i < local_height; ++i) {
176  DataType& x = values_buffer[i+j*values_ldim];
177  const DataType g = gradient_buffer[i+j*gradient_ldim] + m_eps;
178  DataType& m1 = moment1_buffer[i+j*moment1_ldim];
179  DataType& m2 = moment2_buffer[i+j*moment2_ldim];
180  DataType& old_c = old_gradient_buffer[i+j*old_gradient_ldim];
181  m1 = m_beta1 * m1 + (DataType(1) - m_beta1) * g;
182  m2 = m_beta2 * m2 + (DataType(1) - m_beta2) * g * g;
183  old_c = correction * m1 / (std::sqrt(m2) + m_eps);
184  x -= m_learning_rate * old_c;
185  }
186  }
187  } else {
188  // Contiguous data.
189  #pragma omp parallel for
190  for (int i = 0; i < local_height * local_width; ++i) {
191  DataType& x = values_buffer[i];
192  // Add eps here to avoid denormalized floats.
193  const DataType g = gradient_buffer[i] + m_eps;
194  DataType& m1 = moment1_buffer[i];
195  DataType& m2 = moment2_buffer[i];
196  DataType& old_c = old_gradient_buffer[i];
197  // Update the first/second moment estimates.
198  m1 = m_beta1 * m1 + (DataType(1) - m_beta1) * g;
199  m2 = m_beta2 * m2 + (DataType(1) - m_beta2) * g * g;
200  // Compute the unbiased gradient estimate.
201  old_c = correction * m1 / (std::sqrt(m2) + m_eps);
202  // Parameter update.
203  x -= m_learning_rate * old_c;
204  }
205  }
206 }
