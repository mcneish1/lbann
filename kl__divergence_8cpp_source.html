59 EvalType kl_divergence::evaluate() {
60 
61  //Get matrices of input (latent space) layers
62  AbsDistMat& z_mean_acts = m_z_mean_layer->get_activations();
63  AbsDistMat& z_log_sigma_acts = m_z_log_sigma_layer->get_activations();
64 
65  AbsDistMat* z_mean = z_mean_acts.Construct(z_mean_acts.Grid(),
66  z_mean_acts.Root());
67  AbsDistMat* z_log1 = z_log_sigma_acts.Construct(z_log_sigma_acts.Grid(),
68  z_log_sigma_acts.Root());
69 
70  DataType mean_value = DataType(0);
71  DataType std_value = DataType(0);
72 
73  auto sq = [&](const DataType& x) {
74  return (x*x);
75  };
76  auto expn = [&](const DataType& x) {
77  return std::exp(x);
78  };
79  auto addone = [&](const DataType& x) {
80  return (1+x);
81  };
82 
83  //Local copies
84  El::Copy(z_mean_acts, *z_mean);
85  El::Copy(z_log_sigma_acts, *z_log1);
86  AbsDistMat* z_log2 = z_log1->Copy();
87 
88  //Compute entrywise add, square and exponent of the matrices
89  El::EntrywiseMap(*z_log1,El::MakeFunction(addone));
90  El::EntrywiseMap(*z_mean,El::MakeFunction(sq));
91  El::EntrywiseMap(*z_log2,El::MakeFunction(expn));
92 
93  //Entrywise substraction of matrices
94  El::Axpy(DataType(-1),*z_mean,*z_log1);
95  El::Axpy(DataType(-1),*z_log2,*z_log1);
96 
97  //Entrywise mean of all the previous computation (stats)
98  entrywise_mean_and_stdev(*z_log1,mean_value,std_value);
99 
100  delete z_mean;
101  delete z_log1;
102 
103  return (0.5 * mean_value);
104 }

106 void kl_divergence::differentiate() {
107  const AbsDistMat& z_mean_acts = m_z_mean_layer->get_activations();
108  const AbsDistMat& z_log_sigma_acts = m_z_log_sigma_layer->get_activations();
109  auto height= z_mean_acts.Height();
110 
111  AbsDistMat* z_mean_gradient = z_mean_acts.Construct(z_mean_acts.Grid(),
112  z_mean_acts.Root());
113 
114  AbsDistMat* z_log_sigma_gradient = z_log_sigma_acts.Construct(z_log_sigma_acts.Grid(),
115  z_log_sigma_acts.Root());
116  El::Copy(z_mean_acts,*z_mean_gradient);
117  El::Copy(z_log_sigma_acts,*z_log_sigma_gradient);
118 
119  const DataType scale = DataType(1) / DataType(2)*height;
120 
121  //Compute z_log_sigma_gradient = -(1/2n)(1-exp(Z_logsigma))
122  auto expn = [&](const DataType& x) {
123  return (1 - std::exp(x));
124  };
125  El::EntrywiseMap(*z_log_sigma_gradient,El::MakeFunction(expn));
126 
127  //add gradients to respective error signal
128  //z_mean_gradient = (1/2n)z_mean
129  m_z_mean_layer->add_to_error_signal(*z_mean_gradient, (DataType(1)/height));
130  m_z_log_sigma_layer->add_to_error_signal(*z_log_sigma_gradient,DataType(-scale));
131 
132  delete z_mean_gradient;
133  delete z_log_sigma_gradient;
134 }
