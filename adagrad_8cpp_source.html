78 void adagrad::step_compute(AbsDistMat& values, const AbsDistMat& gradient) {
79 
80  // Get local matrix data
81  const int local_height = values.LocalHeight();
82  const int local_width = values.LocalWidth();
83  DataType* __restrict__ values_buffer = values.Buffer();
84  const int values_ldim = values.LDim();
85  const DataType* __restrict__ gradient_buffer = gradient.LockedBuffer();
86  const int gradient_ldim = gradient.LDim();
87  DataType* __restrict__ cache_buffer = m_cache->Buffer();
88  const int cache_ldim = m_cache->LDim();
89 
90  // Check if matrix data is contiguous
91  if (values_ldim != local_height
92  || gradient_ldim != local_height
93  || cache_ldim != local_height) {
94  // Update with non-contiguous data
95  #pragma omp parallel for collapse(2)
96  for (int j=0; j<local_width; ++j) {
97  for (int i=0; i<local_height; ++i) {
98  DataType& x = values_buffer[i+j*values_ldim];
99  const DataType g = gradient_buffer[i+j*gradient_ldim];
100  DataType& c = cache_buffer[i+j*cache_ldim];
101  c += g * g;
102  x -= m_learning_rate * g / (std::sqrt(c) + m_eps);
103  }
104  }
105  } else {
106  // Update with contiguous data
107  #pragma omp parallel for
108  for (int i=0; i<local_height*local_width; ++i) {
109  DataType& x = values_buffer[i];
110  const DataType g = gradient_buffer[i];
111  DataType& c = cache_buffer[i];
112  c += g * g;
113  x -= m_learning_rate * g / (std::sqrt(c) + m_eps);
114  }
115  }
116 }
