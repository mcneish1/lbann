68 int lbann::DataReader_ImageNet::fetch_data(Mat& X)
69 {
70  static bool testme = true;
71  if(!DataReader::position_valid()) {
72  stringstream err;
73  err << __FILE__<<" "<<__LINE__<< " :: Imagenet data reader load error: !position_valid";
74  throw lbann_exception(err.str());
75  }
76 
77  int pixelcount = m_image_width * m_image_height * m_image_depth;
78  int current_batch_size = getBatchSize();
79 
80  int n = 0;
81  for (n = CurrentPos; n < CurrentPos + current_batch_size; n++) {
82  if (n >= (int)ShuffledIndices.size())
83  break;
84 
85  int k = n - CurrentPos;
86  int index = ShuffledIndices[n];
87  string imagepath = m_image_dir + ImageList[index].first;
88 
89  int width, height;
90  bool ret = lbann::image_utils::loadJPG(imagepath.c_str(), width, height, false, m_pixels);
91  if(!ret) {
92  throw lbann_exception("ImageNet: image_utils::loadJPG failed to load");
93  }
94  if(width != m_image_width || height != m_image_height) {
95  throw lbann_exception("ImageNet: mismatch data size -- either width or height");
96  }
97 
98  for (int p = 0; p < pixelcount; p++) {
99  X.Set(p, k, m_pixels[p]);
100  }
101 
102  auto pixel_col = X(IR(0, X.Height()), IR(k, k + 1));
103  augment(pixel_col, m_image_height, m_image_width, m_image_depth);
104  normalize(pixel_col, m_image_depth);
105  }
106 
107  return (n - CurrentPos);
108 }

133 bool lbann::DataReader_ImageNet::load(string imageDir, string imageListFile)
134 {
135  m_image_dir = imageDir;
136  ImageList.clear();
137 
138  // load image list
139  FILE* fplist = fopen(imageListFile.c_str(), "rt");
140  if (!fplist) {
141  cerr << __FILE__ << " " << __LINE__ << "failed to open: " << imageListFile << endl;
142  }
143 
144  while (!feof(fplist)) {
145  char imagepath[512];
146  int imagelabel;
147  if (fscanf(fplist, "%s%d", imagepath, &imagelabel) <= 1)
148  break;
149  ImageList.push_back(make_pair(imagepath, imagelabel));
150  }
151  fclose(fplist);
152 
153  // reset indices
154  ShuffledIndices.clear();
155  ShuffledIndices.resize(ImageList.size());
156  for (size_t n = 0; n < ImageList.size(); n++) {
157  ShuffledIndices[n] = n;
158  }
159 
160  return true;
161 }
