240  void bp_compute_gpu() {
241  #ifndef __LIB_CUDNN
242  std::stringstream err;
243  err << __FILE__ << " " << __LINE__ << " :: slice_layer: cuDNN not detected";
244  throw lbann_exception(err.str());
245  #else
246 
247  // Split the error signal tensor into slices of width 1 along the
248  // slice axis
249  const int num_slices
250  = std::accumulate(this->m_prev_neuron_dims.begin(),
251  this->m_prev_neuron_dims.begin() + m_slice_axis,
252  1,
253  std::multiplies<int>());
254  const int slice_unit_size
255  = std::accumulate(this->m_prev_neuron_dims.begin() + m_slice_axis + 1,
256  this->m_prev_neuron_dims.end(),
257  1,
258  std::multiplies<int>());
259  const int output_slice_dim = this->m_prev_neuron_dims[m_slice_axis];
260  const int output_slice_size = output_slice_dim * slice_unit_size;
261 
262  // Copy entries in each child to error signal tensor
263  const int num_gpus = this->m_cudnn->get_num_gpus();
264  for(size_t child_index = 0; child_index < this->m_child_layers.size(); ++child_index) {
265  const Layer* child = this->m_child_layers[child_index];
266 
267  // Get child error signal on GPUs
268  std::vector<DataType*> input;
269  if(child_index == 0) {
270  input = this->m_prev_error_signal_dv;
271  }
272  else {
273  std::vector<DataType*> work_spaces;
274  for(int i=0; i<num_gpus; ++i) {
275  work_spaces.push_back((DataType*)this->m_cudnn->get_work_space(i));
276  }
277  if(child->using_gpus()) {
278  child->get_gpu_bp_output(input, work_spaces, this);
279  }
280  else {
281  child->get_bp_output(*this->m_prev_error_signal_v, this);
282  this->m_cudnn->scatter_to_gpus(work_spaces,
283  this->m_prev_error_signal_v->LockedMatrix(),
284  this->m_mini_batch_size_per_gpu);
285  input = work_spaces;
286  }
287  }
288 
289  // Split previous error signal tensor into slices
290  const int input_slice_dim = m_slice_points[child_index+1] - m_slice_points[child_index];
291  const int input_slice_size = input_slice_dim * slice_unit_size;
292  const int input_size = num_slices * input_slice_size;
293  const int slice_offset = m_slice_points[child_index] * slice_unit_size;
294 
295  // Copy slices from previous error signal tensor into error signal tensor
296  for(int slice = 0; slice < num_slices; ++slice) {
297  std::vector<DataType*> input_slice(num_gpus), output_slice(num_gpus);
298  for(int i = 0; i < num_gpus; ++i) {
299  input_slice[i] = input[i] + slice * input_slice_size;
300  output_slice[i] = this->m_error_signal_d[i] + slice * output_slice_size + slice_offset;
301  }
302  this->m_cudnn->copy_on_gpus(output_slice,
303  input_slice,
304  input_slice_size,
305  this->m_mini_batch_size_per_gpu,
306  input_size,
307  this->m_num_prev_neurons);
308  }
309 
310  }
311 
312  #endif // #ifndef __LIB_CUDNN

315  void bp_compute_cpu() {
316 
317  // Clear error signal if needed
318  if(m_slice_points.front() != 0
319  || m_slice_points.back() != this->m_neuron_dims[m_slice_axis]) {
320  El::Zero(*this->m_error_signal_v);
321  }
322 
323  // Split the error signal tensor into slices of width 1 along the
324  // slice axis
325  const int num_slices
326  = std::accumulate(this->m_prev_neuron_dims.begin(),
327  this->m_prev_neuron_dims.begin() + m_slice_axis,
328  1,
329  std::multiplies<int>());
330  const int slice_unit_size
331  = std::accumulate(this->m_prev_neuron_dims.begin() + m_slice_axis + 1,
332  this->m_prev_neuron_dims.end(),
333  1,
334  std::multiplies<int>());
335  const int output_slice_dim = this->m_prev_neuron_dims[m_slice_axis];
336  const int output_slice_size = output_slice_dim * slice_unit_size;
337 
338  // Copy entries in each child to error signal tensor
339  for(size_t i = 0; i < this->m_child_layers.size(); ++i) {
340 
341  // Split previous error signal tensor into slices
342  this->m_child_layers[i]->get_bp_output(*this->m_prev_error_signal_v, this);
343  const int input_slice_dim = m_slice_points[i+1] - m_slice_points[i];
344  const int input_slice_size = input_slice_dim * slice_unit_size;
345  const int slice_offset_start = m_slice_points[i] * slice_unit_size;
346  const int slice_offset_end = m_slice_points[i+1] * slice_unit_size;
347 
348  // Copy slices from previous error signal tensor into error signal tensor
349  for(int slice = 0; slice < num_slices; ++slice) {
350  El::LockedView(*m_input_slice_v,
351  *this->m_prev_error_signal_v,
352  El::IR(slice * input_slice_size,
353  (slice+1) * input_slice_size),
354  El::ALL);
355  El::View(*m_output_slice_v,
356  *this->m_error_signal_v,
357  El::IR(slice * output_slice_size + slice_offset_start,
358  slice * output_slice_size + slice_offset_end),
359  El::ALL);
360  El::Copy(*m_input_slice_v, *m_output_slice_v);
361  }
362 
363  }

367  void get_fp_output(AbsDistMat& fp_output, const Layer* next_layer) const override {
368 
369  // Check if input is in the list of child layers
370  const int child_index = (std::find(this->m_child_layers.begin(),
371  this->m_child_layers.end(),
372  next_layer)
373  - this->m_child_layers.begin());
374  if(child_index >= (int) this->m_child_layers.size()) {
375  transform::get_fp_output(fp_output, next_layer);
376  }
377 
378  // Split the activations tensor into slices of width 1 along the
379  // slice axis
380  const int num_slices
381  = std::accumulate(this->m_prev_neuron_dims.begin(),
382  this->m_prev_neuron_dims.begin() + m_slice_axis,
383  1,
384  std::multiplies<int>());
385  const int slice_unit_size
386  = std::accumulate(this->m_prev_neuron_dims.begin() + m_slice_axis + 1,
387  this->m_prev_neuron_dims.end(),
388  1,
389  std::multiplies<int>());
390  const int input_slice_dim = this->m_prev_neuron_dims[m_slice_axis];
391  const int output_slice_dim = m_slice_points[child_index+1] - m_slice_points[child_index];
392  const int input_slice_size = input_slice_dim * slice_unit_size;
393  const int output_slice_size = output_slice_dim * slice_unit_size;
394  const int slice_offset_start = m_slice_points[child_index] * slice_unit_size;
395  const int slice_offset_end = m_slice_points[child_index+1] * slice_unit_size;
396 
397  if(num_slices == 1
398  && m_activations_v->DistData() == fp_output.DistData()) {
399  // Return view of activations tensor slice
400  El::LockedView(fp_output,
401  *this->m_activations_v,
402  El::IR(slice_offset_start, slice_offset_end),
403  El::ALL);
404  }
405  else {
406  // Copy slices from activations tensor into output
407  fp_output.Empty(false);
408  fp_output.Resize(output_slice_size, m_activations_v->Width());
409  AbsDistMat* output_slice_v
410  = fp_output.Construct(fp_output.Grid(), fp_output.Root());
411  for(int slice = 0; slice < num_slices; ++slice) {
412  El::LockedView(*m_input_slice_v,
413  *this->m_activations_v,
414  El::IR(slice * input_slice_size + slice_offset_start,
415  slice * input_slice_size + slice_offset_end),
416  El::ALL);
417  El::View(*output_slice_v,
418  fp_output,
419  El::IR(slice * output_slice_size,
420  (slice+1) * output_slice_size),
421  El::ALL);
422  El::Copy(*m_input_slice_v, *output_slice_v);
423  }
424  delete output_slice_v;
425  }
