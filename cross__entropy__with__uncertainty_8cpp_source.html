61 void cross_entropy_with_uncertainty::compute_value(const AbsDistMat& predictions,
62  const AbsDistMat& ground_truth) {
63 
64  // Get local matrices and matrix parameters
65  const Mat& predictions_local = predictions.LockedMatrix();
66  const Mat& ground_truth_local = ground_truth.LockedMatrix();
67  const int width = predictions.Width();
68  const int local_height = predictions_local.Height();
69  const int local_width = predictions_local.Width();
70 
71  AbsDistMat *workspace;
72  if (predictions.DistData().colDist == El::MC) {
73  //model parallel
74  workspace = new StarMRMat(predictions.Grid());
75  } else {
76  //data parallel
77  workspace = new StarVCMat(predictions.Grid());
78  }
79  workspace->Resize(1, width);
80  Mat& local_workspace = workspace->Matrix();
81 
82  for (int col = 0; col < local_width; ++col) {
83  double pred_sum = 0;
84  for (int row = 0; row < local_height; row++) {
85  if (ground_truth_local(row, col) != DataType(0)) {
86  pred_sum += predictions_local(row, col);
87  }
88  }
89  local_workspace(0, col) = pred_sum;
90  }
91  m_objective_function->get_model()->get_comm()->allreduce(
92  *workspace, workspace->RedundantComm());
93 
94  double sum_cross_entropy = 0.0;
95  for (int i = 0; i < local_width; i++) sum_cross_entropy -= std::log(local_workspace(0, i));
96  sum_cross_entropy = m_objective_function->get_model()->get_comm()->allreduce(
97  sum_cross_entropy, workspace->DistComm());
98 
99  // Update objective function value
100  add_to_value(sum_cross_entropy/width);
101
