74 DataType cross_entropy_with_uncertainty::evaluate(const AbsDistMat& predictions,
75  const AbsDistMat& ground_truth) {
76  lbann_comm* comm = m_objective_function->get_model()->get_comm();
77 
78  // Initialize workspace
79  m_prediction_sums->Resize(1, predictions.Width());
80  Mat& prediction_sums_local = m_prediction_sums->Matrix();
81 
82  // Local matrices
83  const Mat& predictions_local = predictions.LockedMatrix();
84  const Mat& ground_truth_local = ground_truth.LockedMatrix();
85 
86  // Matrix parameters
87  const int width = predictions.Width();
88  const int local_height = predictions_local.Height();
89  const int local_width = predictions_local.Width();
90 
91  // Compute sum of predictions
92  #pragma omp parallel for
93  for (int col = 0; col < local_width; ++col) {
94  DataType pred_sum = DataType(0);
95  for (int row = 0; row < local_height; ++row) {
96  if (ground_truth_local(row, col) != DataType(0)) {
97  pred_sum += predictions_local(row, col);
98  }
99  }
100  prediction_sums_local(0, col) = pred_sum;
101  }
102  comm->allreduce(*m_prediction_sums, m_prediction_sums->RedundantComm());
103 
104  // Compute mean objective function value
105  DataType local_sum = DataType(0);
106  for (int col = 0; col < local_width; ++col) {
107  local_sum += -std::log(prediction_sums_local(0, col));
108  }
109  return comm->allreduce(local_sum / width, m_prediction_sums->DistComm());
110 
111 }

120  Mat& gradient_local = gradient.Matrix();
121 
122  // Matrix parameters
123  const El::Int local_height = gradient_local.Height();
124  const El::Int local_width = gradient_local.Width();
125 
126  // Compute gradient
127  #pragma omp parallel for collapse(2)
128  for (El::Int col = 0; col < local_width; ++col) {
129  for (El::Int row = 0; row < local_height; ++row) {
130  const DataType true_val = ground_truth_local(row, col);
131  DataType& grad_val = gradient_local(row, col);
132  if (true_val != DataType(0)) {
133  grad_val = -true_val / prediction_sums_local(0, col);
134  } else {
135  grad_val = DataType(0);
136  }
137  }
