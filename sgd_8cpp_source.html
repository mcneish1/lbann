123 void sgd::step_compute(AbsDistMat& values, const AbsDistMat& gradient) {
124 
125  // SGD without momentum is just an Axpy
126  if (m_momentum == DataType(0)) {
127  El::Axpy(-m_learning_rate, gradient, values);
128  return;
129  }
130 
131  // Get local matrix data
132  const int local_height = values.LocalHeight();
133  const int local_width = values.LocalWidth();
134  DataType* __restrict__ values_buffer = values.Buffer();
135  const int values_ldim = values.LDim();
136  const DataType* __restrict__ gradient_buffer = gradient.LockedBuffer();
137  const int gradient_ldim = gradient.LDim();
138  DataType* __restrict__ velocity_buffer = m_velocity->Buffer();
139  const int velocity_ldim = m_velocity->LDim();
140 
141  // Check if matrix data is contiguous
142  if (values_ldim != local_height
143  || gradient_ldim != local_height
144  || velocity_ldim != local_height) {
145  // (Nesterov) momentum SGD for non-contiguous data
146  #pragma omp parallel for collapse(2)
147  for (int j=0; j<local_width; ++j) {
148  for (int i=0; i<local_height; ++i) {
149  const DataType g = gradient_buffer[i+j*gradient_ldim];
150  DataType& v = velocity_buffer[i+j*velocity_ldim];
151  DataType& x = values_buffer[i+j*values_ldim];
152  v = m_momentum * v + g;
153  x -= (m_nesterov ?
154  m_learning_rate * (m_momentum * v + g) :
155  m_learning_rate * v);
156  }
157  }
158  } else {
159  if (m_nesterov) {
160  // Nesterov's accelerated gradient descent for contiguous data
161  #pragma omp parallel for
162  for (int i=0; i<local_height*local_width; ++i) {
163  DataType& x = values_buffer[i];
164  const DataType g = gradient_buffer[i];
165  DataType& v = velocity_buffer[i];
166  v = m_momentum * v + g;
167  x -= m_learning_rate * (m_momentum * v + g);
168  }
169  } else {
170  // Momentum SGD for contiguous data
171 #pragma omp parallel for
172  for (int i=0; i<local_height*local_width; ++i) {
173  DataType& x = values_buffer[i];
174  const DataType g = gradient_buffer[i];
175  DataType& v = velocity_buffer[i];
176  v = m_momentum * v + g;
177  x -= m_learning_rate * v;
178  }
179  }
180  }
