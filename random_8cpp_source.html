72 void init_random(int seed, lbann_comm *comm) {
73  if (seed != -1) {
74  // Seed every OpenMP thread, if present.
75  // Note: Threadprivate OMP variables don't work with dynamic threads.
76 #ifdef _OPENMP
77  #pragma omp parallel
78  {
79  get_generator().seed((seed << 8) | omp_get_thread_num());
80  get_fast_generator().seed((seed << 8) | omp_get_thread_num());
81  }
82 #else
83  get_generator().seed(seed);
84  get_fast_generator().seed(seed);
85 #endif
86 #ifdef LBANN_SET_EL_RNG
87  if (comm != nullptr) {
88  El::Generator().seed(seed ^ comm->get_rank_in_model());
89  } else {
90  El::Generator().seed(seed ^ El::mpi::Rank(El::mpi::COMM_WORLD));
91  }
92 #endif
93  } else {
94  // Seed with a random value.
95  std::random_device rd;
96  unsigned rand_val = rd();
97 #ifdef _OPENMP
98  #pragma omp parallel
99  {
100  get_generator().seed((rand_val << 8) | omp_get_thread_num());
101  get_fast_generator().seed((rand_val << 8) | omp_get_thread_num());
102  }
103 #else
104  get_generator().seed(rand_val);
105  get_fast_generator().seed(rand_val);
106 #endif
107 #ifdef LBANN_SET_EL_RNG
108  El::Generator().seed(rand_val);
109 #endif
110  }
111 }

158 void gaussian_fill_procdet(AbsDistMat& mat, El::Int m, El::Int n, DataType mean,
159  DataType stddev) {
160  El::Zeros(mat, m, n);
161  if (mat.Grid().Rank() == 0) {
162  mat.Reserve(n * m);
163  auto& gen = get_generator();
164  std::normal_distribution<DataType> dist(mean, stddev);
165  for (El::Int col = 0; col < n; ++col) {
166  for (El::Int row = 0; row < m; ++row) {
167  mat.QueueUpdate(row, col, dist(gen));
168  }
169  }
170  }
171  mat.ProcessQueues();
172 }

174 void bernoulli_fill_procdet(AbsDistMat& mat, El::Int m, El::Int n, double p) {
175  El::Zeros(mat, m, n);
176  if (mat.Grid().Rank() == 0) {
177  mat.Reserve(m * n);
178  auto& gen = get_generator();
179  std::bernoulli_distribution dist(p);
180  for (El::Int col = 0; col < n; ++col) {
181  for (El::Int row = 0; row < m; ++row) {
182  mat.QueueUpdate(row, col, dist(gen) ? 1.0f : 0.0f);
183  }
184  }
185  }
186  mat.ProcessQueues();
187 }

189 void uniform_fill_procdet(AbsDistMat& mat, El::Int m, El::Int n, DataType center,
190  DataType radius) {
191  El::Zeros(mat, m, n);
192  if (mat.Grid().Rank() == 0) {
193  mat.Reserve(n * m);
194  auto& gen = get_generator();
195  std::uniform_real_distribution<DataType> dist(center - radius,
196  center + radius);
197  for (El::Int col = 0; col < n; ++col) {
198  for (El::Int row = 0; row < m; ++row) {
199  mat.QueueUpdate(row, col, dist(gen));
200  }
201  }
202  }
203  mat.ProcessQueues();
204 }
