35 void lbann_callback_timeline::on_train_begin(model *m) {
36  // Set up layers and weights.
37  for (const auto& l : m->get_layers()) {
38  m_fp_times.emplace(l->get_name(), 0);
39  m_bp_times.emplace(l->get_name(), 0);
40  }
41  for (const auto& w : m->get_weights()) {
42  m_opt_times.emplace(w->get_name(), 0);
43  }
44  // Ensure the model is synchronized at the start.
45  m->get_comm()->model_barrier();
46  m_start_time = get_time();
47 }

49 void lbann_callback_timeline::on_train_end(model *m) {
50  const std::string path = m_outdir + "/timeline.m" +
51  std::to_string(m->get_comm()->get_model_rank()) + "." +
52  std::to_string(m->get_comm()->get_rank_in_model()) + ".txt";
53  std::ofstream f(path);
54  for (const auto& kv : m_fp_times) {
55  const std::string name = "fp-" + kv.first;
56  for (const auto& time : kv.second) {
57  f << name << ":" << time.first << ":" << time.second << '\n';
58  }
59  }
60  for (const auto& kv : m_bp_times) {
61  const std::string name = "bp-" + kv.first;
62  for (const auto& time : kv.second) {
63  f << name << ":" << time.first << ":" << time.second << '\n';
64  }
65  }
66  for (const auto& kv : m_opt_times) {
67  const std::string name = "opt-" + kv.first;
68  for (const auto& time : kv.second) {
69  f << name << ":" << time.first << ":" << time.second << '\n';
70  }
71  }
72 }
