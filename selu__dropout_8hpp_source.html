43 class selu_dropout : public regularizer_layer {
44  public:
46  selu_dropout(lbann_comm *comm,
47  float keep_prob=0.95f,
48  DataType alpha = DataType(1.6732632423543772848170429916717),
49  DataType scale = DataType(1.0507009873554804934193349852946)) :
50  regularizer_layer(comm),
51  m_keep_prob(keep_prob) {
52 #ifdef LBANN_PROCDET_DROPOUT
53  throw lbann_exception("selu_dropout: deterministic dropout not supported");
54 #endif
55  // Setup the data distribution
56  initialize_distributed_matrices();
57  // Compute alpha' and the affine transform.
58  m_alpha_prime = -scale*alpha;
59  m_a = keep_prob +
60  m_alpha_prime*m_alpha_prime*keep_prob*(DataType(1) - keep_prob);
61  m_a = DataType(1) / std::sqrt(m_a);
62  m_b = -m_a * m_alpha_prime*(DataType(1) - keep_prob);
63  }
64 
65  selu_dropout(const selu_dropout& other) :
66  regularizer_layer(other),
67  m_alpha_prime(other.m_alpha_prime),
68  m_a(other.m_a),
69  m_b(other.m_b),
70  m_keep_prob(other.m_keep_prob) {
71  m_cur_mask = new Mat(*other.m_cur_mask);
72  }

100  void fp_compute() override {
101  if (this->m_model->get_execution_mode() != execution_mode::training ||
102  m_keep_prob < 0.0f) {
103  // Do nothing if dropout is disabled
104  El::Copy(*(this->m_prev_activations_v), *(this->m_activations_v));
105  } else {
106 
107  AbsDistMat *input_acts = this->m_prev_activations_v;
108  const El::Int local_height = input_acts->LocalHeight();
109  const El::Int local_width = input_acts->LocalWidth();
110 
111  Mat& local_input_acts = input_acts->Matrix();
112  Mat& local_output_acts = this->m_activations_v->Matrix();
113 
114  // Construct and apply mask and the affine transform.
115  // TODO: Optimize.
116  El::Bernoulli(*m_cur_mask, local_height, local_width, m_keep_prob);
117  for (El::Int col = 0; col < local_width; ++col) {
118  for (El::Int row = 0; row < local_height; ++row) {
119  local_output_acts(row, col) = m_a *
120  (local_input_acts(row, col)*(*m_cur_mask)(row, col) +
121  m_alpha_prime*(1 - (*m_cur_mask)(row, col))) + m_b;
122  }
123  }
124 
125  }
126  }
127 
129  void bp_compute() override {
130  if (this->m_model->get_execution_mode() != execution_mode::training
131  || m_keep_prob < 0.0f) {
132  El::LockedView(*this->m_error_signal_v, *this->m_prev_error_signal_v);
133  } else {
134 
135  Mat& local_prev_error_signal = this->m_prev_error_signal_v->Matrix();
136  Mat& local_error_signal = this->m_error_signal_v->Matrix();
137  const El::Int local_height = local_prev_error_signal.Height();
138  const El::Int local_width = local_prev_error_signal.Width();
139  // Reweight with the affine scale factor and the dropout mask.
140  for (El::Int col = 0; col < local_width; ++col) {
141  for (El::Int row = 0; row < local_height; ++row) {
142  local_error_signal(row, col) =
143  m_a * local_prev_error_signal(row, col) * (*m_cur_mask)(row, col);
144  }
145  }
146 
147  }
148  }
