38 void lbann_callback_early_stopping::on_validation_end(model* m) {
39  double score = m->obj_fn->report_aggregate_avg_obj_fn(execution_mode::validation);
40  if (score < last_score) {
41  if (m->get_comm()->am_model_master()) {
42  std::cout << "Model " << m->get_comm()->get_model_rank() <<
43  " early stopping: score is improving " << last_score << " >> " << score << std::endl;
44  }
45  last_score = score;
46  wait = 0;
47  } else {
48  if (wait >= patience) {
49  m->set_terminate_training(true);
50  if (m->get_comm()->am_model_master()) {
51  std::cout << "Model " << m->get_comm()->get_model_rank() <<
52  " terminating training due to early stopping: " << score << " score and " << last_score << " last score" << std::endl;
53  }
54  } else {
55  ++wait;
56  }
57  }
58 }
