41 // FullyConnectedLayer : single network layer class
43 // WB structure: (num units "neurons / filters" x (num features + 1))
44 // Each row represents a neuron / filter
45 // There is a column for each feature coming in from the previous layer plus 1 for the bias
46 // [W00 ... B0]
47 // [| |]
48 // [Wn0 Bn]
49 //
50 // WB_D structure:
51 // [dW dB]
52 // D structure:
53 // [D ]
54 // Z, Zs, Act, Acts structure:
55 // [Acts ]

87 void lbann::FullyConnectedLayer::setup(int numPrevNeurons) {
88  Layer::setup(numPrevNeurons);
89  if(optimizer != NULL) {
90  optimizer->setup(numPrevNeurons+1, NumNeurons);
91  }
92 
93  // Initialize weight-bias matrix
94  // Note that the weight-bias matrix has an extra column so that it will include
95  // the bias term from the previous layer's activations in the linear combination
96  Zeros(*m_weights, NumNeurons, numPrevNeurons+1);
97 
99  // Initialize weights
100  DistMat weights;
101  View(weights, *m_weights, ALL, IR(0,numPrevNeurons));
102  switch(m_weight_initialization) {
103  case weight_initialization::uniform:
104  uniform_fill(weights, weights.Height(), weights.Width(),
105  DataType(0), DataType(1));
106  break;
107  case weight_initialization::normal:
108  gaussian_fill(weights, weights.Height(), weights.Width(),
109  DataType(0), DataType(1));
110  break;
111  case weight_initialization::glorot_normal: {
112  const DataType var = 2.0 / (numPrevNeurons + NumNeurons);
113  gaussian_fill(weights, weights.Height(), weights.Width(),
114  DataType(0), sqrt(var));
115  break;
116  }
117  case weight_initialization::glorot_uniform: {
118  const DataType var = 2.0 / (numPrevNeurons + NumNeurons);
119  uniform_fill(weights, weights.Height(), weights.Width(),
120  DataType(0), sqrt(3*var));
121  break;
122  }
123  case weight_initialization::he_normal: {
124  const DataType var = 1.0 / numPrevNeurons;
125  gaussian_fill(weights, weights.Height(), weights.Width(),
126  DataType(0), sqrt(var));
127  break;
128  }
129  case weight_initialization::he_uniform: {
130  const DataType var = 1.0 / numPrevNeurons;
131  uniform_fill(weights, weights.Height(), weights.Width(),
132  DataType(0), sqrt(3*var));
133  break;
134  }
135  case weight_initialization::zero: // Zero initialization is default
136  default:
137  Zero(weights);
138  break;
139  }
140 
141  // Initialize other matrices
142  Zeros(*m_weights_gradient, NumNeurons, numPrevNeurons + 1);
143  Zeros(*m_prev_error_signal, NumNeurons, m_mini_batch_size);
144  Zeros(*m_error_signal, numPrevNeurons, m_mini_batch_size); // m_error_signal holds the product of m_weights^T * m_prev_error_signal
145  Zeros(*m_weighted_sum, NumNeurons, m_mini_batch_size);
146  Zeros(*m_activations, NumNeurons, m_mini_batch_size);
147  Zeros(*m_prev_activations, numPrevNeurons, m_mini_batch_size);
148 
150  View(m_activation_weights_v, *m_weights, ALL, IR(0, numPrevNeurons));
151  View(m_bias_weights_v, *m_weights, ALL, IR(numPrevNeurons));
152 
154  View(m_activation_weights_gradient_v, *m_weights_gradient, ALL, IR(0, numPrevNeurons));
155  View(m_bias_weights_gradient_v, *m_weights_gradient, ALL, IR(numPrevNeurons));
156 
158  Ones(m_bias_bp_t, m_mini_batch_size, 1);
159 }

171 void lbann::FullyConnectedLayer::fp_linearity()
172 {
173  // Apply forward prop linearity
174 
175  // Apply bias
176  DistMatrix<DataType,MC,STAR> bias_weights_mc_star(m_bias_weights_v);
177  const Mat& local_bias_weights = bias_weights_mc_star.Matrix();
178  IndexDependentFill(m_weighted_sum_v->Matrix(), (std::function<DataType(El::Int,El::Int)>)
179  ([&local_bias_weights](El::Int r, El::Int c)->DataType {
180  return local_bias_weights.Get(r);
181  }));
182  Scale(m_bias_term, *m_weighted_sum_v);
183 
184  // Apply weight matrix
185  Gemm(NORMAL, NORMAL, DataType(1), m_activation_weights_v, *m_prev_activations_v,
186  DataType(1), *m_weighted_sum_v);
187 
188  // Copy result to output matrix
189  Copy(*m_weighted_sum_v, *m_activations_v);
190 }

192 void lbann::FullyConnectedLayer::bp_linearity()
193 {
194  // Compute the partial delta update for the next lower layer
195  Gemm(TRANSPOSE, NORMAL, DataType(1), m_activation_weights_v, *m_prev_error_signal_v,
196  DataType(0), *m_error_signal_v);
197  // Compute update for activation weights
198  Gemm(NORMAL, TRANSPOSE, DataType(1)/get_effective_minibatch_size(), *m_prev_error_signal_v,
199  *m_prev_activations_v, DataType(0), m_activation_weights_gradient_v);
200  // Compute update for bias terms
201  Gemv(NORMAL, DataType(1)/get_effective_minibatch_size(), *m_prev_error_signal_v,
202  m_bias_bp_t_v, DataType(0), m_bias_weights_gradient_v);
203 }

205 DataType lbann::FullyConnectedLayer::computeCost(DistMat &deltas) {
206  DataType avg_error = 0.0, total_error = 0.0;
207  // Compute the L2 norm on the deltas (activation - y)
208  ColSumMat norms;
209  ColumnTwoNorms(deltas, norms);
210  int c = 0;
211  // Sum the local, total error
212  for(int r = 0; r < norms.LocalHeight(); r++) {
213  total_error += norms.GetLocal(r,c);
214  }
215  mpi::AllReduce(total_error, norms.DistComm());
216  avg_error = total_error / norms.Height();
217  return avg_error;
218 }

237 {
238  DistMat WB_E1(m_weights->Grid());
239  DistMat WB_E2(m_weights->Grid());
240  DistMat Zs_E1(m_weighted_sum->Grid());
241  DistMat Zs_E2(m_weighted_sum->Grid());
242  DistMat Acts_E1(m_activations->Grid());
243  DistMat Acts_E2(m_activations->Grid());
244  DataType grad_diff = 0;
245  DataType grad_sum = 0;
246 
247  Zeros(WB_E1, m_weights->Height(), m_weights->Width());
248  Zeros(WB_E2, m_weights->Height(), m_weights->Width());
249  Zeros(Zs_E1, m_weighted_sum->Height(), m_weighted_sum->Width());
250  Zeros(Zs_E2, m_weighted_sum->Height(), m_weighted_sum->Width());
251  Zeros(Acts_E1, m_activations->Height(), m_activations->Width());
252  Zeros(Acts_E2, m_activations->Height(), m_activations->Width());
253 
254  Copy(*m_weights, WB_E1);
255  Copy(*m_weights, WB_E2);
256 
257  DataType sum_error = 0;
258  Int prow = 0;
259  Int pcol = 0;
260 
261  if(WB_E1.IsLocal(prow, pcol)) {
262  Int _prow = WB_E1.LocalRow(prow);
263  Int _pcol = WB_E1.LocalCol(pcol);
264  WB_E1.SetLocal(_prow, _pcol, WB_E1.GetLocal(_prow, _pcol) + Epsilon);
265  }
266  if(WB_E2.IsLocal(prow, pcol)) {
267  Int _prow = WB_E2.LocalRow(prow);
268  Int _pcol = WB_E2.LocalCol(pcol);
269  WB_E2.SetLocal(_prow, _pcol, WB_E2.GetLocal(_prow, _pcol) - Epsilon);
270  }
271  for (Int row = 0; row < m_weights->Height(); row++) {
272  for (Int col = 0; col < m_weights->Width(); col++) {
273  // printf("Updating %d: %d x %d\n", Index, row, col);
274  if(WB_E1.IsLocal(prow, pcol)) {
275  Int _prow = WB_E1.LocalRow(prow);
276  Int _pcol = WB_E1.LocalCol(pcol);
277  WB_E1.SetLocal(_prow, _pcol, WB_E1.GetLocal(_prow, _pcol) - Epsilon);
278  }
279  if(WB_E2.IsLocal(prow, pcol)) {
280  Int _prow = WB_E2.LocalRow(prow);
281  Int _pcol = WB_E2.LocalCol(pcol);
282  WB_E2.SetLocal(_prow, _pcol, WB_E2.GetLocal(_prow, _pcol) + Epsilon);
283  }
284  if(WB_E1.IsLocal(row, col)) {
285  Int _row = WB_E1.LocalRow(row);
286  Int _col = WB_E1.LocalCol(col);
287  WB_E1.SetLocal(_row, _col, WB_E1.GetLocal(_row, _col) + Epsilon);
288  }
289  if(WB_E2.IsLocal(row, col)) {
290  Int _row = WB_E2.LocalRow(row);
291  Int _col = WB_E2.LocalCol(col);
292  WB_E2.SetLocal(_row, _col, WB_E2.GetLocal(_row, _col) - Epsilon);
293  }
294 
295  // J(theta)
296  // this->fp_linearity(*WB, *(PrevLayer.Acts), *Zs, *Acts);
297  //this->fp_nonlinearity(*Acts);
298 
299  // J(thetaPlus(i))
300  // this->fp_linearity(WB_E1, *(PrevLayer.Acts), Zs_E1, Acts_E1);
301  //this->fp_nonlinearity(Acts_E1);
302 
303  // J(thetaMinus(i))
304  // this->fp_linearity(WB_E2, *(PrevLayer.Acts), Zs_E2, Acts_E2);
305  //this->fp_nonlinearity(Acts_E2);
306 
307  // this->getCost();
308 
309  // gradApprox(i) = J(thetaPlus(i)) - J(thetaMinus(i)) / (2*Epsilon)
310  Axpy(-1.0, *m_activations, Acts_E1);
311  Axpy(-1.0, *m_activations, Acts_E2);
312 
313  bool bad_E1 = false, bad_E2 = false;
314  for(Int r = 0; r < m_activations->Height(); r++) {
315  for(Int c = 0; c < m_activations->Width(); c++) {
316  if(Acts_E1.IsLocal(r,c)) {
317  Int _r = Acts_E1.LocalRow(r);
318  Int _c = Acts_E1.LocalCol(c);
319  if((Acts_E1.GetLocal(_r,_c) > 1e-12 || Acts_E1.GetLocal(_r,_c) < -1e-12) && r != row) {
320  bad_E1 = true;

322  }
323  }
324  if(Acts_E2.IsLocal(r,c)) {
325  Int _r = Acts_E2.LocalRow(r);
326  Int _c = Acts_E2.LocalCol(c);
327  if((Acts_E2.GetLocal(_r,_c) > 1e-12 || Acts_E2.GetLocal(_r,_c) < -1e-12) && r != row) {
328  bad_E2 = true;
329  // cout << "Acts_E2=["<< r << "," << c << "]="<<Acts_E2.GetLocal(_r,_c)<<endl;
330  }
331  }
332  }
333  }
334 
335  if(bad_E1) {
336  if(Acts_E1.Grid().Rank() == 0) {
337  printf("BAD ENTRY Acts_E1 %d x %d\n", row, col);
338  }
339  cout.precision(20);
340  Print(Acts_E1);
341  if(Acts_E1.Grid().Rank() == 0) {
342  printf("Acts\n");
343  }
344  Print(*m_activations);
345  }
346  if(bad_E2) {
347  if(Acts_E2.Grid().Rank() == 0) {
348  printf("BAD ENTRY Acts_E2 %d x %d\n", row, col);
349  }
350  Print(Acts_E2);
351  }
352 
353  prow = row;
354  pcol = col;
355  }
356  }
357  DataType grad_error = sqrt(grad_diff / grad_sum);
358  return grad_error;
359 }
