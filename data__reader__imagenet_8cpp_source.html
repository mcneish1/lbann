86 bool imagenet_reader::replicate_processor(const cv_process& pp) {
87  const int nthreads = omp_get_max_threads();
88  m_pps.resize(nthreads);
89 
90  // Construct thread private preprocessing objects out of a shared pointer
91  #pragma omp parallel for schedule(static, 1)
92  for (int i = 0; i < nthreads; ++i) {
93  //auto ppu = std::make_unique<cv_process>(pp); // c++14
94  std::unique_ptr<cv_process> ppu(new cv_process(pp));
95  m_pps[i] = std::move(ppu);
96  }
97 
98  bool ok = true;
99  for (int i = 0; ok && (i < nthreads); ++i) {
100  if (!m_pps[i]) ok = false;
101  }
102 
103  if (!ok || (nthreads <= 0)) {
104  std::stringstream err;
105  err << __FILE__<<" "<<__LINE__<< " :: Imagenet data reader construction error: cannot replicate image processor";
106  throw lbann_exception(err.str());
107  return false;
108  }
109 
110  const std::vector<unsigned int> dims = pp.get_data_dims();
111  if ((dims.size() == 2u) && (dims[0] != 0u) && (dims[1] != 0u)) {
112  m_image_width = static_cast<int>(dims[0]);
113  m_image_height = static_cast<int>(dims[1]);
114  }
115 
116  return true;
117 }

123 bool imagenet_reader::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
124  const int num_channel_values = m_image_width * m_image_height * m_image_num_channels;
125  const std::string imagepath = get_file_dir() + m_image_list[data_id].first;
126 
127  int width=0, height=0, img_type=0;
128 
129  ::Mat X_v = create_datum_view(X, mb_idx);
130  const bool ret = lbann::image_utils::load_image(imagepath, width, height, img_type, *(m_pps[tid]), X_v);
131 
132  if(!ret) {
133  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__)
134  + "ImageNet: image_utils::load_image failed to load - "
135  + imagepath);
136  }
137  if((width * height * CV_MAT_CN(img_type)) != num_channel_values) {
138  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__)
139  + "ImageNet: mismatch data size -- either width, height or channel - "
140  + imagepath + "[w,h,c]=[" + std::to_string(width) + "x" + std::to_string(height)
141  + "x" + std::to_string(CV_MAT_CN(img_type)) + "]");
142  }
143  return true;
144 }
