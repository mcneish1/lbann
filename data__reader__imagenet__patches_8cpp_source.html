89 bool imagenet_reader_patches::replicate_processor(const cv_process_patches& pp) {
90  const int nthreads = omp_get_max_threads();
91  m_pps.resize(nthreads);
92 
93  // Construct thread private preprocessing objects out of a shared pointer
94  #pragma omp parallel for schedule(static, 1)
95  for (int i = 0; i < nthreads; ++i) {
96  //auto ppu = std::make_unique<cv_process_patches>(pp); // c++14
97  std::unique_ptr<cv_process_patches> ppu(new cv_process_patches(pp));
98  m_pps[i] = std::move(ppu);
99  }
100 
101  bool ok = true;
102  for (int i = 0; ok && (i < nthreads); ++i) {
103  if (!m_pps[i]) ok = false;
104  }
105 
106  if (!ok || (nthreads <= 0)) {
107  std::stringstream err;
108  err << __FILE__<<" "<<__LINE__<< " :: Imagenet data reader construction error: cannot replicate image processor";
109  throw lbann_exception(err.str());
110  return false;
111  }
112  const std::vector<unsigned int> dims = pp.get_data_dims();
113  if ((dims.size() == 3u) && (dims[0] != 0u) && (dims[1] != 0u) && (dims[2] != 0u)) {
114  m_num_patches = static_cast<int>(dims[0]);
115  m_image_width = static_cast<int>(dims[1]);
116  m_image_height = static_cast<int>(dims[2]);
117  }
118 
119  return true;
120 }

148 bool imagenet_reader_patches::fetch_datum(Mat& X, int data_id, int mb_idx, int tid) {
149  const int num_channel_values = m_image_width * m_image_height * m_image_num_channels;
150  const std::string imagepath = get_file_dir() + m_image_list[data_id].first;
151 
152  int width=0, height=0, img_type=0;
153  std::vector<::Mat> X_v = create_datum_views(X, mb_idx);
154 
155  const bool ret = lbann::image_utils::load_image(imagepath, width, height, img_type, *(m_pps[tid]), X_v);
156 
157  if (m_pps[tid]->is_self_labeling()) {
158  m_image_list[data_id].second = m_pps[tid]->get_patch_label();
159  }
160 
161  if(!ret) {
162  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__)
163  + "ImageNet: image_utils::load_image failed to load - "
164  + imagepath);
165  }
166  if((width * height * CV_MAT_CN(img_type)) != num_channel_values) {
167  throw lbann_exception(std::string{} + __FILE__ + " " + std::to_string(__LINE__)
168  + "ImageNet: mismatch data size -- either width, height or channel - "
169  + imagepath + " [w,h,c]=[" + std::to_string(width) + "x" + std::to_string(height)
170  + "x" + std::to_string(CV_MAT_CN(img_type)) + "] != " + std::to_string(num_channel_values));
171  }
172  return true;
173 }

202 int imagenet_reader_patches::fetch_data(std::vector<::Mat>& X) {
203  const int nthreads = omp_get_max_threads();
204  if(!position_valid()) {
205  throw lbann_exception(
206  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
207  + " :: imagenet data reader load error: !position_valid"
208  + " -- current pos = " + std::to_string(m_current_pos)
209  + " and there are " + std::to_string(m_shuffled_indices.size()) + " indices");
210  }
211  if (X.size() == 0u) {
212  throw lbann_exception(
213  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
214  " :: imagenet data reader fetch error: X.size()==0");
215  }
216  const El::Int x_height = X[0].Height();
217  const El::Int x_width = X[0].Width();
218 
221  #pragma omp parallel for schedule(static, 1)
222  for (int t = 0; t < nthreads; t++) {
223  preprocess_data_source(omp_get_thread_num());
224  }
225 
226  const int current_batch_size = get_current_mini_batch_size();
227  const int end_pos = std::min(static_cast<size_t>(m_current_pos+current_batch_size),
228  m_shuffled_indices.size());
229  const int mb_size = std::min(
230  El::Int{((end_pos - m_current_pos) + m_sample_stride - 1) / m_sample_stride},
231  x_width);
232 
233  for (auto & x : X) {
234  El::Zeros(x, x_height, x_width);
235  }
236  El::Zeros(m_indices_fetched_per_mb, mb_size, 1);
237  #pragma omp parallel for schedule(static, 1)
238  for (int s = 0; s < mb_size; s++) {
239  // Catch exceptions within the OpenMP thread.
240  try {
241  const int n = m_current_pos + (s * m_sample_stride);
242  const int index = m_shuffled_indices[n];
243  fetch_datum(X, index, s, omp_get_thread_num());
244  m_indices_fetched_per_mb.Set(s, 0, index);
245  } catch (lbann_exception& e) {
246  lbann_report_exception(e);
247  } catch (std::exception& e) {
248  El::ReportException(e);
249  }
250  }
251 
254  #pragma omp parallel for schedule(static, 1)
255  for (int t = 0; t < nthreads; t++) {
256  postprocess_data_source(omp_get_thread_num());
257  }
258 
259  return mb_size;
260 }
