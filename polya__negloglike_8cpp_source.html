29 double digamma(double x) {
30  double result = 0.0;
31  double z = x;
32  while (z < 8.0) {
33  result -= 1.0/z;
34  z += 1.0;
35  }
36  double s = 1.0/z;
37  double s2 = s*s;
38  double s4 = s2*s2;
39  double s6 = s2*s4;
40  double s8 = s4*s4;
41  result += std::log(z) - s/2.0 - s2/12.0 + s4/120.0 - s6/252.0 + s8/240.0 - (5.0/660.0)*s4*s6 +(691.0/32760.0)*s6*s6 - s6*s8/12.0;
42  return result;
43 }

129 DataType polya_negloglike::evaluate(const AbsDistMat& predictions,
130  const AbsDistMat& ground_truth) {
131  lbann_comm* comm = m_objective_function->get_model()->get_comm();
132 
133  // Initialize workspace
134  m_counts->Resize(1, predictions.Width());
135  m_alpha_sums->Resize(1, predictions.Width());
136  m_lgamma_alpha_sums->Resize(1, predictions.Width());
137  m_lgamma_alpha_level_count_sums->Resize(1, predictions.Width());
138  Mat& counts_local = m_counts->Matrix();
139  Mat& alpha_sums_local = m_alpha_sums->Matrix();
140  Mat& lgamma_alpha_sums_local = m_lgamma_alpha_sums->Matrix();
141  Mat& lgamma_alpha_level_count_sums_local = m_lgamma_alpha_level_count_sums->Matrix();
142 
143  // Local matrices
144  const Mat& predictions_local = predictions.LockedMatrix();
145  const Mat& ground_truth_local = ground_truth.LockedMatrix();
146 
147  // Matrix parameters
148  const int width = predictions.Width();
149  const int local_height = predictions_local.Height();
150  const int local_width = predictions_local.Width();
151 
152  #pragma omp parallel for
153  for (int col = 0; col < local_width; ++col) {
154  DataType count = DataType(0);
155  DataType alpha_sum = DataType(0);
156  DataType lgamma_alpha_sum = DataType(0);
157  DataType lgamma_alpha_level_count_sum = DataType(0);
158  for (int row = 0; row < local_height; ++row) {
159  const DataType true_val = ground_truth_local(row, col);
160  const DataType pred_val = predictions_local(row, col);
161  count += true_val;
162  alpha_sum += pred_val;
163  lgamma_alpha_sum += std::lgamma(pred_val);
164  lgamma_alpha_level_count_sum += std::lgamma(pred_val + true_val);
165  }
166  counts_local(0, col) = count;
167  alpha_sums_local(0, col) = alpha_sum;
168  lgamma_alpha_sums_local(0, col) = lgamma_alpha_sum;
169  lgamma_alpha_level_count_sums_local(0, col) = lgamma_alpha_level_count_sum;
170  }
171  comm->allreduce(*m_counts, m_counts->RedundantComm());
172  comm->allreduce(*m_alpha_sums, m_alpha_sums->RedundantComm());
173  comm->allreduce(*m_lgamma_alpha_sums, m_lgamma_alpha_sums->RedundantComm());
174  comm->allreduce(*m_lgamma_alpha_level_count_sums,
175  m_lgamma_alpha_level_count_sums->RedundantComm());
176 
177  // Compute mean objective function value across mini-batch
178  DataType local_sum = DataType(0);
179  for (int col = 0; col < local_width; ++col) {
180  local_sum += (- std::lgamma(alpha_sums_local(0, col))
181  + std::lgamma(counts_local(0, col) + alpha_sums_local(0, col))
182  - lgamma_alpha_level_count_sums_local(0, col)
183  + lgamma_alpha_sums_local(0, col));
184  }
185  return comm->allreduce(local_sum / width, m_counts->DistComm());
186 
187 }

189 void polya_negloglike::differentiate(const AbsDistMat& predictions,
190  const AbsDistMat& ground_truth,
191  AbsDistMat& gradient) {
192 
193  // Local matrices
194  const Mat& predictions_local = predictions.LockedMatrix();
195  const Mat& ground_truth_local = ground_truth.LockedMatrix();
196  const Mat& counts_local = m_counts->LockedMatrix();
197  const Mat& alpha_sums_local = m_alpha_sums->LockedMatrix();
198  Mat& gradient_local = gradient.Matrix();
199 
200  // Matrix parameters
201  const El::Int local_height = gradient_local.Height();
202  const El::Int local_width = gradient_local.Width();
203 
204  // Compute gradient
205  #pragma omp parallel for collapse(2)
206  for (El::Int col = 0; col < local_width; ++col) {
207  for (El::Int row = 0; row < local_height; ++row) {
208  const DataType true_val = ground_truth_local(row, col);
209  const DataType pred_val = predictions_local(row, col);
210  const DataType alpha_sum = alpha_sums_local(0, col);
211  const DataType count = counts_local(0, col);
212  gradient_local(row, col) = (- digamma(alpha_sum)
213  + digamma(count + alpha_sum)
214  - digamma(true_val + pred_val)
215  + digamma(pred_val));
216  }
217  }
