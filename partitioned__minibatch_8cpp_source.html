110 void lbann::partitioned_minibatch::calculate_num_iterations_per_epoch_spanning_models(int max_mini_batch_size, generic_data_reader *data_reader) {
111  if(data_reader == nullptr) { return; }
112  // If the data reader does not have any data bail out (e.g. unused validation reader)
113  if(data_reader->get_use_percent() == double(0.0)) { return; }
114 
116  if(max_mini_batch_size > data_reader->get_num_data()) {
117  max_mini_batch_size = data_reader->get_num_data();
118  }
119 
121  int num_parallel_readers_per_model = compute_max_num_parallel_readers(data_reader->get_num_data(), max_mini_batch_size, m_requested_max_num_parallel_readers);
122  data_reader->set_num_parallel_readers(num_parallel_readers_per_model);
123  if(num_parallel_readers_per_model == 0
124  || (num_parallel_readers_per_model != m_comm->get_procs_per_model() && num_parallel_readers_per_model != max_mini_batch_size)) {
125  throw lbann_exception(
126  std::string{} + __FILE__ + " " + std::to_string(__LINE__)
127  + " :: partitioned_minibatch: number of parallel readers is " + std::to_string(num_parallel_readers_per_model)
128  + " and there are " + std::to_string(m_comm->get_procs_per_model()) + " processes in the model");
129  }
130 
132  int batch_stride = m_comm->get_num_models() * max_mini_batch_size;
133  int base_offset = m_comm->get_rank_in_model();
134  int model_offset = m_comm->get_model_rank() * max_mini_batch_size;
136  data_reader->set_mini_batch_size(max_mini_batch_size);
137  data_reader->set_stride_to_next_mini_batch(batch_stride);
138  data_reader->set_sample_stride(num_parallel_readers_per_model);
139  data_reader->set_iteration_stride(1);
141  data_reader->set_base_offset(base_offset);
142  data_reader->set_model_offset(model_offset);
143  data_reader->set_initial_position();
144 
145  int min_stride_across_models = max_mini_batch_size * m_comm->get_num_models();
146 
147  data_reader->set_global_mini_batch_size(min_stride_across_models);
148 
149  data_reader->set_last_mini_batch_size(max_mini_batch_size);
150  data_reader->set_global_last_mini_batch_size(min_stride_across_models);
151 
152  int num_whole_mini_batches_per_model = floor(data_reader->get_num_data() / min_stride_across_models);
153  int global_partial_mini_batch_size = data_reader->get_num_data() - (num_whole_mini_batches_per_model * min_stride_across_models);
154  int per_model_partial_mini_batch_size = global_partial_mini_batch_size / m_comm->get_num_models();
155  int world_master_remainder_data = 0;
156 
157  // Compute how many full "parallel" mini-batches are available
158  int last_mini_batch_threshold = num_whole_mini_batches_per_model * min_stride_across_models;
159 
160  int world_master_remainder_adjustment = data_reader->get_num_data()
161  - (num_whole_mini_batches_per_model * min_stride_across_models)
162  - (per_model_partial_mini_batch_size * m_comm->get_num_models());
163  if(m_comm->get_model_rank() == 0) {
164  world_master_remainder_data = world_master_remainder_adjustment;
165  world_master_remainder_adjustment = 0;
166  }
167  per_model_partial_mini_batch_size += world_master_remainder_data;
168 
169  if(per_model_partial_mini_batch_size > 0 || world_master_remainder_adjustment > 0) {
170  data_reader->set_last_mini_batch_size(per_model_partial_mini_batch_size);
171  data_reader->set_global_last_mini_batch_size(global_partial_mini_batch_size);
172  }
173 
174  if(global_partial_mini_batch_size != 0) {
175  data_reader->set_num_iterations_per_epoch(num_whole_mini_batches_per_model+1);
176  }else {
177  data_reader->set_num_iterations_per_epoch(num_whole_mini_batches_per_model);
178  }
179 
180  if(data_reader->get_last_mini_batch_size() > max_mini_batch_size) {
181  throw new lbann_exception("Error in calculating the partial mini-batch size, exceeds the max mini-batch size");
182  }
183 
186 
188  data_reader->set_stride_to_last_mini_batch(data_reader->get_stride_to_next_mini_batch());
189 
192  int last_mini_batch_offset = std::max(0, num_whole_mini_batches_per_model - 1) * data_reader->get_stride_to_next_mini_batch();
193 
195  if(per_model_partial_mini_batch_size > 0 || world_master_remainder_adjustment > 0) {
196  data_reader->set_stride_to_last_mini_batch((last_mini_batch_threshold - data_reader->get_base_offset() - data_reader->get_model_offset() - last_mini_batch_offset) + m_comm->get_model_rank() * per_model_partial_mini_batch_size + m_comm->get_rank_in_model());
197  }
198

200 //cout << "[" << m_comm->get_rank_in_world() << "] " << m_comm->get_model_rank() << " model rank, "<< m_comm->get_rank_in_model() << " rank in model, num_whole_mini_batches_per_model " << num_whole_mini_batches_per_model << " num_whole_mini_batches_per_reader " << num_whole_mini_batches_per_reader << "(m_num_mini_batches_per_reader=" << data_reader->get_num_mini_batches_per_reader() << ") parallel_readers_with_extra_mini_batch " << /*parallel_readers_with_extra_mini_batch <<*/ " partial_mini_batch_size=" << per_model_partial_mini_batch_size << " last mini bath size=" << data_reader->get_last_mini_batch_size() << " world_master_remainder_data=" << world_master_remainder_data << " threshold " << data_reader->get_last_mini_batch_threshold() << " with a last stride of " << data_reader->get_stride_to_last_mini_batch() << " and stride of " << data_reader->get_batch_stride() << " and there are " << num_parallel_readers_per_model << " parallel readers per model" << " last mini batch offset = " << last_mini_batch_offset << " parallel reader with extra minibatch = " << /*parallel_readers_with_extra_mini_batch << */" model bracket = " << (/*parallel_readers_with_extra_mini_batch **/ max_mini_batch_size + per_model_partial_mini_batch_size + world_master_remainder_data) <<" base ofset "<< data_reader->get_base_offset() << " model offset " << data_reader->get_model_offset() <<endl;

204 void lbann::partitioned_minibatch::calculate_num_iterations_per_epoch_single_model(int max_mini_batch_size, generic_data_reader *data_reader) {
205  if(data_reader == nullptr) { return; }
206  // If the data reader does not have any data bail out (e.g. unused validation reader)
207  if(data_reader->get_use_percent() == double(0.0)) { return; }
208 
209  if(max_mini_batch_size > data_reader->get_num_data()) {
210  max_mini_batch_size = data_reader->get_num_data();
211  }
212 
214  int num_parallel_readers_per_model = compute_max_num_parallel_readers(data_reader->get_num_data(), max_mini_batch_size, m_requested_max_num_parallel_readers);
215  data_reader->set_num_parallel_readers(num_parallel_readers_per_model);
216  if(num_parallel_readers_per_model == 0
217  || (num_parallel_readers_per_model != m_comm->get_procs_per_model() && num_parallel_readers_per_model != max_mini_batch_size)) {
218  throw lbann_exception(
219  std::string{} + __FILE__ + " " + std::to_string(__LINE__) +
220  " :: generic_data_distribution: number of parallel readers is zero");
221  }
222 
224  int batch_stride = max_mini_batch_size;
225  int base_offset = m_comm->get_rank_in_model();
227  data_reader->set_mini_batch_size(max_mini_batch_size);
228  data_reader->set_stride_to_next_mini_batch(batch_stride);
229  data_reader->set_sample_stride(num_parallel_readers_per_model);
230  data_reader->set_iteration_stride(1);
232  data_reader->set_base_offset(base_offset);
233  data_reader->set_model_offset(0);
234  data_reader->set_initial_position();
235 
237  int num_iterations_per_epoch = ceil((float) data_reader->get_num_data() / (float) max_mini_batch_size);
238  int last_mini_batch_size = data_reader->get_num_data() % max_mini_batch_size;
239  if(last_mini_batch_size == 0) {
240  last_mini_batch_size = max_mini_batch_size;
241  }
242  data_reader->set_num_iterations_per_epoch(num_iterations_per_epoch);
243  data_reader->set_last_mini_batch_size(last_mini_batch_size);
244  data_reader->set_stride_to_last_mini_batch(data_reader->get_stride_to_next_mini_batch());
245 
246  data_reader->set_global_mini_batch_size(max_mini_batch_size);
247  data_reader->set_global_last_mini_batch_size(last_mini_batch_size);
248  return;
249 }
