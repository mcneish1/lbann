71 int lbann::DataReader_MNIST::fetch_data(Mat& X)
72 {
73  if(!DataReader::position_valid()) {
74  stringstream err;
75  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader load error: !position_valid";
76  throw lbann_exception(err.str());
77  }
78 
79  int pixelcount = m_image_width * m_image_height;
80  int current_batch_size = getBatchSize();
81 
82  int n = 0;
83  std::vector<float> pixels(pixelcount);
84  for (n = CurrentPos; n < CurrentPos + current_batch_size; n++) {
85  if (n >= (int)ShuffledIndices.size())
86  break;
87 
88  int k = n - CurrentPos;
89  int index = ShuffledIndices[n];
90  unsigned char* data = m_image_data[index];
91  unsigned char* pixels = &data[1];
92  unsigned char label = data[0];
93 
94  for (int p = 0; p < pixelcount; p++) {
95  X.Set(p, k, pixels[p]);
96  }
97 
98  auto pixel_col = X(IR(0, X.Height()), IR(k, k + 1));
99  augment(pixel_col, m_image_height, m_image_width, 1);
100  normalize(pixel_col, 1);
101  }
102 
103  return (n - CurrentPos);
104 }

131 bool lbann::DataReader_MNIST::load(string FileDir, string ImageFile, string LabelFile)
132 {
133  this->free();
134 
135  // set filepath
136  string imagepath = FileDir + __DIR_DELIMITER + ImageFile;
137  string labelpath = FileDir + __DIR_DELIMITER + LabelFile;
138 
139  // read labels
140  FILE* fplbl = fopen(labelpath.c_str(), "rb");
141  if (!fplbl) {
142  stringstream err;
143  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader: failed to open file: " << labelpath;
144  throw lbann_exception(err.str());
145  }
146 
147  int magicnum1, numitems1;
148  fread(&magicnum1, 4, 1, fplbl);
149  fread(&numitems1, 4, 1, fplbl);
150  __swapEndianInt((unsigned int&)magicnum1);
151  __swapEndianInt((unsigned int&)numitems1);
152 
153  // read images
154  FILE* fpimg = fopen(imagepath.c_str(), "rb");
155  if (!fpimg) {
156  stringstream err;
157  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader: failed to open file: " << imagepath;
158  throw lbann_exception(err.str());
159  }
160 
161  int magicnum2, numitems2, imgwidth, imgheight;
162  fread(&magicnum2, 4, 1, fpimg);
163  fread(&numitems2, 4, 1, fpimg);
164  fread(&imgwidth, 4, 1, fpimg);
165  fread(&imgheight, 4, 1, fpimg);
166  __swapEndianInt((unsigned int&)magicnum2);
167  __swapEndianInt((unsigned int&)numitems2);
168  __swapEndianInt((unsigned int&)imgwidth);
169  __swapEndianInt((unsigned int&)imgheight);
170 
171  if (numitems1 != numitems2) {
172  fclose(fplbl);
173  fclose(fpimg);
174  stringstream err;
175  err << __FILE__<<" "<<__LINE__<< " :: MNIST data reader: numitems1 != numitems2";
176  throw lbann_exception(err.str());
177  }
178 
179  // set to array
180  for (int n = 0; n < numitems1; n++) {
181  unsigned char* data = new unsigned char[1 + imgwidth * imgheight];
182  fread(&data[0], 1, 1, fplbl);
183  fread(&data[1], imgwidth * imgheight, 1, fpimg);
184 
185  m_image_data.push_back(data);
186  }
187  fclose(fpimg);
188  fclose(fplbl);
189 
190  // reset indices
191  ShuffledIndices.clear();
192  ShuffledIndices.resize(m_image_data.size());
193  for (size_t n = 0; n < ShuffledIndices.size(); n++) {
194  ShuffledIndices[n] = n;
195  }
196 
197  return true;
198 }

269 void lbann::DataReader_MNIST::clone_image_data(const DataReader_MNIST& source)
270 {
271  // m_image_data has pointers, so we need to deep copy them
272  for (size_t n = 0; n < source.m_image_data.size(); n++) {
273  unsigned char* data = new unsigned char[1 + m_image_width * m_image_height];
274  unsigned char* src_data = source.m_image_data[n];
275 
276  for (size_t i = 0; i < 1 + m_image_width * m_image_height; i++) {
277  data[i] = src_data[i];
278  }
279  m_image_data.push_back(data);
280  }
281  return;
282 }
