89  if(src != nullptr && dst != nullptr) { \

150 AbsDistMat& optimizer::get_gradient() {
151 
152  // Check if gradient is initialized
153  if (!is_initialized()) {
154  std::stringstream err;
155  err << __FILE__ << " " << __LINE__ << " :: "
156  << "attempted to access gradients before they are set up";
157  throw lbann_exception(err.str());
158  }
159 
160  // Accumulate CPU allreduce staging matrix
161  if (m_cpu_staging_is_nonzero) {
162  m_comm->allreduce(*m_staging,
163  m_staging->RedundantComm());
164  add_to_gradient(*m_staging);
165  El::Zero(*m_staging);
166  m_cpu_staging_is_nonzero = false;
167  }
168 
169 #if __LIB_CUDNN
170 
171  // Get matrix dimensions
172  const int height = m_weights->get_height();
173  const int width = m_weights->get_width();
174 
175  // Accumulate GPU allreduce staging matrix
176  if (m_gpu_staging_is_nonzero) {
177  m_cudnn->global_allreduce_on_gpus(m_staging_d,
178  height,
179  width,
180  m_gradient->RedundantComm());
181  add_to_gradient_gpu(m_staging_d);
182  m_cudnn->clear_on_gpus(m_staging_d, height, width);
183  m_gpu_staging_is_nonzero = false;
184  }
185 
186  // Accumulate gradient in CPU
187  if (m_gpu_gradient_is_nonzero) {
188  m_cudnn->copy_from_gpu(0,
189  m_staging->Matrix(),
190  m_gradient_d[0]);
191  m_cudnn->clear_on_gpus(m_gradient_d, height, width);
192  m_cudnn->synchronize();
193  add_to_gradient(*m_staging);
194  El::Zero(*m_staging);
195  m_gpu_gradient_is_nonzero = false;
196  }
197 
198 #endif // __LIB_CUDNN
199 
200  // Return full gradient
201  return *m_gradient;
202 
203 }

383 void optimizer::setup(weights& w) {
384  if (is_initialized()) {
385  std::stringstream err;
386  err << __FILE__ << " " << __LINE__ << " :: "
387  << "attempted to setup an optimizer that is already set up";
388  throw lbann_exception(err.str());

390  set_weights(w);
391 
392  // Initialize matrices
393  const int height = m_weights->get_height();
394  const int width = m_weights->get_width();
395  const AbsDistMat& values = m_weights->get_values();
396  m_gradient = values.Construct(values.Grid(), values.Root());
397  m_staging = values.Construct(values.Grid(), values.Root());
398  El::Zeros(*m_gradient, height, width);

410  // Initialize with zero gradient
411  clear_gradient();

415 void optimizer::step() {
416  if (!is_initialized()) {
417  std::stringstream err;
418  err << __FILE__ << " " << __LINE__ << " :: "
419  << "optimizer must be set up before performing optimization step";
420  throw lbann_exception(err.str());
421  }
422 
423  double step_start = get_time();
424  // Apply optimization step
425  if (m_cudnn != nullptr) {
426  #if __LIB_CUDNN
427  std::vector<DataType*> values_d = m_weights->m_values_d;
428  std::vector<DataType*> gradient_d = get_gradient_gpu();
429  step_compute_gpu(values_d, gradient_d);
430  #endif // __LIB_CUDNN
431  } else {
432  m_weights->get_values(); // Move data to CPU
433  AbsDistMat& values = *m_weights->m_values;
434  const AbsDistMat& gradient = get_gradient();
435  step_compute(values, gradient);
436  }
437 
438  // Clear gradients
439  clear_gradient();
440 
441  m_step_time += get_time() - step_start;
442 
443 }
