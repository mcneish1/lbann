57  double compute_metric(AbsDistMat& predictions_v, AbsDistMat& groundtruth_v) override {
58 
59  double corr = 0.0;
60 
61  // Compute mean and stdev
62  DataType pred_mean = 0;
63  DataType pred_std = 0;
64  DataType true_mean = 0;
65  DataType true_std = 0;
66  DataType corr_mean = 0;
67  DataType corr_std = 0;
68 
69  entrywise_mean_and_stdev(predictions_v,pred_mean, pred_std);
70  entrywise_mean_and_stdev(groundtruth_v,true_mean, true_std);
71 
72  //Compute covariance
73  auto sub_pred_mean = [&](const DataType& z) {return z - pred_mean;};
74  auto sub_true_mean = [&](const DataType& z) {return z - true_mean;};
75 
76  AbsDistMat* fsp = predictions_v.Construct(predictions_v.Grid(),
77  predictions_v.Root());
78  AbsDistMat* fst = groundtruth_v.Construct(groundtruth_v.Grid(),
79  groundtruth_v.Root());
80 
81  Copy(predictions_v,*fsp);
82  Copy(groundtruth_v,*fst);
83 
84  El::EntrywiseMap(*fsp, El::MakeFunction(sub_pred_mean));
85  El::EntrywiseMap(*fst, El::MakeFunction(sub_true_mean));
86 
87  AbsDistMat* covariance_mat = groundtruth_v.Construct(groundtruth_v.Grid(),
88  groundtruth_v.Root());
89 
90  El::Hadamard(*fsp,*fst, *covariance_mat);
91 
92  entrywise_mean_and_stdev(*covariance_mat, corr_mean, corr_std);
93  //Compute correlation
94  corr = corr_mean/(pred_std*true_std);
95 
96  return corr;
97 
98  }
