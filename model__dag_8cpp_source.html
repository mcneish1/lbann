78 void dag_model::topologically_sort_layers() {
79  /* Note: This sort must be deterministic so that it produces
80  * identical orderings when applied on different MPI processes.
81  */
82 
83  // Initialize data structures for topological sort
84  std::stack<const Layer*> sorted_stack;
85  std::stack<const Layer*> search_stack;
86  std::unordered_map<const Layer*,bool> is_sorted;
87  std::unordered_map<const Layer*,bool> is_visited;
88  for (const Layer* layer : m_layers) {
89  is_sorted[layer] = false;
90  is_visited[layer] = false;
91  search_stack.push(layer);
92  }
93 
94  // Perform depth-first searches until DAG has been traversed
95  while(!search_stack.empty()) {
96  const Layer* layer = search_stack.top();
97  search_stack.pop();
98  if (!is_sorted[layer]) {
99  if (is_visited[layer]) {
100  // Move search layer to sorted stack if we have visited already
101  sorted_stack.push(layer);
102  is_sorted[layer] = true;
103  } else {
104  // Visit search layer by adding children to search stack
105  search_stack.push(layer);
106  is_visited[layer] = true;
107  for (const Layer* child_layer : layer->get_child_layers()) {
108  if (is_visited[child_layer] && !is_sorted[child_layer]) {
109  throw lbann_exception("model_dag: detected a cycle in network graph");
110  }
111  search_stack.push(child_layer);
112  }
113  }
114  }
115  }
116 
117  // Record topologically sorted ordering
118  m_layers.clear();
119  while (!sorted_stack.empty()) {
120  m_layers.push_back(const_cast<Layer*>(sorted_stack.top()));
121  sorted_stack.pop();
122  }
123 
124 }
