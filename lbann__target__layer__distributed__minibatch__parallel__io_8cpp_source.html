87 void lbann::target_layer_distributed_minibatch_parallel_io::fp_linearity() {
88  int num_samples_in_batch = fetch_to_local_matrix(Y_local);
89  if(is_current_root()) {
91  target_layer::update_num_samples_processed(num_samples_in_batch);
92  }
93 
94  int64_t curr_mini_batch_size = neural_network_model->get_current_mini_batch_size();
95  if(is_current_root() && num_samples_in_batch != curr_mini_batch_size) {
96  throw lbann_exception("lbann_target_layer_distributed_minibatch_parallel_io: number of labels does not match the current mini-batch size.");
97  }
99  distribute_from_local_matrix(Y_local, Ys);
100  Copy(Ys, *m_activations);
101 
103  DataType avg_error = neural_network_model->obj_fn->compute_obj_fn(*m_prev_activations_v, *m_activations_v);
104  neural_network_model->obj_fn->record_obj_fn(m_execution_mode, avg_error);
105 
106  for (auto&& m : neural_network_model->metrics) {
107  double num_errors = (int) m->compute_metric(*m_prev_activations_v, *m_activations_v);
108  m->record_error(num_errors, curr_mini_batch_size);
109  }
110 
111  return;
112 }
