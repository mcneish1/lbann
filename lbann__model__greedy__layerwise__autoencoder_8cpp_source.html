67 bool lbann::greedy_layerwise_autoencoder::load_from_checkpoint_shared(lbann::persist& p)
68 {
69  // have rank 0 read whether we have a mirror layer inserted
70  struct lbann_model_greedy_layerwise_autoencoder_header header;
71  if (p.m_rank == 0) {
72  p.read_uint32(persist_type::train, "gla_phase_index", &header.phase_index);
73  p.read_uint32(persist_type::train, "gla_have_mirror", &header.have_mirror);
74  }
75 
76  // TODO: this assumes homogeneous processors
77  // broadcast state from rank 0
78  MPI_Bcast(&header, sizeof(header), MPI_BYTE, 0, MPI_COMM_WORLD);
79 
80  // insert the mirror layer if needed
81  uint32_t phase_index = header.phase_index;
82  uint32_t have_mirror = header.have_mirror;
83  if (have_mirror) {
84  // note that this calls setup on the layers,
85  // and setup reinitializes a bunch of values like data reader positions
86  // and optimization layer cache values that we'll overwrite
87  // in load_from_checkpoint_shared below
88  insert_mirror(phase_index);
89  }
90 
91  // read parameters from base class first
92  sequential_model::load_from_checkpoint_shared(p);
93 
94  return true;
95 }

104 void lbann::greedy_layerwise_autoencoder::insert_mirror(uint32_t layer_index)
105 {
106  // compute layer index for mirrror
107  size_t mirror_index = layer_index + 2;
108 
109  // build mirror layer
110  Layer* original_layer = m_layers[layer_index];
111  Optimizer *optimizer = optimizer_fac->create_optimizer();
112  target_layer_unsupervised* mirror_layer = new target_layer_unsupervised(mirror_index, comm, optimizer, m_mini_batch_size, original_layer);
113 
114  // insert mirror layer into model
115  insert(mirror_index, mirror_layer);
116 
117  //call base model set up at each phase to reindex and set appropriate matrices, fp and bp input
118  //assume that necessary layer parameters are set e.g., NumNeurons when layers were constructed
119  setup(layer_index, mirror_index+1); //set up all active layers
120 
121  // set flag to indicate that we have a mirror layer inserted
122  m_have_mirror = 1;
123 }

147 void lbann::greedy_layerwise_autoencoder::train(int num_epochs, int evaluation_frequency)
148 {
149  size_t num_phases = m_layers.size()-1;
150  // get to training, layer by layer
151  while(m_current_phase < num_phases){
152  //m_current_phase = phase_index;
153  m_phase_end = m_current_phase+2;
154  Layer* original_layer = m_layers[m_current_phase];
155  Optimizer *optimizer = optimizer_fac->create_optimizer();
156  target_layer_unsupervised* mirror_layer = new target_layer_unsupervised(m_phase_end, comm, optimizer, m_mini_batch_size,original_layer);
157  Layer* tmp;
158  //if not at the last layer/phase, swap otherwise insert new
159  if(m_current_phase < num_phases-1) tmp = swap(m_phase_end,mirror_layer);
160  else insert(m_phase_end,mirror_layer);
161  //call base model set up at each phase to reindex and set appropriate matrices, fp and bp input
162  //assume that necessary layer parameters are set e.g., NumNeurons when layers were constructed
163  setup(m_phase_end,m_phase_end+1); //set up just the added (new) layers
164  train_phase(num_epochs,evaluation_frequency);
165 
166  if (comm->am_world_master()) {
167  //end of phase cbs e.g., save a number of image to file
168  do_phase_end_cbs();
169  }
170  m_reconstruction_layers.insert(m_reconstruction_layers.begin(),mirror_layer);
171  //swap back
172  if(m_current_phase < num_phases-1) swap(m_phase_end,tmp);
173 
174  // move on to the next phase
175  m_current_phase++;
176  }
177  //evaluate and save all layers for i.e., (1) global cost (2) image reconstruction
178  evaluate(execution_mode::testing);
179 
180 }

215 void lbann::greedy_layerwise_autoencoder::train_phase(int num_epochs, int evaluation_frequency)
216 {
217  do_train_begin_cbs();
218 
219  // Epoch main loop
220  while (get_cur_epoch() < num_epochs) {
221  // Check if training has been terminated
222  if (get_terminate_training()) break;
223 
224  // due to restart, may not always be at start of epoch
225  // use mini batch index in data reader to signify start of epoch
226  if (at_epoch_start()) {
227  ++m_current_epoch;
228  do_epoch_begin_cbs(); // needed for selected callback e.g., dump matrices
229  }
230 
231  //Overide default print callback
232  if (comm->am_world_master()) {
233  std::cout << "-----------------------------------------------------------" << std::endl;
234  std::cout << "Phase [" << m_current_phase << "] Epoch [" << m_current_epoch << "]" << std::endl;
235  std::cout << "-----------------------------------------------------------" << std::endl;
236  }
237 
239  m_execution_mode = execution_mode::training;
240  for (size_t l =0; l < m_layers.size(); l++) {
241  m_layers[l]->m_execution_mode = execution_mode::training;
242  }
243 
244  // Train on mini-batches until data set is traversed
245  // Note: The data reader shuffles the data after each epoch
246  for (auto&& m : metrics) { m->reset_metric(); }
247  bool finished_epoch;
248  do {
249  finished_epoch = train_mini_batch();
250 
251  // save a checkpoint if needed
252  if (need_checkpoint()) {
253  checkpointShared();
254  }
255  } while(!finished_epoch);
256 
257 
258  //print training reconstruction cost
259  if (comm->am_world_master()) std::cout << "Layer-wise training ";
260  m_layers[m_phase_end]->epoch_print();
261 
262 
263  do_epoch_end_cbs(); //needed for selected callback e.g., dump matrices
264 
265  for (Layer* layer : m_layers) {
266  layer->epoch_reset();
267  } // train epoch end, this reset cost
268 
269  evaluate_phase(execution_mode::validation);
270 
271  //print validation reconstruction cost
272  if (comm->am_world_master()) std::cout << "Layer-wise validation ";
273  m_layers[m_phase_end]->epoch_print();
274 
275  //Reset cost again
276  for (Layer* layer : m_layers) {
277  layer->epoch_reset();
278  } // train epoch
279 
280  // Reset execution mode back to training
281  m_execution_mode = execution_mode::training;
282  for (Layer* layer : m_layers) {
283  layer->m_execution_mode = execution_mode::training;
284  }
285 
286  // save checkpoint after epoch
287  if (need_checkpoint()) {
288  checkpointShared();
289  }
290  }
291 
292  do_train_end_cbs();
293  m_current_epoch = 0; //reset epoch counter
294 }

296 bool lbann::greedy_layerwise_autoencoder::train_mini_batch()
297 {
298  do_batch_begin_cbs();
299 
300  // Forward propagation
301  do_model_forward_prop_begin_cbs();
302  //@todo; optimize this? change start index from 0 to phase_index
303  for (size_t l = 0; l <= m_phase_end; ++l) {
304  do_layer_forward_prop_begin_cbs(m_layers[l]);
305  m_layers[l]->forwardProp();
306  do_layer_forward_prop_end_cbs(m_layers[l]);
307  }
308  do_model_forward_prop_end_cbs();
309 
310  ++m_current_step;
311 
312  // Backward propagation
313  do_model_backward_prop_begin_cbs();
314  //@todo; optimize to backprop up to phase_index and not 0
315  for (size_t l = m_phase_end+1; l-- > 0;) {
316  do_layer_backward_prop_begin_cbs(m_layers[l]);
317  m_layers[l]->backProp();
318  do_layer_backward_prop_end_cbs(m_layers[l]);
319  }
320  do_model_backward_prop_end_cbs();
321 
324  for (size_t l = m_phase_end; l > m_current_phase; --l) {
325  m_layers[l]->update();
326  }
327  const bool data_set_processed = m_layers[0]->update();
328 
329  do_batch_end_cbs();
330  return data_set_processed;
331 }
