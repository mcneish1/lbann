53 void lbann::target_layer_unsupervised::setup(int num_prev_neurons) {
54  target_layer::setup(num_prev_neurons);
55  Layer::setup(num_prev_neurons);
56  if(optimizer != NULL) {
57  optimizer->setup(num_prev_neurons, NumNeurons);
58  }
59  // Initialize weight-bias matrix
60  Zeros(*m_weights, NumNeurons, num_prev_neurons);
61 
62  // Initialize weights
63  DistMat weights;
64  View(weights, *m_weights, IR(0,NumNeurons), IR(0,num_prev_neurons));
65  switch(m_weight_initialization) {
66  case weight_initialization::uniform:
67  uniform_fill(weights, weights.Height(), weights.Width(),
68  DataType(0), DataType(1));
69  break;
70  case weight_initialization::normal:
71  gaussian_fill(weights, weights.Height(), weights.Width(),
72  DataType(0), DataType(1));
73  break;
74  case weight_initialization::glorot_normal: {
75  const DataType var = 2.0 / (num_prev_neurons + NumNeurons);
76  gaussian_fill(weights, weights.Height(), weights.Width(),
77  DataType(0), sqrt(var));
78  break;
79  }
80  case weight_initialization::glorot_uniform: {
81  const DataType var = 2.0 / (num_prev_neurons + NumNeurons);
82  uniform_fill(weights, weights.Height(), weights.Width(),
83  DataType(0), sqrt(3*var));
84  break;
85  }
86  case weight_initialization::he_normal: {
87  const DataType var = 1.0 / num_prev_neurons;
88  gaussian_fill(weights, weights.Height(), weights.Width(),
89  DataType(0), sqrt(var));
90  break;
91  }
92  case weight_initialization::he_uniform: {
93  const DataType var = 1.0 / num_prev_neurons;
94  uniform_fill(weights, weights.Height(), weights.Width(),
95  DataType(0), sqrt(3*var));
96  break;
97  }
98  case weight_initialization::zero: // Zero initialization is default
99  default:
100  Zero(weights);
101  break;
102  }
103 
104  // Initialize other matrices
105  Zeros(*m_error_signal, num_prev_neurons, m_mini_batch_size); // m_error_signal holds the product of m_weights^T * m_prev_error_signal
106  Zeros(*m_activations, NumNeurons, m_mini_batch_size); //clear up m_activations before copying fp_input to it
107  Zeros(*m_weights_gradient, NumNeurons,num_prev_neurons); //clear up before filling with new results
108  Zeros(*m_prev_error_signal, NumNeurons, m_mini_batch_size); //clear up before filling with new results
109  Zeros(*m_prev_activations, num_prev_neurons, m_mini_batch_size);
110 
111 }

113 void lbann::target_layer_unsupervised::fp_linearity()
114 {
115  //m_activations is linear transformation of m_weights * m_prev_activations^T
116  Gemm(NORMAL, NORMAL, (DataType) 1., *m_weights, *m_prev_activations_v, (DataType) 0.0, *m_activations_v);
117 
118  int64_t curr_mini_batch_size = neural_network_model->get_current_mini_batch_size();
119  DistMatrixReadProxy<DataType,DataType,MC,MR> DsNextProxy(*m_original_layer->m_activations);
120  DistMat& DsNext = DsNextProxy.Get();
121  DistMat DsNext_v;
122  View(DsNext_v, DsNext, IR(0, DsNext.Height()), IR(0, curr_mini_batch_size));
123  //DsNext is proxy of original layer
124  // Compute cost will be sum of squared error of fp_input (linearly transformed to m_activations)
125  // and original layer fp_input/original input (DsNext)
126  DataType avg_error = neural_network_model->obj_fn->compute_obj_fn(*m_activations_v, DsNext_v);
127  aggregate_cost += avg_error;
128  num_forwardprop_steps++;
129 }

131 void lbann::target_layer_unsupervised::bp_linearity()
132 {
133 
135 
136  DistMatrixReadProxy<DataType,DataType,MC,MR> DsNextProxy(*m_original_layer->m_activations);
137  DistMat& DsNext = DsNextProxy.Get();
138  // delta = (activation - y)
139  // delta_w = delta * activation_prev^T
140  //@todo: Optimize (may be we dont need this double copy)
141  //Activation in this layer is same as linear transformation of its input, no nonlinearity
142  //@todo: Optimize (check that may be we dont need this double copy)
143 
144  int64_t curr_mini_batch_size = neural_network_model->get_current_mini_batch_size();
145  DistMat DsNext_v;
146  View(DsNext_v, DsNext, IR(0, DsNext.Height()), IR(0, curr_mini_batch_size));
147  Copy(*m_activations_v, *m_prev_error_signal_v);
148  Axpy(-1., DsNext_v, *m_prev_error_signal_v); // Per-neuron error
149  // Compute the partial delta update for the next lower layer
150  Gemm(TRANSPOSE, NORMAL, (DataType) 1., *m_weights, *m_prev_error_signal_v, (DataType) 0., *m_error_signal_v);
151 
152  // Compute update for activation weights
153  Gemm(NORMAL, TRANSPOSE, (DataType) 1.0/get_effective_minibatch_size(), *m_prev_error_signal_v,
154  *m_prev_activations_v, (DataType) 0., *m_weights_gradient);
155 }
