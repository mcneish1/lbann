42 void data_store_imagenet::setup() {
43  double tm1 = get_time();

46  bool run_tests = false;

48  run_tests = true;
49  }
50 
51  if (m_rank == 0) {
52  std::cout << "starting data_store_imagenet::setup() for data reader with role: " << m_reader->get_role() << std::endl;
53  }
54 
55  if (! m_in_memory) {
56  std::stringstream err;
57  err << __FILE__ << " " << __LINE__ << " :: "
58  << "not yet implemented";
59  throw lbann_exception(err.str());
60  m_buffers.resize( omp_get_max_threads() );
61  }
62 
63  else {
64  //sanity check
65  imagenet_reader *reader = dynamic_cast<imagenet_reader*>(m_reader);
66  if (reader == nullptr) {
67  std::stringstream err;
68  err << __FILE__ << " " << __LINE__ << " :: "
69  << "dynamic_cast<imagenet_reader*>(m_reader) failed";
70  throw lbann_exception(err.str());
71  }
72 
73  // fill in global index -> owning processor
74  for (size_t j=0; j<m_num_global_indices; j++) {
75  int owner = j % m_num_readers;
76  m_owner_mapping[(*m_shuffled_indices)[j]] = owner;
77  }
78 
79  m_my_data.resize(m_my_minibatch_indices.size());
80 
81  get_file_sizes();
82  allocate_memory();
83  read_files();

85  exchange_data();
86  if (run_tests) {
87  test_file_sizes();
88  test_data();
89  }
90 
91  MPI_Barrier(m_comm->get_model_comm().comm);
92  double tm2 = get_time();
93  if (m_rank == 0) {

153 void data_store_imagenet::get_file_sizes() {
154  std::vector<int> num_images(m_num_readers, 0);
155  for (size_t j=0; j<m_num_global_indices; j++) {
156  num_images[j % m_num_readers] += 1;
157  }
158  //num_images[j] contains the number of image files owned by P_j
159 
160  //get the number of bytes for files owned by this processor;
161  //my_file_sizes[j] is a global index (wrt m_reader->get_image_list()),
162  //and my_file_sizes[j+1] is the corresponding file size
163  imagenet_reader *reader = dynamic_cast<imagenet_reader*>(m_reader);
164  const std::vector<std::pair<std::string, int> > & image_list = reader->get_image_list();
165  int my_num_images = num_images[m_rank];
166  std::vector<Triple> my_file_sizes(my_num_images);
167  std::string file_dir = m_reader->get_file_dir();
168  size_t cur_offset = 0;
169  for (size_t j=0; j<m_my_datastore_indices.size(); j++) {
170  if (j >= image_list.size()) throw lbann_exception("err 2\n");
171  size_t index = m_my_datastore_indices[j];
172  my_file_sizes[j].global_index = index;
173  my_file_sizes[j].num_bytes = get_file_size(file_dir, image_list[index].first);
174  my_file_sizes[j].offset = cur_offset;
175  cur_offset += my_file_sizes[j].num_bytes;
176  }
177 
178  //exchange files sizes
179  std::vector<Triple> global_file_sizes(m_num_global_indices);
180  std::vector<int> disp(m_num_readers);
181  disp[0] = 0;
182  for (int h=1; h<(int)m_num_readers; h++) {
183  disp[h] = disp[h-1] + num_images[h-1]*sizeof(Triple);
184  }
185 
186  for (size_t j=0; j<num_images.size(); j++) {
187  num_images[j] *= sizeof(Triple);
188  }
189 
190  //m_comm->model_gatherv(&my_file_sizes[0], my_file_sizes.size(),
191  // &global_file_sizes[0], &num_images[0], &disp[0]);
192  MPI_Allgatherv(&my_file_sizes[0], my_file_sizes.size()*sizeof(Triple), MPI_BYTE,
193  &global_file_sizes[0], &num_images[0], &disp[0], MPI_BYTE, m_comm->get_model_comm().comm);
194 
195  for (auto t : global_file_sizes) {
196  m_file_sizes[t.global_index] = t.num_bytes;
197  m_offsets[t.global_index] = t.offset;
198  }
199 }

239 void data_store_imagenet::exchange_data() {
240  double tm1 = get_time();
241  MPI_Win_fence(MPI_MODE_NOSTORE|MPI_MODE_NOPUT, m_win);
242  m_my_data_hash.clear();
243  for (size_t j = 0; j<m_my_minibatch_indices.size(); j++) {
244  int idx = (*m_shuffled_indices)[m_my_minibatch_indices[j]];
245  int offset = m_offsets[idx];
246  int file_len = m_file_sizes[idx];
247  if (file_len <= 0) throw lbann_exception(std::string{} + "ERROR A " + std::to_string(file_len));
248  if (file_len > 100000000) throw lbann_exception(std::string{} + "ERROR B " + std::to_string(file_len));
249  if (j >= m_my_data.size()) {
250  throw lbann_exception(std::string{} + "ERROR C; j: " + std::to_string(j) + " m_my_data.size: " + std::to_string(m_my_data.size()));
251  }
252  m_my_data[j].resize(file_len);
253  m_my_data_hash[idx] = j;
254  int owner = m_owner_mapping[idx];
255  MPI_Get(&m_my_data[j][0], file_len, MPI_BYTE,
256  owner, offset, file_len, MPI_BYTE, m_win);
257  }
258  MPI_Win_fence(MPI_MODE_NOSTORE|MPI_MODE_NOPUT, m_win);
259  double tm2 = get_time();
260  if (m_rank == 0) {
261  std::cout << "data_store_imagenet::exchange_data() time: " << tm2 - tm1 << std::endl;
262  }
263 }
