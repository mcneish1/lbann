504  if ((it != this->m_data_readers.end()) && it->second) {
505  (it->second)->saveToCheckpointShared(p, "data_reader_validation");
506  }
507 
508  it = this->m_data_readers.find(execution_mode::testing);
509  if ((it != this->m_data_readers.end()) && it->second) {
510  (it->second)->saveToCheckpointShared(p, "data_reader_testing");
511  }
512 
513  // save our own state
514  // rank 0 writes the file
515  if (p.get_rank() == 0) {
516  p.write_uint64(persist_type::train, "reader_train_processed",
517  (uint64_t) m_training_dataset.get_num_samples_processed());
518  p.write_uint64(persist_type::train, "reader_train_total",
519  (uint64_t) m_training_dataset.get_total_samples());
520 
521  p.write_uint64(persist_type::train, "reader_test_processed",
522  (uint64_t) m_testing_dataset.get_num_samples_processed());
523  p.write_uint64(persist_type::train, "reader_test_total",

547  // save state of data readers from input layer
548  data_reader_map_t::const_iterator it;
549 
550  it = this->m_data_readers.find(execution_mode::training);
551  if ((it != this->m_data_readers.end()) && it->second) {
552  (it->second)->loadFromCheckpointShared(p, "data_reader_training");
553  }
554 
555  it = this->m_data_readers.find(execution_mode::validation);
556  if ((it != this->m_data_readers.end()) && it->second) {
557  (it->second)->loadFromCheckpointShared(p, "data_reader_validation");
558  }
559 
560  it = this->m_data_readers.find(execution_mode::testing);
561  if ((it != this->m_data_readers.end()) && it->second) {
562  (it->second)->loadFromCheckpointShared(p, "data_reader_testing");
563  }
564 
565  // save our own state
566  // rank 0 reads the file
567  dataset_header header;
568  if (p.get_rank() == 0) {
569  p.read_uint64(persist_type::train, "reader_train_processed", &header.train_proc);
570  p.read_uint64(persist_type::train, "reader_train_total", &header.train_total);
571  p.read_uint64(persist_type::train, "reader_test_processed", &header.test_proc);
572  p.read_uint64(persist_type::train, "reader_test_total", &header.test_total);
573  p.read_uint64(persist_type::train, "reader_validate_processed", &header.validate_proc);
574  p.read_uint64(persist_type::train, "reader_validate_total", &header.validate_total);
575  }
576 
577  // TODO: assumes homogeneous hardware
578  // broadcast data from rank 0
579  MPI_Bcast(&header, sizeof(header), MPI_BYTE, 0, MPI_COMM_WORLD);
580 
581  // set our fields
582  m_training_dataset.num_samples_processed() = (long) header.train_proc;
583  m_training_dataset.total_samples() = (long) header.train_total;
584  m_testing_dataset.num_samples_processed() = (long) header.test_proc;
585  m_testing_dataset.total_samples() = (long) header.test_total;
586  m_validation_dataset.num_samples_processed() = (long) header.validate_proc;
587  m_validation_dataset.total_samples() = (long) header.validate_total;
588 
589  io_layer::loadFromCheckpointShared(p);
590 
591  return true;
592  }
